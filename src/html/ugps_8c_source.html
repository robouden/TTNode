73 bool ugps_ok_to_update() {
74 
75  // Don't update under any circumstances if the battery is low, or if
76  // we're in burn mode. This is because a) during test we are constantly
77  // tapping the unit to wake up the screen, and the GPS seek is an annoyance,
78  // and b) the GPS sucks a lot of power, and this is inappropriate when
79  // the battery is low.
80  switch (battery_status()) {
81  case BAT_BURN:
82  case BAT_TEST:
83  case BAT_EMERGENCY:
84  case BAT_WARNING:
85  case BAT_LOW:
86  return false;
87 
88  }
89 
90  return true;
91 
92 }

143 // One-time initialization of sensor
144 bool s_ugps_init(void *s, uint16_t param) {
145 
146  // Exit if we're updating when it's simply inappropriate to do so
147  // because of how little energy remains.
148  if (initialized_ever && !ugps_ok_to_update())
149  return false;
150 
151  // Defensively, exit if we're alraedy running
152  if (initialized)
153  return false;
154 
155  // Proceed
156  completed_iobufs_available = 0;
157  sentences_received = 0;
158  iobuf_filling = 0;
159  iobuf[0].linesize = 0;
160  initialized = initialized_ever = true;
161  seconds = 0;
162  shutdown = false;
163 
164  // Send update rate
165  DEBUG_PRINTF("s-gps initializing\n");
166  serial_send_string("$PMTK300,2000,0,0,0,0*18");
167 
168  // Tell it that we'd like updates on which antenna is being used
169  if (!displayed_antenna_status)
170  serial_send_string("$PGCMD,33,1*6C");
171 
172  return true;
173 }

180 // Drain the completed I/O buffer and point to the next
181 bool iobuf_pop(iobuf_t *piobuf) {
182  if (completed_iobufs_available <= 0)
183  return false;
184  *piobuf = iobuf[iobuf_completed];
185  completed_iobufs_available--;
186  if (++iobuf_completed >= IOBUFFERS)
187  iobuf_completed = 0;
188  return true;

191 // Process a data-received events
192 void gps_process_sentence(char *line, uint16_t linelen) {
193 
194  // Bump the number received
195  sentences_received++;
196 
197  // For extreme debugging of GPS abort/retry logic at the comm level
198 #if 0
199  reported_have_location = true;
200  reported_latitude = 1;
201  reported_longitude = 2;
202  reported_altitude = 3;
203  reported_have_full_location = true;
204  reported_have_improved_location = true;
205  trying_to_improve_location = false;
206  if (!reported_have_timedate) {
207  reported_time = 111111;
208  reported_date = 111117;
209  set_timestamp(reported_date, reported_time);
210  reported_have_timedate = true;
211  }
212  return;
213 #endif
214 
215  // Process the GPS sentence
216  if (debug(DBG_GPS_MAX))
217  DEBUG_PRINTF("%s%s%s%s %s\n", reported_have_location ? "l" : "-", reported_have_full_location ? "L" : "-", reported_have_improved_location ? "I" : "-", reported_have_timedate ? "T" : "-", line);
218 
219  // Process $GPGGA, which should give us lat/lon/alt
220  if (memcmp(line, "$GPGGA", 6) == 0) {
221  int j;
222  char *lat, *ns, *lon, *ew, *alt, *fix, *time;
223  bool haveLat, haveLon, haveNS, haveEW, haveAlt, haveFix, haveTime;
224  uint16_t commas;
225 
226  commas = 0;
227  haveLat = haveLon = haveNS = haveEW = haveAlt = haveFix = haveTime = false;
228  lat = ns = lon = ew = alt = time = fix = "";
229 
230  for (j = 0; j < linelen; j++)
231  if (line[j] == ',') {
232  line[j] = '\0';
233  commas++;
234  if (commas == 1) {
235  // Sitting at comma before Time
236  time = (char *) &line[j + 1];
237  }
238  if (commas == 2) {
239  haveTime = (time[0] != '\0');
240  // Sitting at comma before Lat
241  lat = (char *) &line[j + 1];
242  }
243  if (commas == 3) {
244  // Sitting at comma before NS
245  haveLat = (lat[0] != '\0');
246  ns = (char *) &line[j + 1];
247  }
248  if (commas == 4) {
249  // Sitting at comma before Lon
250  haveNS = (ns[0] != '\0');
251  lon = (char *) &line[j + 1];
252  }
253  if (commas == 5) {
254  // Sitting at comma before EW
255  haveLon = (lon[0] != '\0');
256  ew = (char *) &line[j + 1];
257  }
258  if (commas == 6) {
259  // Sitting at comma before Quality
260  haveEW = (ew[0] != '\0');
261  fix = (char *) &line[j + 1];
262  }
263  if (commas == 7) {
264  // Sitting at comma before NumSat
265  // 1 is valid GPS fix, 2 is valid DGPS fix
266  haveFix = (fix[0] == '1' || fix[0] == '2');
267  }
268  if (commas == 8) {
269  // Sitting at comma before Hdop
270  }
271  if (commas == 9) {
272  // Sitting at comma before Alt
273  alt = (char *) &line[j + 1];
274  }
275  if (commas == 10) {
276  // Sitting at comma before M unit
277  haveAlt = (alt[0] != '\0');
278  break;
279  }
280  }
281 
282  // If we've got what we need, process it and exit.
283  if (haveFix && haveLat && haveNS & haveLon && haveEW) {
284  float fLatitude = GpsEncodingToDegrees(lat, ns);
285  float fLongitude = GpsEncodingToDegrees(lon, ew);
286  if (fLatitude != 0 || fLongitude != 0) {
287  last_sampled_loc++;
288  reported_have_location = true;
289  reported_latitude = fLatitude;
290  reported_longitude = fLongitude;
291  if (haveAlt) {
292  reported_altitude = atof(alt);
293  reported_have_full_location = true;
294  reported_have_improved_location = true;
295  trying_to_improve_location = false;
296  }
297  if (!saved_lkg_this_session) {
298  STORAGE *f = storage();
299  f->lkg_gps_latitude = reported_latitude;
300  f->lkg_gps_longitude = reported_longitude;
301  f->lkg_gps_altitude = reported_altitude;
302  storage_save(false);
303  saved_lkg_this_session = true;
304  }
305  }
306 
307  }
308 
309  } // if GPGGA
310 
311  // Process $GPRMC, which should give us lat/lon and time
312  if (memcmp(line, "$GPRMC", 6) == 0) {
313  int j;
314  char *lat, *ns, *lon, *ew, *time, *date, *valid;
315  bool haveLat, haveLon, haveNS, haveEW, haveTime, haveDate, haveValid;
316  uint16_t commas;
317 
318  commas = 0;
319  haveLat = haveLon = haveNS = haveEW = haveTime = haveDate = haveValid = false;
320  lat = ns = lon = ew = date = time = valid = "";
321 
322  for (j = 0; j < linelen; j++)
323  if (line[j] == ',') {
324  line[j] = '\0';
325  commas++;
326  if (commas == 1) {
327  // Sitting at comma before Time
328  time = (char *) &line[j + 1];
329  }
330  if (commas == 2) {
331  haveTime = (time[0] != '\0');
332  // Sitting at comma before Validity
333  valid = (char *) &line[j + 1];
334  }
335  if (commas == 3) {
336  // Sitting at comma before Lat
337  haveValid = (valid[0] == 'A');
338  lat = (char *) &line[j + 1];
339  }
340  if (commas == 4) {
341  // Sitting at comma before NS
342  haveLat = (lat[0] != '\0');
343  ns = (char *) &line[j + 1];
344  }
345  if (commas == 5) {
346  // Sitting at comma before Lon
347  haveNS = (ns[0] != '\0');
348  lon = (char *) &line[j + 1];
349  }
350  if (commas == 6) {
351  // Sitting at comma before EW
352  haveLon = (lon[0] != '\0');
353  ew = (char *) &line[j + 1];
354  }
355  if (commas == 7) {
356  // Sitting at comma before Speed
357  haveEW = (ew[0] != '\0');
358  }
359  if (commas == 8) {
360  // Sitting at comma before True Course
361  }
362  if (commas == 9) {
363  // Sitting at comma before Date
364  date = (char *) &line[j + 1];
365  }
366  if (commas == 10) {
367  // Sitting at comma before Variation
368  haveDate = (date[0] != '\0');
369  break;
370  }
371  }
372 
373  // If we've got lat/lon, process it
374  if (haveValid && haveLat && haveNS & haveLon && haveEW) {
375  float fLatitude = GpsEncodingToDegrees(lat, ns);
376  float fLongitude = GpsEncodingToDegrees(lon, ew);
377  if (fLatitude != 0 || fLongitude != 0) {
378  last_sampled_loc++;
379  reported_have_location = true;
380  reported_latitude = fLatitude;
381  reported_longitude = fLongitude;
382  reported_have_improved_location = true;
383  trying_to_improve_location = false;
384  }
385  if (!saved_lkg_this_session) {
386  STORAGE *f = storage();
387  f->lkg_gps_latitude = reported_latitude;
388  f->lkg_gps_longitude = reported_longitude;
389  f->lkg_gps_altitude = 0;
390  storage_save(false);
391  saved_lkg_this_session = true;
392  }
393  }
394 
395  // If we've got what we need, process it and exit.
396  if (!reported_have_timedate && haveValid && haveTime && haveDate) {
397  reported_time = atol(time);
398  reported_date = atol(date);
399  // Do one final check to make sure that the date is valid.
400  // We do this because we've seen dates of 1980 being reported
401  // by GPS chips, and it's reasonable to bracket valid year values.
402  // If this source code lasts beyond this range, I'll be very happy
403  // if you relax this check because maybe chips will function
404  // properly by then :-)
405  float reported_year = reported_date % 100;
406  if (reported_year >= 17 && reported_year < 50) {
407  set_timestamp(reported_date, reported_time);
408  reported_have_timedate = true;
409  } else {
410  DEBUG_PRINTF("GPS: Invalid year: %.0f\n", reported_year);
411  }
412  }
413 
414  } // if GPRMC
415 
416  // Process $PGTOP, which tells us which antenna is being used
417  if (memcmp(line, "$PGTOP", 6) == 0) {
418  int j;
419  char *arg1, *arg2;
420  bool haveArg1, haveArg2;
421  uint16_t commas;
422 
423  commas = 0;
424  haveArg1 = haveArg2 = false;
425  arg1 = arg2 = "";
426 
427  for (j = 0; j < linelen; j++)
428  if (line[j] == ',' || line[j] == '*') {
429  line[j] = '\0';
430  commas++;
431  if (commas == 1) {
432  // Sitting at comma before Arg1
433  arg1 = (char *) &line[j + 1];
434  }
435  if (commas == 2) {
436  // Sitting at comma before Arg2
437  haveArg1 = (arg1[0] != '\0');
438  arg2 = (char *) &line[j + 1];
439  }
440  if (commas == 3) {
441  // Sitting at asterisk before checksum
442  haveArg2 = (arg2[0] != '\0');
443  break;
444  }
445  }
446 
447  // If we've got what we need, process it and exit.
448  UNUSED_VARIABLE(haveArg1);
449  if (haveArg2) {
450  if (!displayed_antenna_status) {
451  displayed_antenna_status = true;
452  switch (arg2[0]) {
453  case '1':
454  DEBUG_PRINTF("GPS antenna failure.\n");
455  break;
456  case '2':
457  DEBUG_PRINTF("GPS using internal antenna.\n");
458  break;
459  case '3':
460  DEBUG_PRINTF("GPS using external antenna.\n");
461  break;
462  }
463  }
464  }
465 
466  } // if $PGTOP
467 
468 }

479 // Bump to the next I/O buffer, dropping the line if we overflow I/O buffers
480 bool iobuf_push() {
481  bool dropped = false;
482 
483  if (completed_iobufs_available < IOBUFFERS) {
484  if (++iobuf_filling >= IOBUFFERS)
485  iobuf_filling = 0;
486  iobuf_reset();
487  completed_iobufs_available++;
488  app_sched_event_put(NULL, 0, line_event_handler);
489  } else {
490  iobuf_reset();
491  dropped = true;
492  }
493 
494  return !dropped;
495 }

497 // Process byte received from gps
498 void s_ugps_received_byte(uint8_t databyte) {
499 
500  // Discard results if we happen to get here if not yet initialized
501  if (!initialized)
502  return;
503 
504  // If we get here, we're expecting to receive an ASCII text command terminated in \r\n.
505  // If this is the CR of the CRLF sequence, skip it.
506  if (databyte == '\r')
507  return;
508 
509  // If this the final byte of the CRLF sequence, process it
510  if (databyte == '\n') {
511 
512  // If it's just a blank line (or the second char of \r\n), skip it
513  if (iobuf[iobuf_filling].linesize == 0)
514  return;
515 
516  // Null terminate the line because it's a string
517  iobuf[iobuf_filling].linebuf[iobuf[iobuf_filling].linesize] = '\0';
518 
519  // Enqueue it
520  iobuf_push();
521 
522  return;
523  }
524 
525  // If somehow a non-text character got here, substitute.
526  if (databyte < 0x20 || databyte > 0x7f)
527  databyte = '.';
528 
529  // Add the char to the line buffer
530  if (iobuf[iobuf_filling].linesize < (sizeof(iobuf[iobuf_filling].linebuf)-2))
531  iobuf[iobuf_filling].linebuf[iobuf[iobuf_filling].linesize++] = (char) databyte;
532 
533 }

540 // Group skip handler
541 bool g_ugps_skip(void *g) {
542 
543  // Skip if we've got a static value
544  if (!reported && comm_gps_get_value(NULL, NULL, NULL) == GPS_LOCATION_FULL)
545  return true;
546 
547  // Don't skip if we'd aborted and it's time for an hourly retry
548  if (sensor_op_mode() != OPMODE_TEST_BURN)
549  if (comm_gps_get_value(NULL, NULL, NULL) == GPS_LOCATION_ABORTED) {
550  if (!ShouldSuppress(&last_retry, GPS_RETRY_MINUTES*60)) {
551  reported = false;
552  skip = false;
553  return false;
554  } else {
555  if (debug(DBG_GPS_MAX))
556  DEBUG_PRINTF("Need to resample aborted GPS\n");
557  }
558  }
559 
560  // Don't let the GPS sleep if we're in mobile mode
561  if (sensor_op_mode() == OPMODE_MOBILE)
562  return false;
563 
564  // Skip if we've already reported
565  return(skip);
566 }

568 // Poller
569 void s_ugps_poll(void *s) {
570 
571  // Exit if we're not supposed to be here
572  if (!sensor_is_polling_valid(s))
573  return;
574 
575  // Keep track of active/inactive
576  if (sentences_received == sentences_received_last_poll)
577  gps_active = false;
578  else {
579  gps_active = true;
580  sentences_received_last_poll = sentences_received;
581  }
582 
583  // Keep track of how long we've been waiting for lock
584  seconds += GPS_POLL_SECONDS;
585  if (debug(DBG_GPS_MAX)) {
586  DEBUG_PRINTF("GPS poll %ds (%d)\n", seconds, comm_gps_get_value(NULL, NULL, NULL));
587  }
588 
589  // If we're in DFU mode, we don't need to hold things up waiting for GPS
590  if (storage()->dfu_status == DFU_PENDING && s_ugps_get_value(NULL, NULL, NULL) != GPS_NO_DATA) {
591  skip = true;
592  comm_gps_abort();
593  sensor_measurement_completed(s);
594  gpio_indicators_off();
595  DEBUG_PRINTF("GPS present but aborted because of DFU mode\n");
596  return;
597  }
598 
599  // If we're in burn mode and we've received at least some data, short circuit it
600  if (sensor_op_mode() == OPMODE_TEST_BURN && s_ugps_get_value(NULL, NULL, NULL) != GPS_NO_DATA) {
601  skip = true;
602  comm_gps_abort();
603  sensor_measurement_completed(s);
604  gpio_indicators_off();
605  DEBUG_PRINTF("GPS present but aborted because of burn mode\n");
606  return;
607  }
608 
609  // If we're in battery test mode, short circuit all this
610  if (battery_status() == BAT_TEST) {
611  skip = true;
612  comm_gps_abort();
613  sensor_measurement_completed(s);
614  gpio_indicators_off();
615  DEBUG_PRINTF("GPS aborted because of test mode\n");
616  return;
617  }
618 
619  // If we're in mobile mode, don't ever stop unless something needs the UART to transmit.
620  if (sensor_op_mode() == OPMODE_MOBILE && comm_would_be_buffered(false)) {
621  if (debug(DBG_SENSOR_MAX)) {
622  static uint32_t prev_last_sampled_loc = 0;
623  if (last_sampled_loc != prev_last_sampled_loc) {
624  prev_last_sampled_loc = last_sampled_loc;
625  DEBUG_PRINTF("gps %.3f %.3f %s\n", reported_latitude, reported_longitude, gps_active ? "^" : "-");
626  } else {
627  DEBUG_PRINTF("gps (%s%s%s%s) %s\n", reported_have_location ? "l" : "-", reported_have_full_location ? "L" : "-", reported_have_improved_location ? "I" : "-", reported_have_timedate ? "T" : "-", gps_active ? "^" : "-");
628  }
629  }
630  return;
631  }
632 
633  // Determine whether or not it's time to stop polling
634  if (reported_have_location && reported_have_timedate) {
635  uint32_t abort_seconds = GPS_ABORT_FIRST_MINUTES * 60;
636  if (reported_have_full_location && !reported_have_improved_location)
637  abort_seconds = GPS_ABORT_IMPROVE_MINUTES * 60;
638  if (seconds > abort_seconds || (reported_have_full_location && reported_have_improved_location && reported_have_timedate)) {
639  reported = true;
640  skip = true;
641  if (debug(DBG_GPS_MAX))
642  DEBUG_PRINTF("%.3f/%.3f/%.3f %lu:%lu\n", reported_latitude, reported_longitude, reported_altitude, reported_date, reported_time);
643  }
644  }
645 
646  // If we've been asked to shut down, terminate
647  if (shutdown) {
648  skip = true;
649  sensor_measurement_completed(s);
650  if (reported)
651  DEBUG_PRINTF("GPS: %.3f %.3f\n", reported_latitude, reported_longitude);
652  return;
653  }
654 
655  // If we've already got the full location, terminate the polling just to save battery life
656  if (!trying_to_improve_location) {
657  if ((comm_gps_get_value(NULL, NULL, NULL) == GPS_LOCATION_FULL)) {
658  skip = true;
659  sensor_measurement_completed(s);
660  if (reported)
661  DEBUG_PRINTF("GPS: %.3f %.3f\n", reported_latitude, reported_longitude);
662  return;
663  }
664  }
665 
666  // If the GPS hardware isn't even present, terminate the polling to save battery life.
667  uint32_t abort_seconds = GPS_ABORT_FIRST_MINUTES * 60;
668  if (reported_have_full_location && !reported_have_improved_location)
669  abort_seconds = GPS_ABORT_IMPROVE_MINUTES * 60;
670  if (sensor_op_mode() == OPMODE_TEST_BURN)
671  abort_seconds = 30;
672  if (seconds > abort_seconds) {
