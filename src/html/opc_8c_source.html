139 // Wait for SPI transaction to complete
140 bool opc_spi_transfer_succeeded() {
141 
142  while (true) {
143 
144  // Compute duration of transaction
145  uint32_t duration = get_seconds_since_boot() - spi_began;
146  if (spi_began == 0)
147  break;
148 
149  // Output a debug message
150  if (!ShouldSuppress(&spi_message, 1))
151  DEBUG_PRINTF("SPI command is taking too long!\n");
152 
153  // Exit if it has taken a very long time
154  if (duration > 3)
155  return false;
156 
157  }
158 
159  return true;
160

163 // Extract data as a struct from the raw OPC data, pointing at the 0xf3
164 bool unpack_opc_version(char *ver, uint16_t ver_len, uint8_t *spiData, uint16_t spiDataLen)
165 {
166  int i;
167 
168  // Bump to just after the 0xf3
169  spiData++;
170  spiDataLen--;
171 
172  // Compute the maximum length as the buffer length
173  for (i=0; ; i++) {
174  // If we've run out of target buffer length, stop
175  if (i >= ver_len-1)
176  break;
177  // If we've run out of spi data, stop.
178  if (i >= spiDataLen-1)
179  break;
180  // If we see a run of at least two '.' characters, we're done.
181  if (spiData[i] == 0x2e && spiData[i+1] == 0x2e)
182  break;
183  // Copy to output
184  ver[i] = spiData[i];
185  }
186 
187  // Null-terminate
188  ver[i] = '\0';
189 
190  // Success if we've copied anything at all
191  return (i != 0);
192 
193 }

195 // Return true if this is a valid PM value, noting that using SPI numbers can get corrupted
196 // Also, invalidate the number if it is
197 bool valid_float(char *what, float *val) {
198  float pm = *val;
199  bool fValid = true;
200  char *why = "";
201 
202  switch (fpclassify(pm)) {
203  case FP_ZERO:
204  case FP_NORMAL:
205  fValid = true;
206  break;
207  case FP_NAN:
208  fValid = false;
209  why = "FP_NAN";
210  break;
211  case FP_INFINITE:
212  fValid = false;
213  why = "FP_INFINITE";
214  break;
215  case FP_SUBNORMAL:
216  fValid = false;
217  why = "FP_SUBNORMAL";
218  break;
219  default:
220  fValid = false;
221  why = "????";
222  break;
223  }
224 
225  // ***
226  // Reset the FP hardware state because something, potentially during SPI
227  // initialization, has placed the FP hardware into a bad state. Note
228  // that if you remove this code you will take an FP trap (and crash the nRF)
229  // on the very first floating point operation after this point.
230  // Ray 2017-05-31
231  // ***
232  volatile float f = 0.0;
233  isless(f, 0.0);
234 
235  // If not valid, output it
236  if (!fValid && *what != '\0') {
237  uint8_t *p = (uint8_t *) val;
238  DEBUG_PRINTF("*** %s invalid (%s): ", what, why);
239  for (int i=0; i<sizeof(float); i++)
240  DEBUG_PRINTF("%02x", p[i]);
241  DEBUG_PRINTF("\n");
242  }
243 
244  // If not valid, zero it out so that we don't get an FP fault when calculating with it
245  if (!fValid)
246  *val = 0.0;
247 
248  // Valid
249  return fValid;
250 }

252 // Extract data as a struct from the raw OPC data, pointing at the 0xf3
253 bool unpack_opc_data(opc_t *opc, uint8_t *spiData, uint16_t spiDataLen)
254 {
255  uint8_t pos = 0;
256 
257  // If debugging, print but ignore ALL data
258  if (debug(DBG_SENSOR_SUPERMAX)) {
259  DEBUG_PRINTF("OPC raw data: ");
260  int i;
261  for (i=0; i<spiDataLen; i++) {
262  DEBUG_PRINTF("%02x", spiData[i]);
263  if ((i % 8) == 7)
264  DEBUG_PRINTF(" ");
265  }
266  DEBUG_PRINTF("\n");
267  }
268 
269  // Exit if we're skipping samples, waiting for settling, and remember when
270  // the sampling actually began.
271  if (num_samples_left_to_skip != 0) {
272  if (--num_samples_left_to_skip == 0) {
273  if (debug(DBG_SENSOR_SUPERMAX))
274  DEBUG_PRINTF("Sample should be skipped.\n");
275  count_began = get_seconds_since_boot();
276  }
277  return false;
278  }
279 
280  // Get Bin counts, assuming that our local machine arch is little-endian
281  for (int i=0; i<NumHistogramBins; i++) {
282  opc->binCount[i] = 0;
283  for (int j=0; j<2; j++) {
284  pos++;
285  opc->binCount[i] |= (((uint16_t)spiData[pos]) << 8*j);
286  }
287  }
288 
289  // Get mtof data
290  opc->bin1_mtof = spiData[33];
291  opc->bin3_mtof = spiData[34];
292  opc->bin5_mtof = spiData[35];
293  opc->bin7_mtof = spiData[36];
294 
295  // Get flow rate, and zero out any corrupt FP numbers
296  opc->flowRate = *(float *) &spiData[37];
297  valid_float("", &opc->flowRate);
298 
299  // Get Temperature or pressure (alternating)
300  uint32_t tempPressVal = 0;
301  tempPressVal |= spiData[41];
302  tempPressVal |= spiData[42] << 8;
303  tempPressVal |= spiData[43] << 16;
304  tempPressVal |= spiData[44] << 24;
305  if (tempPressVal < 1000) {
306  opc->temperature = tempPressVal;
307  opc->pressure = 0;
308  opc->haveTemperature = true;
309  opc->havePressure = false;
310  } else {
311  opc->pressure = tempPressVal;
312  opc->temperature = 0;
313  opc->haveTemperature = false;
314  opc->havePressure = true;
315  }
316 
317  // Get sampling period, and zero out any corrupt FP numbers
318  opc->samplePeriod = *(float *) &spiData[45];
319  valid_float("", &opc->samplePeriod);
320 
321  // Get checksom
322  opc->checksum = ((uint16_t)spiData[49]);
323  opc->checksum |= ((uint16_t)spiData[50]) << 8;
324 
325  // Check PM and ensure that the values are valid floating point numbers
326  bool isValid = true;
327  opc->PM1 = *(float *) &spiData[51];
328  if (!valid_float("PM01", &opc->PM1))
329  isValid = false;
330  opc->PM2_5 = *(float *) &spiData[55];
331  if (!valid_float("PM25", &opc->PM2_5))
332  isValid = false;
333  opc->PM10 = *(float *) &spiData[59];
334  if (!valid_float("PM10", &opc->PM10))
335  isValid = false;
336 
337  // Zero value testing, just to make sure the code works and allows 0's all the way through
338 #ifdef AIR_ZERO_TEST
339  for (int i=0; i<NumHistogramBins; i++)
340  opc->binCount[i] = 0;
341  opc->PM1 = opc->PM2_5 = opc->PM10 = 0.0;
342 #endif
343 
344  // Check buckets
345  uint32_t bin_sum = 0;
346  for (int i=0; i<NumHistogramBins; i++)
347  bin_sum += opc->binCount[i];
348 
349  // Report total samples analyzed
350  num_samples++;
351 
352  // Report nonzero values analyzed
353  if (bin_sum != 0 || opc->PM1 != 0.0 || opc->PM2_5 != 0.0 || opc->PM10 != 0.0)
354  num_nonzero_samples++;
355 
356  // Report valid values analyzed
357  if (isValid)
358  num_valid_samples++;
359 
360  // Done
361  return isValid;
362 
363 }

438 // Transmit an SPI command, and return the value in rx_buf buffer
439 bool spi_cmd(uint8_t *tx, uint16_t txlen, uint16_t rxlen) {
440  uint32_t err_code;
441  int i;
442 
443  // This of course will never happen. Defensive programming.
444  if (rxlen > sizeof(rx_buf)) {
445  DEBUG_PRINTF("Buffer overrun!\n");
446  rxlen = sizeof(rx_buf);
447  }
448 
449  // We've found that we cannot execute commands too quickly else we get garbage,
450  // as indicated by the very first byte of the reply not being 0xf3
451  nrf_delay_ms(500);
452 
453  // Do special handling for the commands requiring large receives
454  if (tx[0] != 0x3f && tx[0] != 0x30) {
455 
456  // Issue the normal command if it's not a special one
457  opc_spi_transfer_begin();
458  err_code = nrf_drv_spi_transfer(spi_context(), tx, txlen, rx_buf, rxlen);
459  if (err_code != NRF_SUCCESS) {
460  DEBUG_PRINTF("SPI Transfer (rcv) result = %04x\n", err_code);
461  stats()->errors_spi++;
462  stats()->errors_opc++;
463  return false;
464  }
465  if (!opc_spi_transfer_succeeded()) {
466  DEBUG_PRINTF("SPI Transfer TIMEOUT\n");
467  stats()->errors_spi++;
468  stats()->errors_opc++;
469  return false;
470  }
471 
472  } else {
473 
474  // Send just the first byte of the command. If we send the second byte, it
475  // has an impact on the first byte of what is ultimately received. No, I don't know why.
476  opc_spi_transfer_begin();
477  err_code = nrf_drv_spi_transfer(spi_context(), tx, txlen, &rx_buf[0], 1);
478  if (err_code != NRF_SUCCESS) {
479  DEBUG_PRINTF("SPI Transfer (cmd) result = %04x\n", err_code);
480  stats()->errors_spi++;
481  stats()->errors_opc++;
482  return false;
483  }
484  if (!opc_spi_transfer_succeeded()) {
485  DEBUG_PRINTF("SPI Transfer TIMEOUT\n");
486  stats()->errors_spi++;
487  stats()->errors_opc++;
488  return false;
489  }
490 
491  if (rx_buf[0] == 0xf3) {
492 
493  // Wait 5ms so that we skip over whatever trash was returned to us immediately
494  // following the command. This ensures that whatever we get afterward, which
495  // comes after quite a bit of a delay, will start cleanly.
496  nrf_delay_ms(5);
497 
498  // Receive each of these bytes individually. We've found that we can't do a single large
499  // read because the bytes apparently aren't yet available, and this technique introduces
500  // sufficient delay so as to pick them up individually successfully.
501  for (i=1; i<rxlen; i++) {
502  opc_spi_transfer_begin();
503  err_code = nrf_drv_spi_transfer(spi_context(), NULL, 0, &rx_buf[i], 1);
504  if (err_code != NRF_SUCCESS) {
505  DEBUG_PRINTF("OPC %02x error rcv[%d]\n", tx[0], i);
506  stats()->errors_spi++;
507  stats()->errors_opc++;
508  return false;
509  }
510  if (!opc_spi_transfer_succeeded()) {
511  DEBUG_PRINTF("OPC %02x xfer #%d TIMEOUT\n", tx[0], i);
512  stats()->errors_spi++;
513  stats()->errors_opc++;
514  return false;
515  }
516  }
517 
518  }
519  }
520 
521 
522  // Not ok if the first returned byte wasn't our OPC signature
523  if (rx_buf[0] != 0xf3) {
524  // Allow several pieces corrupt piece of data per reading, silently. This is
525  // an arbitrary number, however a) randomness on the SPI bus does indeed happen from
526  // time to time, and 2) we don't want to leave ALL corruption unflagged
527  if (++num_errors > OPC_IGNORED_ERRORS) {
528  stats()->errors_opc++;
529  DEBUG_PRINTF("OPC cmd 0x%02x bad header 0x%02x != 0xF3\n", tx[0], rx_buf[0]);
530  }
531  return false;
532  }
533 
534  // Do special command processing to unfold into other statics
535  bool good = true;
536  if (tx[0] == 0x30)
537  good = unpack_opc_data(&opc_data, rx_buf, 63);
538  else if (tx[0] == 0x3f)
539  good = unpack_opc_version(version, sizeof(version), rx_buf, 61);
540 
541  if (!good) {
542  // The first piece of data that comes back is ALWAYS invalid, and is skipped.
543  // But even beyond this, aw allow several pieces corrupt piece of data per reading,
544  // silently. This is a somewhat arbitrary number, however a) randomness on the
545  // SPI bus does indeed happen from time to time, and 2) we don't want to leave
546  // ALL corruption unflagged.
547  if (++num_errors > OPC_IGNORED_ERRORS) {
548  stats()->errors_opc++;
549  DEBUG_PRINTF("OPC error: zero %d, invalid %d, total %d\n", num_samples-num_nonzero_samples, num_samples-num_valid_samples, num_samples);
550  }
551  if (debug(DBG_SENSOR_SUPERMAX))
552  DEBUG_PRINTF("Skipping sample.\n");
553  return false;
554  }
555 
556  return (true);
557 
558 }

565 // Measure it
566 void s_opc_measure(void *s) {
567  float std1, std2_5, std10;
568 
569  // Compute the reported
570  reported_pm_1 = reported_pm_2_5 = reported_pm_10 = 0.0;
571  reported_std_1 = reported_std_2_5 = reported_std_10 = 0.0;
572  std1 = std2_5 = std10 = 0.0;
573  reported_count_00_38 = 0;
574  reported_count_00_54 = 0;
575  reported_count_01_00 = 0;
576  reported_count_02_10 = 0;
577  reported_count_05_00 = 0;
578  reported_count_10_00 = 0;
579  reported_count_seconds = 0;
580 
581  // Avoid div by zero!
582  if (num_samples_recorded) {
583  int i;
584 
585  for (i=0; i<num_samples_recorded; i++) {
586  reported_pm_1 += samples_PM1[i];
587  reported_pm_2_5 += samples_PM2_5[i];
588  reported_pm_10 += samples_PM10[i];
589  }
590  reported_pm_1 = reported_pm_1 / num_samples_recorded;
591  reported_pm_2_5 = reported_pm_2_5 / num_samples_recorded;
592  reported_pm_10 = reported_pm_10 / num_samples_recorded;
593 
594  reported_count_00_38 = count_00_38;
595  reported_count_00_54 = count_00_54;
596  reported_count_01_00 = count_01_00;
597  reported_count_02_10 = count_02_10;
598  reported_count_05_00 = count_05_00;
599  reported_count_10_00 = count_10_00;
600 
601  // Compute the standard deviations
602  reported_std_1 = std1 = compute_maximum_deviation(samples_PM1, num_samples_recorded);
603  reported_std_2_5 = std2_5 = compute_maximum_deviation(samples_PM2_5, num_samples_recorded);
604  reported_std_10 = std10 = compute_maximum_deviation(samples_PM10, num_samples_recorded);
605 
606  // Apply a filter to the reported STD values to save bandwidth
607  if (reported_pm_1 < AIR_MATERIAL_PM || reported_std_1 < (reported_pm_1*AIR_MATERIAL_STD_MULTIPLE))
608  reported_std_1 = 0;
609  if (reported_pm_2_5 < AIR_MATERIAL_PM || reported_std_2_5 < (reported_pm_2_5*AIR_MATERIAL_STD_MULTIPLE))
610  reported_std_2_5 = 0;
611  if (reported_pm_10 < AIR_MATERIAL_PM || reported_std_10 < (reported_pm_10*AIR_MATERIAL_STD_MULTIPLE))
612  reported_std_10 = 0;
613 
614  // Valid
615  reported_count_seconds = count_seconds;
616  num_valid_reports++;
617 
618  }
619 
620  // If we haven't measured sufficiently long, it's an error
621  if (num_samples_recorded >= OPC_SAMPLE_MIN_BINS)
622  reported = ever_reported = true;
623  else
624  stats()->errors_opc++;
625 
626  // If high variance, don't allow it to pollute our data. If repeated high variance, it's an error.
627  if (reported_std_1 != 0 || reported_std_2_5 != 0 || reported_std_10 != 0) {
628  if (++consecutive_std > 3)
629  stats()->errors_opc++;
630  } else
631  consecutive_std = 0;
632 
633  // Debug
634  if (debug(DBG_SENSOR_MAX)) {

683  if (num_samples_recorded < OPC_SAMPLE_MAX_BINS) {
684

687  samples_PM2_5[num_samples_recorded] = opc_data.PM2_5;
688  samples_PM10[num_samples_recorded] = opc_data.PM10;
689  num_samples_recorded++;

730 bool opc_version() {
731 
732  // Exit if we're not supposed to be here
733  if (!request_opc_version)
734  return true;
735 
736  // Get the version IF we can, but only try once. The most common failure is that
737  // we get the first byte "O", but that the remainder of the command fails
738  static uint8_t req_version[] = {0x3f};
739  static uint16_t rsp_version_length = 61;
740  if (!spi_cmd(req_version, sizeof(req_version), rsp_version_length))
741  strcpy(version, "(cannot get version)");
742 
743  DEBUG_PRINTF("%s\n", version);
744 
745  // Only give this a single attempt
746  request_opc_version = false;
747 
748  // Force a timer quantum upon return, to keep SPI commands from stacking up
749  return false;

752 // The real init, which is performed during polling
753 bool opc_init() {
754  uint16_t opc_init_retry = (OPC_LASER_RETRIES - opc_init_retries_left) + 1;
755 
756  // Exit if we're not supposed to be here
757  if (!request_opc_initialization)
758  return true;
759 
760  // Turn fan and laser power) ON
761  static uint8_t req_everything_on[] = {0x03, 0x00};
762  static uint8_t rsp_everything_on[] = {0xf3, 0x03};
763  if (!spi_cmd(req_everything_on, sizeof(req_everything_on), sizeof(rsp_everything_on)) && (rx_buf[1] == rsp_everything_on[1])) {
764  if (debug(DBG_SENSOR_SUPERMAX))
765  DEBUG_PRINTF("OPC Fan+Laser FAILURE (%d/%d)\n", opc_init_retry, OPC_LASER_RETRIES);
766  if (--opc_init_retries_left == 0) {
767  stats()->errors_opc++;
768  request_opc_initialization = false;
769  DEBUG_PRINTF("Giving up on OPC.\n");
770  }
771  return false;
772  }
773 
774  // Success
775  if (debug(DBG_SENSOR_SUPERMAX)) {
776  if (opc_init_retry == 1)
777  DEBUG_PRINTF("OPC Fan+Laser ON\n");
778  else
779  DEBUG_PRINTF("OPC Fan+Laser ON (%d/%d)\n", opc_init_retry, OPC_LASER_RETRIES);
780  }
781 
782  // Initialize state
783  s_opc_clear_measurement();
784  num_errors = 0;
785  num_valid_reports = 0;
786  num_samples = 0;
787  num_valid_samples = 0;
788  num_nonzero_samples = 0;
789  opc_polling_ok = true;
790 
791  // The first value that we receive is ALWAYS zero
792  num_samples_left_to_skip = 1;
793 
794  // Don't come back
795  request_opc_initialization = false;
796 
797  // Force a timer quantum upon return, to keep SPI commands from stacking up
798  return false;
