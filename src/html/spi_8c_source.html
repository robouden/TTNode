43 // One-time initialization of SPI subsystem
44 bool spi_init() {
45  uint32_t err_code;
46  uint16_t pin;
47  nrf_drv_spi_handler_t handler;
48 
49  // Exit if already initialized
50  if (fInit) {
51  DEBUG_PRINTF("SPI Init: already initialized!\n");
52  return true;
53  }
54 
55  if (debug(DBG_SENSOR_MAX))
56  DEBUG_PRINTF("SPI Init\n");
57 
58  // Get OPC setup info
59 #ifdef SPIOPC
60  s_opc_get_spi(&pin, &handler);
61 #else
62  Error: what SPI device??
63 #endif
64 
65  // Initialize SPI handling. This code only works
66  // for a single slave device and needs to be enhanced
67  // if we ever have a second slave.
68  nrf_drv_spi_config_t spi_config = NRF_DRV_SPI_DEFAULT_CONFIG;
69  spi_config.bit_order = NRF_DRV_SPI_BIT_ORDER_MSB_FIRST;
70  spi_config.mode = NRF_DRV_SPI_MODE_1;
71  spi_config.frequency = NRF_DRV_SPI_FREQ_1M;
72  spi_config.miso_pin = SPI_PIN_MISO;
73  spi_config.mosi_pin = SPI_PIN_MOSI;
74  spi_config.sck_pin = SPI_PIN_SCLK;
75  spi_config.ss_pin = pin;
76  err_code = nrf_drv_spi_init(&m_spi, &spi_config, handler);
77  if (err_code == NRF_SUCCESS)
78  fInit = true;
79  else {
80  DEBUG_PRINTF("SPI init error = 0x%04x\n", err_code);
81  fInit = false;
82  stats()->errors_spi++;
83  }
84 
85  return fInit;
86 
87 }

89 // Terminate SPI
90 bool spi_term() {
91 
92  // Exit if already ununinitialized
93  if (!fInit)
94  return false;
95 
96  if (debug(DBG_SENSOR_MAX))
97  DEBUG_PRINTF("SPI Term\n");
98 
99  // Uninitialize SPI, to stop the current drain on the about-to-be powered-off device
100  nrf_drv_spi_uninit(&m_spi);
101 
102  // Clear the select pin, because THIS IS THE CAUSE of a 7ma drain and we can safely do this
103  // once the SPI subsystem no longer owns the pin.
104  gpio_pin_set(SPI_PIN_SS_OPC, false);
105  gpio_cfg_input(SPI_PIN_SS_OPC);
106 
107  // Done
108  fInit = false;
109 
110  return true;
111 }
