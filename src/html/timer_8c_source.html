54 // Access to our app-maintained system clock
55 uint32_t get_seconds_since_boot() {
56  uint32_t ticks;
57 
58 #if defined(NSDKV10) || defined(NSDKV11)
59  app_timer_cnt_get(&ticks);
60 #else
61  ticks = app_timer_cnt_get();
62 #endif
63 
64  // If the clock has wrapped, just return the clock in our clock's granularity
65  if (ticks < ticks_at_measurement)
66  return seconds_since_boot;
67 
68  // Compute seconds since last increment of our clock
69  uint32_t elapsed_ticks = ticks - ticks_at_measurement;
70  uint32_t elapsed_seconds = elapsed_ticks / APP_TIMER_TICKS_PER_SECOND;
71 
72  // Return finer-grained time
73  return (seconds_since_boot + elapsed_seconds);
74 
75 }

77 // Set the date/time
78 void set_timestamp(uint32_t ddmmyy, uint32_t hhmmss) {
79 
80  // If we're in mobile mode AND we already have it, don't set it again
81  // or else it will cause our stamps to be blown away on every resampling
82  if (dt_seconds_since_boot_when_set != 0 && sensor_op_mode() == OPMODE_MOBILE)
83  return;
84 
85  // http://aprs.gids.nl/nmea/#rmc
86  // 225446 Time of fix 22:54:46 UTC
87  // 191194 Date of fix 19 November 1994
88  dt_date = ddmmyy;
89  dt_time = hhmmss;
90  dt_seconds_since_boot_when_set = get_seconds_since_boot();
91 
92  // Just for debugging, so we can see when we actually acquire a timestamp
93  uint16_t yr = (ddmmyy % 100) + 2000;
94  uint16_t mo = (ddmmyy/100) % 100;
95  uint16_t dy = (ddmmyy/10000) % 100;
96  uint16_t ss = hhmmss % 100;
97  uint16_t mm = (hhmmss/100) % 100;
98  uint16_t hh = (hhmmss/10000) % 100;
99  DEBUG_PRINTF("%02d/%02d/%02d %02d:%02d:%02d UTC\n", mo, dy, yr, hh, mm, ss);
100 
101  // Force a service update so that the new timestamp is uploaded ASAP
102  comm_initiate_service_update(false);
103 
104 }

106 // Get the date/time
107 bool get_current_timestamp(uint32_t *date, uint32_t *time, uint32_t *offset) {
108 
109  // Exit if we've never acquired date/time from gps
110  if (dt_seconds_since_boot_when_set == 0)
111  return false;
112 
113  // Return this to the caller. This is obviously crufty, however it's best
114  // to allow the server to do the offset calculation on the dates. The other
115  // benefit is that this is more compressable than a textual date/time.
116  if (date != NULL)
117  *date = dt_date;
118  if (time != NULL)
119  *time = dt_time;
120  if (offset != NULL)
121  *offset = (get_seconds_since_boot() - dt_seconds_since_boot_when_set);
122 
123  return true;
124 
125 }

127 // Get time since boot
128 char *time_since_boot() {
129  uint32_t secs = get_seconds_since_boot();
130  uint32_t mins = secs / 60;
131  uint32_t hrs = mins / 60;
132  uint32_t days = hrs / 24;
133  secs -= mins * 60;
134  mins -= hrs * 60;
135  hrs -= days * 24;
136  timebuf[0] = '\0';
137  if (days)
138  sprintf(timebuf, "%lud ", days);
139  if (hrs)
140  sprintf(&timebuf[strlen(timebuf)], "%luh ", hrs);
141  if (mins)
142  sprintf(&timebuf[strlen(timebuf)], "%lum ", mins);
143  sprintf(&timebuf[strlen(timebuf)], "%lus", secs);
144  return timebuf;
145 }

186 // Primary app timer
187 void tt_timer_handler(void *p_context) {
188  static bool overcurrent = false;
189  static uint32_t overcurrent_report = 0;
190 
191  // Remember the number of ticks the last time we set seconds_since_boot
192  uint32_t ticks;
193 #if defined(NSDKV10) || defined(NSDKV11)
194  app_timer_cnt_get(&ticks);
195 #else
196  ticks = app_timer_cnt_get();
197 #endif
198 
199  // Bump the number of seconds since boot, leaving overflow to be dealt with by users
200  seconds_since_boot += tt_fast_timer_mode ? TT_FAST_TIMER_SECONDS : TT_SLOW_TIMER_SECONDS;
201  ticks_at_measurement = ticks;
202 
203  // Exit if we've somehow gone re-entrant
204  static int inside_timer = 0;
205  if (inside_timer++ != 0) {
206  inside_timer--;
207  DEBUG_PRINTF("tt_timer REENTRANCY!\n");
208  return;
209  }
210 
211  // Refresh the timer operating mode, if necessary
212  timer_refresh_mode();
213 
214  // Update the status of whether or not the device is currently in-motion
215  gpio_motion_sense(MOTION_UPDATE);
216 
217  // Display handling
218 #ifdef SSD
219 
220  // Turn on the display after boot or if we've detected movement
221  if (!ssd1306_active())
222  if (get_seconds_since_boot() < (DROP_DISPLAY_MINUTES*60))
223  ssd1306_init();
224 
225  // Remember when we've first noticed the display having been turned on, for whatever reason
226  static uint32_t ssd_enabled = 0;
227  if (ssd_enabled == 0 && ssd1306_active())
228  ssd_enabled = get_seconds_since_boot();
229 
230  // IF we're armed, so we can turn it on again, turn off the display if it has been on for too long
231 #ifndef SSDKEEPALIVE
232  if (gpio_motion_sense(MOTION_QUERY_ARMED) && ssd_enabled != 0)
233  if (!ShouldSuppress(&ssd_enabled, DROP_DISPLAY_MINUTES*60)) {
234  while (ssd1306_active())
235  ssd1306_term();
236  ssd_enabled = 0;
237  }
238 #endif
239 
240 #endif
241 
242  // Notifiy if overcurrent is sensed
243  if (gpio_power_overcurrent_sensed()) {
244  overcurrent = true;
245  DEBUG_PRINTF("Overcurrent sensed!\n");
246  }
247 
248  // Report overcurrent only ever 15m
249  if (!ShouldSuppress(&overcurrent_report, 60 * 15))
250  if (overcurrent) {
251  overcurrent = false;
252  stats()->overcurrent_events++;
253  }
254 
255  // Checkpoint deferred NVRAM I/O if serial I/O is not in progress. In the case of
256  // oneshot mode, this means that neither comms nor PMS are using the uart. In the case
257  // of non-oneshot mode where the uart is always busy, this just means when comms is not active.
258  if (comm_uart_switching_allowed()) {
259  if (gpio_current_uart() == UART_NONE)
260  storage_checkpoint();
261  } else {
262  if (!comm_is_busy())
263  storage_checkpoint();
264  }
265 
266  // Restart if it's been requested
267  io_restart_if_requested();
268 
269  // Terminate solicitations if we're optimizing power
270  static bool fDropped = false;
271  if (!fDropped && io_optimize_power()) {
272  fDropped = true;
273  drop_bluetooth();
274  }
275 
276  // This is just defensive programming; they should shut down themselves WAY before this
277  if (seconds_since_boot > 15*60)
278  gpio_indicators_off();
279 
280  // Poll geiger counters
281 #ifdef GEIGERX
282  if (tt_fast_timer_mode)
283  geiger_poll();
284 #endif
285 
286  // Poll the sensor package BEFORE polling comms, so that if there is anything
287  // marked as "completed" by the sensor package it will be immediately communicated
288  sensor_poll();
289 
290  // Poll and advance our communications state machine
291  comm_poll();
292 
293  // Say hello if we're just now connecting to BT
294  welcome_message();
295 
296  // Report any UART errors, but only after comm_poll had a chance to check
297  serial_uart_error_check(false);
298 
299  // Exit
300  inside_timer--;
301 
302 }

304 // Initialize our app timers
305 void timer_init() {
306 
307  // Initialize the clock to a small random number of seconds, so that
308  // if someone intentionally tries to synchronize the power-on of multiple
309  // devices the Lora uploads won't also necessarily be synchronized
310  seconds_since_boot += io_get_random(60);
311 
312  // Init the completed task scheduler that lets us handle command
313  // processing outside the interrupt handlers, and instead via app_sched_execute()
314  // called from the main loop in main.c.
315  APP_SCHED_INIT(SCHED_MAX_EVENT_DATA_SIZE, SCHED_QUEUE_SIZE);
316 
317  // Init app timer support for the entire app, in a way such that it utilizes
318  // the app scheduler. We do this so that we don't execute our (sometimes-lengthy)
319  // timer functions at interrupt time, but rather via the app scheduler.
320  APP_TIMER_APPSH_INIT(APP_TIMER_PRESCALER, APP_TIMER_OP_QUEUE_SIZE, true);
321 
322  // If faking GPS time for debugging, set a timestamp for testing purposes
323 #ifdef FAKEGPSTIME
324  set_timestamp(201116, 123456);
325 #endif
326 
327  // Create our primary app timer
328  app_timer_create(&tt_timer, APP_TIMER_MODE_REPEATED, tt_timer_handler);
329 
330  // Create our debug output timer
331  btdebug_create_timer();
332 
333 }

349 // Start or stop timers based on mode
350 void timer_refresh_mode() {
351  bool fast_timer_mode_needed;
352 
353  // Determine whether or not we're in 'fast mode' based on mobile mode,
354  // which requires the fast clock for geiger sensing. Otherwise, we'll
355  // keep the clock slow for power-savings reasons.
356  if (sensor_op_mode() == OPMODE_MOBILE)
357  fast_timer_mode_needed = true;
358  else
359  fast_timer_mode_needed = false;
360 
361  // Switch from one clock to the other if we're in the wrong mode
362  if (tt_fast_timer_mode != fast_timer_mode_needed) {
363  tt_fast_timer_mode = fast_timer_mode_needed;
364  tt_request_timer_mode_reset = true;
365  };
366 
367 }
