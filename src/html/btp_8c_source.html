18 // Handle the BLE_GATTS_EVT_WRITE event
19 void on_write(btp_t *p_btp, ble_evt_t *p_ble_evt) {
20  ble_gatts_evt_write_t *p_evt_write = &p_ble_evt->evt.gatts_evt.params.write;
21  if (
22  (p_evt_write->handle == p_btp->rx_handles.cccd_handle)
23  &&
24  (p_evt_write->len == 2)
25  ) {
26  if (ble_srv_is_notification_enabled(p_evt_write->data)) {
27  p_btp->is_notification_enabled = true;
28  } else {
29  p_btp->is_notification_enabled = false;
30  }
31  } else if (
32  (p_evt_write->handle == p_btp->tx_handles.value_handle)
33  &&
34  (p_btp->data_handler != NULL)
35  ) {
36  p_btp->data_handler(p_btp, p_evt_write->data, p_evt_write->len);
37  } else {
38  // Do Nothing. This event is not relevant for this service.
39  }
40 }

141 void btp_on_ble_evt(btp_t *p_btp, ble_evt_t *p_ble_evt) {
142 
143  if ((p_btp == NULL) || (p_ble_evt == NULL)) {
144  return;
145  }
146 
147  switch (p_ble_evt->header.evt_id) {
148 
149  case BLE_GATTS_EVT_WRITE:

162 // BTP initialization, called from main services_init
163 void btp_init(btp_t *p_btp) {
164 
165  // Initialize the service structure.
166  p_btp->conn_handle = BLE_CONN_HANDLE_INVALID;
167  p_btp->data_handler = btp_data_handler;
168  p_btp->is_notification_enabled = false;
169 
170  /* [Adding proprietary Service to S110 SoftDevice] */
171  // Add our custom base 128-bit UUID to the internal table.
172  // Note that the second-to-highest order 16 bit word is actually ignored
173  // in all use of that table, because it will be overridden
174  // by a 16-bit UUID "shortcut" specified below.
175  uint32_t err_code;
176  ble_uuid_t ble_uuid;
177  ble_uuid128_t btp_base_uuid = BTP_BASE_UUID;
178 
179  err_code = sd_ble_uuid_vs_add(&btp_base_uuid, &p_btp->uuid_type);
180  if (err_code != NRF_SUCCESS) {
181  DEBUG_CHECK(err_code);
182  return;
183  }
184 
185  // Our 16-bit UUID "shortcut" that will be overlaid on the above
186  ble_uuid.uuid = BLE_UUID_BTP_SERVICE;
187  ble_uuid.type = p_btp->uuid_type;
188 
189  // Add the service.
190  err_code = sd_ble_gatts_service_add(BLE_GATTS_SRVC_TYPE_PRIMARY, &ble_uuid, &p_btp->service_handle);
191  if (err_code != NRF_SUCCESS) {
192  DEBUG_CHECK(err_code);
193  return;
194  }
195 
196  // Add the RX Characteristic.
197  err_code = rx_char_add(p_btp);
198  if (err_code != NRF_SUCCESS) {
199  DEBUG_CHECK(err_code);
200  return;
201  }
202 
203  // Add the TX Characteristic.
204  err_code = tx_char_add(p_btp);
205  if (err_code != NRF_SUCCESS) {
206  DEBUG_CHECK(err_code);
207  return;
208  }
209 
210 }

212 // Method to check to see if we are connected
213 bool btp_can_send(btp_t *p_btp) {
214 
215  if (p_btp == NULL)
216  return false;
217 
218  // We aren't fully connected until we get our first on_write which enables notifications
219  if ((p_btp->conn_handle == BLE_CONN_HANDLE_INVALID) || (!p_btp->is_notification_enabled))
220  return false;
221 
222  return true;
223

226 // Send a single packet to host by generating an HVX notification on the rx characteristic
227 uint32_t btp_string_send(btp_t *p_btp, uint8_t *p_string, uint16_t length) {
228  ble_gatts_hvx_params_t hvx_params;
229  uint32_t status;
230 
231  if (!btp_can_send(p_btp))
232  return NRF_ERROR_INVALID_STATE;
233  if (length > BTP_MAX_DATA_LEN)
234  return NRF_ERROR_INVALID_PARAM;
235 
236  memset(&hvx_params, 0, sizeof(hvx_params));
237 
238  hvx_params.handle = p_btp->rx_handles.value_handle;
239  hvx_params.p_data = p_string;
240  hvx_params.p_len = &length;
241  hvx_params.type = BLE_GATT_HVX_NOTIFICATION;
242 
243  status = sd_ble_gatts_hvx(p_btp->conn_handle, &hvx_params);
244 
245 #ifdef DEBUG_USES_UART
246  if (status == 0x3401) {
247  // Handy debugging note because of the bizarre error code:
248  // https://devzone.nordicsemi.com/question/6085/strange-error-code-13313-0x3401-returned-by-sd_ble_gatts_hvx
249  if (debug(DBG_BT))
250  DEBUG_PRINTF("*** GATTS system attributes missing ***\n");
251  } else if (status == NRF_ERROR_INVALID_STATE) {
252  // Error that cropped up in SDK12 with MTU request not completing?
253  // https://devzone.nordicsemi.com/question/93779/s132-30-mtu-exchange-not-completing/
254  if (debug(DBG_BT))
255  DEBUG_PRINTF("*** Invalid State or MTU Exchange not completed ***\n");
256  } else {
257  if (debug(DBG_BT))
258  DEBUG_PRINTF("sd_ble_gatts_hvx(hdl=0x%04x) error 0x%04x\n", p_btp->conn_handle, status);
259  }
260 #endif
261 
262  return(status);
263 
264 }
