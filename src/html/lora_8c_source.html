162 // Set LPWAN to idle or receive mode, based on modes including LoRaWAN mode
163 void restart_receive() {
164  if (LoRaWAN_mode || !receive_mode_active()) {
165 #ifdef DISABLE_LPWAN_SLEEP
166  // Only if debugging and want to keep the chip awake so we can try out commands
167  comm_cmdbuf_set_state(&fromLora, COMM_STATE_IDLE);
168 #else
169  if (sleepDisabled || !loraInitCompleted) {
170  comm_cmdbuf_set_state(&fromLora, COMM_STATE_IDLE);
171  } else {
172  char command[32];
173  if (fromLora.state == COMM_LORA_SLEEPRPL) {
174  DEBUG_PRINTF("Already sleeping!\n");
175  } else {
176  // Send the sleep command
177  sprintf(command, "sys sleep %d", LPWAN_SLEEP_MILLISECONDS);
178  lora_send(command);
179  comm_cmdbuf_set_state(&fromLora, COMM_LORA_SLEEPRPL);
180  // Totally prohibit ANY serial output while we are sleeping
181  serial_transmit_enable(false);
182  }
183  }
184 #endif
185  } else {
186  lora_send("radio rx 0");
187  comm_cmdbuf_set_state(&fromLora, COMM_LORA_RXRPL);
188  // Totally prohibit ANY serial output while we are sleeping
189  serial_transmit_enable(false);
190  }
191 }

241 // Process the hex-encoded portion of a message received from the LPWAN
242 void process_rx(char *in) {
243  uint8_t buffer[CMD_MAX_LINELENGTH];
244  uint16_t msgtype, decodedBytes;
245  ttproto_Telecast message;
246 
247  // Regardless of what we received, indicate that we're no longer waiting for
248  // a TTServe reply, because we only want to listen for a single receive
249  // window, for power reasons If we don't pick it up now, we'll pick it up
250  // eventually on the next message to the service.
251  awaitingTTServeReply = false;
252 
253  // Decode the message
254  msgtype = comm_decode_received_message(in, &message, buffer, sizeof(buffer) - 1, &decodedBytes);
255  if (msgtype == MSG_NOT_DECODED) {
256  DEBUG_PRINTF("Unrecognized message received.\n");
257  // Bad received message
258  setidlestateL();
259  return;
260  }
261 
262  // Bump stats about what we've received on the wire
263  stats_io(0, decodedBytes);
264 
265  // A reply from an "are you there?" ping we sent to TTGATE?
266  if (msgtype == MSG_REPLY_TTGATE) {
267  awaitingTTGateReply = false;
268  DEBUG_PRINTF("Lora gateway responded\n");
269  processstateL(COMM_LORA_INITCOMPLETED);
270  return;
271  }
272 
273  // The reply is NOT from TTGATE. If we're awaiting one from TTGATE, though,
274  // discard it and just send a ping to TTGATE to attempt to get that reply.
275  if (awaitingTTGateReply) {
276  setstateL(COMM_STATE_IDLE);
277  DEBUG_PRINTF("Unrecognized message received while awaiting Lora gateway response\n");
278  // Only retry for a reasonable amount of time, else a bunch of devices searching
279  // for a gateway would keep each other occupied forever.
280  if (!ShouldSuppress(&beganAwaitingTTGateReply, 30))
281  setidlestateL();
282  else {
283  uint32_t saveBeganAwaitingTTGateReply = beganAwaitingTTGateReply;
284  send_ping_to_service(REPLY_TTGATE);
285  beganAwaitingTTGateReply = saveBeganAwaitingTTGateReply;
286  }
287  return;
288  }
289 
290  // If this is a reply from a text message that we sent to TTSERVE, process it.
291  if (msgtype == MSG_REPLY_TTSERVE) {
292  DEBUG_PRINTF("Reply received from TTServe\n");
293  awaitingTTGateReply = false;
294  // Set idle state before we do this so that we can process recv_message without being "busy"
295  comm_cmdbuf_reset(&fromLora);
296  setidlestateL();
297  recv_message_from_service((char *)buffer);
298  return;
299  }
300 
301  // If this is a text message, just display it.
302  if (msgtype == MSG_TELECAST) {
303  awaitingTTGateReply = false;
304  DEBUG_PRINTF("%s\n", (char *)buffer);
305  comm_cmdbuf_reset(&fromLora);
306  setidlestateL();
307  return;
308  }
309 
310  // This is a Safecast message. If we're not configured as a relay, we're done.
311  STORAGE *s = storage();
312  if (LoRaWAN_mode || (s->flags & FLAG_RELAY) == 0) {
313  if (debug(DBG_COMM_MAX))
314  DEBUG_PRINTF("(safecast message ignored)\n");
315  comm_cmdbuf_reset(&fromLora);
316  setidlestateL();
317  return;
318  }
319 
320  // We know that it's a relay message
321  if (debug(DBG_COMM_MAX))
322  DEBUG_PRINTF("Relaying Safecast message.\n");
323 
324  // This is a Safecast or bGeigie message that we're relaying.
325  // Check to make sure that it has a device ID, and that we haven't already relayed it
326  if (!message.has_device_id) {
327  comm_cmdbuf_reset(&fromLora);
328  setidlestateL();
329  return;
330  }
331 
332  // Determine whether or not we've already relayed it
333  uint32_t thisDeviceID = io_get_device_address();
334  if ((message.has_relay_device1 && message.relay_device1 == thisDeviceID)
335  || (message.has_relay_device2 && message.relay_device2 == thisDeviceID)
336  || (message.has_relay_device3 && message.relay_device3 == thisDeviceID)
337  || (message.has_relay_device4 && message.relay_device4 == thisDeviceID)
338  || (message.has_relay_device5 && message.relay_device5 == thisDeviceID)) {
339  DEBUG_PRINTF("RELAY: already relayed\n");
340  comm_cmdbuf_reset(&fromLora);
341  setidlestateL();
342  return;
343  }
344 
345  // Assign it a routing slot
346  if (!message.has_relay_device1) {
347  message.has_relay_device1 = true;
348  message.relay_device1 = thisDeviceID;
349  } else if (!message.has_relay_device2) {
350  message.has_relay_device2 = true;
351  message.relay_device2 = thisDeviceID;
352  } else if (!message.has_relay_device3) {
353  message.has_relay_device3 = true;
354  message.relay_device3 = thisDeviceID;
355  } else if (!message.has_relay_device4) {
356  message.has_relay_device4 = true;
357  message.relay_device4 = thisDeviceID;
358  } else if (!message.has_relay_device5) {
359  message.has_relay_device5 = true;
360  message.relay_device5 = thisDeviceID;
361  } else {
362  // If no slots left, relay no further.
363  DEBUG_PRINTF("RELAY: too many hops\n");
364  comm_cmdbuf_reset(&fromLora);
365  setidlestateL();
366  return;
367  }
368 
369  // Encode the PB to be transmitted
370  pb_ostream_t stream = pb_ostream_from_buffer(toRelayBuffer, sizeof(toRelayBuffer));
371  uint16_t status = pb_encode(&stream, ttproto_Telecast_fields, &message);
372  if (!status) {
373  DEBUG_PRINTF("Relay pb_encode: %s\n", PB_GET_ERROR(&stream));
374  comm_cmdbuf_reset(&fromLora);
375  setidlestateL();
376  return;
377  }
378  toRelayBufferLength = stream.bytes_written;
379  toRelayDevice = message.device_id;
380 
381  // Get the SNR of the received message, which will then relay it.
382  lora_send("radio get snr");
383  setstateL(COMM_LORA_GETSNRRPL);
384 
385 }

416 // Return true if transmitting would be pointless
417 bool lora_is_busy() {
418 
419  // Take not of whether we're temporarily "not listening" and thus not able to transmit
420  bool fSleeping = (fromLora.state == COMM_LORA_RXRPL || fromLora.state == COMM_LORA_SLEEPRPL);
421 
422  // Exit if we're in the middle of sending a different message,
423  // This can happen because transmits take a finite amount of time,
424  // and we might be being asked to transmit both for the local
425  // geiger counter as well as the URSC-connected bGeigie.
426  if (fromLora.state != COMM_STATE_IDLE && !fSleeping) {
427 
428  // Causes communications issues doing this debug output in the interrupt handler,
429  // , so only use this when doing super debugging.
430 #ifdef FLOWTRACE
431  if (fromLora.state == COMM_LORA_TXRPL2)
432  DEBUG_PRINTF("(Can't send - busy xmitting!)\n");
433  else
434  DEBUG_PRINTF("(Can't send - busy %d!)\n", fromLora.state);
435 #endif
436 
437  return true;
438  }
439 
440  if (fSleeping && deferred_transmit) {
441 #ifdef FLOWTRACE
442  if (debug(DBG_COMM_MAX))
443  DEBUG_PRINTF("(Can't send - something else waiting.)\n");
444 #endif
445  return true;
446  }
447 
448  if (fromLora.state == COMM_LORA_RXRPL) {
449  if (debug(DBG_COMM_MAX))
450  DEBUG_PRINTF("(Busy waiting for reply.)\n");
451  return true;
452  }
453 
454  return false;
455 }

457 // Transmit a well-formed protocol buffer to the LPWAN as a message
458 bool lora_send_to_service(uint8_t *buffer, uint16_t length, uint16_t RequestType) {
459  char *command;
460 
461  // Exit if we're not yet initialized
462  if (!comm_is_initialized()) {
463  if (debug(DBG_TX))
464  DEBUG_PRINTF("Lora not yet initialized.\n");
465  return false;
466  }
467 
468  // Once every [N] minutes, even if this wasn't a request asking for a reply, force a RequestType so
469  // that we give TTGATE an opportunity to send us a "down" message in case it can't reach the service.
470  if (RequestType == REPLY_NONE)
471  if (!ShouldSuppress(&lastGatewayConnectivityCheck, FAILOVER_CHECK_MINUTES * 60L))
472  RequestType = REPLY_TTSERVE;
473 
474  // If this is a request that requires a reply, remember that we're doing so.
475  awaitingTTGateReply = false;
476  awaitingTTServeReply = false;
477  if (RequestType == REPLY_TTGATE) {
478  comm_set_connect_state(CONNECT_STATE_LORA_GATEWAY);
479  DEBUG_PRINTF("Requesting Lora gateway reply\n");
480  awaitingTTGateReply = true;
481  beganAwaitingTTGateReply = get_seconds_since_boot();
482  } if (RequestType == REPLY_TTSERVE)
483  awaitingTTServeReply = true;
484 
485  // If we're busy doing something else, drop this
486  if (lora_is_busy())
487  return false;
488 
489  // Do different types of transmit, based on mode. Start by assuming no retries.
490  xmitReplyRetriesLeft = 0;
491  if (LoRaWAN_mode) {
492  if (RequestType == REPLY_NONE) {
493  command = "mac tx uncnf 1 ";
494  } else {
495  command = "mac tx cnf 1 ";
496  // Indicate that we should do a LoraWAN ping for a reply just for good measure,
497  // simply because we only have a small window to receive.
498  xmitReplyRetriesLeft = XMIT_REPLY_RETRIES_LORAWAN;
499  }
500  } else {
501  command = "radio tx ";
502  if (RequestType == REPLY_TTGATE) {
503  // Indicate that we should do a Lora ping for a reply just for good measure,
504  // simply because we only have a small window to receive.
505  xmitReplyRetriesLeft = XMIT_REPLY_RETRIES_LORA;
506  }
507  }
508 
509  // Convert the binary buffer to a command with hexified data
510  HexCommand(deferred_transmit_buffer, sizeof(deferred_transmit_buffer),
511  command,
512  buffer, length);
513 
514  // Bump stats about what we've transmitted
515  stats_io(length, 0);
516 
517  // If we are sleeping/receiving, defer the xmit until it completes
518  if (fromLora.state == COMM_LORA_RXRPL || fromLora.state == COMM_LORA_SLEEPRPL) {
519  deferred_transmit = true;
520 #ifdef FLOWTRACE
521  DEBUG_PRINTF("(I'll send in a moment.)\n");
522 #endif
523  } else {
524  deferred_transmit = false;
525  lora_send(deferred_transmit_buffer);
526  setstateL(COMM_LORA_TXRPL1);
527  }
528 
529  // Notify caller that we transmitted it (or successfully deferred it)
530  return true;
531 }

533 // Do LPWAN reset processing
534 bool lora_needed_to_be_reset() {
535  uint32_t secondsSinceBoot = get_seconds_since_boot();
536 
537  // Initialize the entire cmd subsystem only after the chip has
538  // had a chance to stabilize after boot.
539  if (!loraInitCompleted && !loraInitInProgress && secondsSinceBoot > BOOT_DELAY_UNTIL_INIT) {
540  lora_reset(false);
541  return true;
542  }
543 
544  if (watchdog_set_time >= secondsSinceBoot)
545  lora_watchdog_reset();
546 
547  // This is a special case that just causes us to go idle if we
548  // happen to miss the OK reply to a sleep request.
549  // This is just intended to jostle it into response
550  if (secondsSinceBoot >= SLEEP_WATCHDOG_SECONDS) {
551  if ((secondsSinceBoot - watchdog_set_time) > SLEEP_WATCHDOG_SECONDS)
552  if (fromLora.state == COMM_LORA_SLEEPRPL) {
553  serial_transmit_enable(true);
554  lora_send("sys get ver");
555  setstateL(COMM_STATE_IDLE);
556  return false;
557  }
558  }
559 
560  // Reset all state and the LPWAN chip if appropriate,
561  // handling clock wrap. Only do this, though, if
562  // the lpWAN is in a non-idle state, because we
563  // don't want to be performing resets of completely
564  // idle devices.
565  if (secondsSinceBoot >= LORA_WATCHDOG_SECONDS) {
566  if ((secondsSinceBoot - watchdog_set_time) > LORA_WATCHDOG_SECONDS) {
567  // Auto-switch if the module is missing or nonresponsive
568  if (fromLora.state == COMM_LORA_GETVERRPL && storage()->wan == WAN_AUTO) {
569  stats()->errors_lora++;
570  comm_select(COMM_FONA, "module failure - lora handoff");
571  return true;
572  }
573  if (fromLora.state != COMM_STATE_IDLE) {
574  DEBUG_PRINTF("WATCHDOG: stuck in fromLora(%d)\n", fromLora.state);
575  lora_reset(true);
576  stats()->errors_lora++;
577  return true;
578  }
579  }
580  }
581 
582  // Not reset
583  return false;
584 
585 }

587 // Initialize or reinitialize the LPWAN chip and state machine
588 void lora_reset(bool force) {
589 
590  // Prevent recursion
591  if (!force && loraInitInProgress)
592  return;
593 
594  // If we're forcing a reset, init as though it had never been completed,
595  // which means for TTN a full re-join
596  if (force)
597  loraInitEverCompleted = false;
598 
599  // Kick off a device reset
600  processstateL(COMM_LORA_RESETREQ);
601 
602 }

652 // Primary state processing of the command buffer
653 void lora_process() {
654  char buffer[CMD_MAX_LINELENGTH];
655 
656  // If there was a spurious app sched event remaining after powering off the module, ignore it
657  if (gpio_current_uart() != UART_LORA)
658  return;
659 
660  // If it's not complete, just exit.
661  if (!fromLora.complete)
662  return;
663 
664  // If we're idle, set the state based on buffer contents
665  if (fromLora.state == COMM_STATE_IDLE)
666  lora_complete();
667 
668  // If extreme debugging
669  if (debug(DBG_RX) && fromLora.buffer[0] != '\0')
670  if (fromLora.state != COMM_LORA_RESETREQ)
671  DEBUG_PRINTF("<%d %s\n", fromLora.state, fromLora.buffer);
672 
683 
684  switch (fromLora.state) {
685 
689 
690  case COMM_LORA_RESETREQ: {
691  DEBUG_PRINTF("LPWAN initializing\n");
692  if (loraFirstResetAfterInit)
693  loraFirstResetAfterInit = false;
694  else
695  stats()->resets++;
696  gpio_indicate(INDICATE_LORA_INITIALIZING);
697  lora_watchdog_reset();
698  loraInitCompleted = false;
699  loraInitInProgress = true;
700  loraInitLastInitiated = get_seconds_since_boot();
701  serial_transmit_enable(true);
702  setstateL(COMM_STATE_IDLE);
703  lora_send("sys get ver");
704  setstateL(COMM_LORA_GETVERRPL);
705  break;
706  }
707 
708  case COMM_LORA_GETVERRPL: {
709  STORAGE *s = storage();
710  // Delay after any get ver. This matters for Microchip timing reasons.
711  nrf_delay_ms(MICROCHIP_LONG_DELAY_MS);
712  // There may be garbage, so retry until we get in sync
713  if (!loraInitEverCompleted)
714  DEBUG_PRINTF("%s\n", &fromLora.buffer[fromLora.args]);
715  if (thisargisL("rn2483")) {
716  isRN2483 = true;
717  strlcpy(stats()->module_lora, "RN2493", sizeof(stats()->module_lora)-1);
718  if (s->lpwan_region[0] == '\0')
719  strlcpy(s->lpwan_region, "eu", sizeof(s->lpwan_region)-1);
720  } else if (thisargisL("rn2903")) {
721  isRN2903 = true;
722  strlcpy(stats()->module_lora, "RN2903", sizeof(stats()->module_lora)-1);
723  if (s->lpwan_region[0] == '\0')
724  strlcpy(s->lpwan_region, "us", sizeof(s->lpwan_region)-1);
725  } else if (thisargisL("invalid_param")) {
726  // This is totally expected, as we are trying to re-sync
727  lora_send("sys get ver");
728  setstateL(COMM_LORA_GETVERRPL);
729  break;
730  } else {
731  DEBUG_PRINTF("Unexpected chip ID/version\n");
732  lora_send("sys get ver");
733  setstateL(COMM_LORA_GETVERRPL);
734  break;
735  }
736  // Init into the appropriate receive mode
737  if ((s->flags & (FLAG_LISTEN|FLAG_RELAY)) != 0)
738  receive_from_lpwan_mode = true;
739  else
740  receive_from_lpwan_mode = false;
741  // If we're restarting, then jump into the appropriate mode
742  if (loraInitEverCompleted) {
743  processstateL(LoRaWAN_mode ? COMM_LORA_LORAWANREQ : COMM_LORA_LORAREQ);
744  break;
745  }
746  // This is a first time init
747  if (s->wan == WAN_LORA || s->wan == WAN_LORAWAN)
748  LoRaWAN_mode_try_the_other_on_failure = false;
749  if (s->wan == WAN_LORA_THEN_LORAWAN || s->wan == WAN_AUTO || s->wan == WAN_LORA)
750  processstateL(COMM_LORA_LORAREQ);
751  else
752  processstateL(COMM_LORA_LORAWANREQ);
753  break;
754  }
755 
756  case COMM_LORA_SYSRESETRPL: {
757  // Let things settle down after the sys reset
758  nrf_delay_ms(MICROCHIP_LONG_DELAY_MS);
759  if (loraInitEverCompleted) {
760  processstateL(COMM_LORA_HWEUIDONE);
761  break;
762  }
763  lora_send("sys get hweui");
764  setstateL(COMM_LORA_HWEUIRPL);
765  break;
766  }
767 
768  case COMM_LORA_HWEUIRPL: {
769  char *devEui = (char *) &fromLora.buffer[fromLora.args];
770  // Save it in NVRAM if it has changed
771  if (strcmp(storage()->ttn_dev_eui, devEui) != 0) {
772  strlcpy(storage()->ttn_dev_eui, devEui, sizeof(storage()->ttn_dev_eui));
773  DEBUG_PRINTF("Saving DevEUI: %s\n", devEui);
774  storage_save(false);
775  nrf_delay_ms(MICROCHIP_LONG_DELAY_MS);
776  } else if (!loraInitEverCompleted) {
777  DEBUG_PRINTF("DevEui: %s\n", devEui);
778  }
779  processstateL(COMM_LORA_HWEUIDONE);
780  break;
781  }
782 
783  case COMM_LORA_HWEUIDONE: {
784  // We're always in this mode after a sys reset
785  LoRaWAN_mode = true;
786  // Dispatch based on what mode we want to be in
787  if (LoRaWAN_mode_desired_after_reset)
788  processstateL(COMM_LORA_MACRESUMERPL);
789  else {
790  lora_send("mac pause");
791  setstateL(COMM_LORA_MACPAUSERPL);
792  LoRaWAN_mode = false;
793  }
794  break;
795  }
796 
800 
801  case COMM_LORA_INITCOMPLETED: {
802  // Stay in whatever mode we were first in after boot, across resets
803  LoRaWAN_mode_try_the_other_on_failure = false;
804  // Done with initialization
805  loraInitInProgress = false;
806  loraInitCompleted = true;
807  loraInitEverCompleted = true;
808  comm_select_completed();
809  if (LoRaWAN_mode) {
810  comm_set_connect_state(CONNECT_STATE_LORAWAN_ACTIVE);
811  gpio_indicate(INDICATE_LORAWAN_CONNECTED);
812  } else {
813  comm_set_connect_state(CONNECT_STATE_LORA_ACTIVE);
814  gpio_indicate(INDICATE_LORA_CONNECTED);
815  }
816  DEBUG_PRINTF("LPWAN online: %s\n", LoRaWAN_mode ? "LoRaWAN" : "LoRa");
817  // Use brute force to ensure that we don't just start jamming commands
818  // down the device's throat after successful initialization. We've
819  // seen cases where just after a reset the lpwan chip goes into a
820  // state in which it just does nothing but say "busy"
821  nrf_delay_ms(250);
822  setidlestateL();
823  // Initiate a service upload if one is pending
824  comm_update_service();
825  break;
826  }
827 
831 
832  case COMM_LORA_LORAREQ: {
833  DEBUG_PRINTF("Entering LoRa mode\n");
834  lorafpRegionCommandNumber = 0;
835  gpio_indicate(INDICATE_LORA_INITIALIZING);
836  LoRaWAN_mode_desired_after_reset = false;
837  setstateL(COMM_STATE_IDLE);
838  lora_send("sys reset");
839  setstateL(COMM_LORA_SYSRESETRPL);
840  break;
841  }
842 
843  case COMM_LORA_MACPAUSERPL: {
844  // Indicate that we are now NOT in LoRaWAN mode
845  STORAGE *s = storage();
846  if ((s->flags & FLAG_RELAY) == 0)
847  sprintf(buffer, "radio set wdt %lu", RECEIVE_TIMEOUT_LISTEN_SECONDS * 1000L);
848  else
849  sprintf(buffer, "radio set wdt %lu", RECEIVE_TIMEOUT_RELAY_SECONDS * 1000L);
850  lora_send(buffer);
851  setstateL(COMM_LORA_SETWDTRPL);
852  break;
853  }
854 
855  case COMM_LORA_SETWDTRPL: {
856  STORAGE *s = storage();
857  // Stay in the same state, sending frequency plan commands until there are none left for the region
858  if (lorafp_get_command(s->lpwan_region, false, lorafpRegionCommandNumber, buffer, sizeof(buffer))) {
859  lorafpRegionCommandNumber++;
860  lora_send(buffer);
861  setstateL(COMM_LORA_SETWDTRPL);
862  break;
863  }
864  }
865  // fallthrough when no more commands to send
866  case COMM_LORA_SENDFQRPL: {
867  setstateL(COMM_STATE_IDLE);
868  // Send a ping in order to see if a gateway is actually there, because
869  // if it isn't there we may give up and switch to LoRaWAN mode
870  if (LoRaWAN_mode_try_the_other_on_failure) {
871  DEBUG_PRINTF("Is Lora gateway present?\n");
872  send_ping_to_service(REPLY_TTGATE);
873  } else
874  processstateL(COMM_LORA_INITCOMPLETED);
875  break;
876  }
877 
881 
882  case COMM_LORA_LORAWANREQ: {
883  DEBUG_PRINTF("Entering LoRaWAN mode\n");
884  lorafpRegionCommandNumber = 0;
885  gpio_indicate(INDICATE_LORAWAN_INITIALIZING);
886  LoRaWAN_mode_desired_after_reset = true;
887  setstateL(COMM_STATE_IDLE);
888  if (loraInitEverCompleted) {
889  lora_send("mac join abp");
890  setstateL(COMM_LORA_RESTORESTATERPL);
891  } else {
892  lora_send("sys reset");
893  setstateL(COMM_LORA_SYSRESETRPL);
894  }
895  break;
896  }
897 
898  case COMM_LORA_MACRESUMERPL: {
899  accept_retries = 0;
900  sprintf(buffer, "mac set deveui %s", storage()->ttn_dev_eui);
901  lora_send(buffer);
902  setstateL(COMM_LORA_SETDEVEUIRPL);
903  break;
904  }
905 
906  case COMM_LORA_SETDEVEUIRPL: {
907  char *appEui = storage()->ttn_app_eui;
908  sprintf(buffer, "mac set appeui %s", appEui);
909  lora_send(buffer);
910  setstateL(COMM_LORA_SETAPPEUIRPL);
911  break;
912  }
913 
914  case COMM_LORA_SETAPPEUIRPL: {
915  char *appKey = storage()->ttn_app_key;
916  sprintf(buffer, "mac set appkey %s", appKey);
917  lora_send(buffer);
918  setstateL(COMM_LORA_SETAPPKEYRPL);
919  break;
920  }
921 
922  case COMM_LORA_SETAPPKEYRPL: {
923 #if 1 // Disabled 2017-05-03 because the commands don't seem to be compatible with EU gateways
924  // Re-enabled 2017-05-05 after upgrading lorafp
925  STORAGE *s = storage();
926  // Stay in the same state, sending frequency plan commands until there are none left for the region
927  if (lorafp_get_command(s->lpwan_region, true, lorafpRegionCommandNumber, buffer, sizeof(buffer))) {
928  lorafpRegionCommandNumber++;
929  lora_send(buffer);
930  setstateL(COMM_LORA_SETAPPKEYRPL);
931  break;
932  }
933 #endif
934  }
935  // fallthrough when no more commands to send
936  // See https://www.microchip.com/forums/m945840.aspx#951895
937  case COMM_LORA_SENDFPRPL: {
938  lora_send("mac set devaddr 00000000");
939  setstateL(COMM_LORA_REJOIN1);
940  break;
941  }
942 
943  // See https://www.microchip.com/forums/m945840.aspx#951895
944  case COMM_LORA_REJOIN1: {
945  lora_send("mac set nwkskey 00000000000000000000000000000000");
946  setstateL(COMM_LORA_REJOIN2);
947  break;
948  }
949 
950  // See https://www.microchip.com/forums/m945840.aspx#951895
951  case COMM_LORA_REJOIN2: {
952  lora_send("mac set appskey 00000000000000000000000000000000");
953  setstateL(COMM_LORA_REJOIN3);
954  break;
955  }
956 
957  case COMM_LORA_REJOIN3: {
958  lora_send("mac set adr off");
959  setstateL(COMM_LORA_SETADRRPL);
960  break;
961  }
962 
963  case COMM_LORA_SAVESTATERPL: {
964  lora_do_term();
965  setstateL(COMM_STATE_IDLE);
966  break;
967  }
968 
969  case COMM_LORA_RESTORESTATERPL: {
970  if (thisargisL("ok")) {
971  setstateL(COMM_LORA_RESTORESTATERPL);
972  } else if (thisargisL("accepted")) {
973  processstateL(COMM_LORA_INITCOMPLETED);
974  } else {
975  processstateL(COMM_LORA_RESETREQ);
976  }
977  break;
978  }
979 
980  case COMM_LORA_SETADRRPL:
981  // fallthrough
982  case COMM_LORA_RETRYJOIN: {
983  comm_set_connect_state(CONNECT_STATE_LORAWAN_GATEWAY);
984  lora_send("mac join otaa");
985  setstateL(COMM_LORA_JOINRPL);
986  break;
987  }
988 
989  case COMM_LORA_JOINRPL: {
990  bool fRetry = false;
991  if (thisargisL("ok")) {
992  // this is expected response from initiating the rcv,
993  // so just reset the buffer and keep waiting for a message to come in
994  setstateL(COMM_LORA_JOINRPL);
995  } else if (thisargisL("accepted")) {
996  stats()->joins++;
997  stats()->joins_today++;
998  processstateL(COMM_LORA_INITCOMPLETED);
999  } else if (thisargisL("busy")) {
1000  DEBUG_PRINTF("Join busy, retrying.\n");
1001  // This is not at all expected, but it means that we're
1002  // moving too quickly and we should try again.
1003  fRetry = true;
1004  } else if (thisargisL("no_free_ch")) {
1005  DEBUG_PRINTF("No free channel on join, retrying.\n");
1006  fRetry = true;
1007  } else if (thisargisL("denied")) {
1008  DEBUG_PRINTF("Join denied, retrying.\n");
1009  stats()->denies++;
1010  stats()->denies_today++;
1011  fRetry = true;
1012  } else {
1013  DEBUG_PRINTF("Unknown join response.\n");
1014  nrf_delay_ms(MICROCHIP_LONG_DELAY_MS);
1015  setstateL(COMM_LORA_JOINRPL);
1016  }
1017 
1018  if (fRetry) {
1019  if (++accept_retries >= LORAWAN_JOIN_RETRIES) {
1020  // If we can't rejoin "optimally" after having joined successfully,
1021  // try a full join just in case something happened on the service side
1022  // to cause state to have fundamentally changed
1023  loraInitEverCompleted = false;
1024  // If we aren't supposed to try the other, we need to just retry this one
1025  if (!LoRaWAN_mode_try_the_other_on_failure) {
1026  processstateL(COMM_LORA_RESETREQ);
1027  } else {
1028 
1029 #ifndef FONA
1030  // If there's no FONA, switch into LORA mode
1031  processstateL(COMM_LORA_LORAREQ);
1032 #else
1033  // If we're supposed to try LORA then LORAWAN and we found none of them, go to Cell.
1034  if (storage()->wan == WAN_AUTO) {
1035  setidlestateL();
1036  comm_select(COMM_FONA, "handoff from lora");
1037  } else {
1038  processstateL(COMM_LORA_LORAREQ);
1039  }
1040 #endif
1041 
1042  }
1043  } else {
1044  DEBUG_PRINTF("Attempt #%d of %d\n", accept_retries + 1, LORAWAN_JOIN_RETRIES);
1045  processstateL(COMM_LORA_RETRYJOIN);
1046  }
1047  }
1048  break;
1049  }
1050 
1054 
1055  case COMM_STATE_IDLE:
1056  setstateL(COMM_STATE_IDLE);
1057  return;
1058 
1059  case COMM_STATE_COMPLETE: {
1060  DEBUG_PRINTF("** wtf?? (%d) %s\n", fromLora.type, &fromLora.buffer[fromLora.args]);
1061  setstateL(COMM_STATE_IDLE);
1062  break;
1063  }
1064 
1065  case COMM_LORA_SLEEPRPL: {
1066  setstateL(COMM_STATE_IDLE);
1067  // Re-enable transmits now that we're safe to do so
1068  serial_transmit_enable(true);
1069  if (fTermAfterSleep) {
1070  fTermAfterSleep = false;
1071  lora_send("mac save");
1072  setstateL(COMM_LORA_SAVESTATERPL);
1073  break;
1074  }
1075  if (!sent_pending_outbound())
1076  restart_receive();
1077  break;
1078  }
1079 
1080  case COMM_LORA_GETSNRRPL: {
1081  uint16_t ms, i;
1082  toRelaySNR = atoi((char *)&fromLora.buffer[fromLora.args]);
1083  // Do a random delay so that we don't step on others who have similar WWAN visibility to the message
1084  ms = 5000 + io_get_random(5000);
1085 #define delay_interval_ms 100
1086  for (i=0; i<ms; i+=delay_interval_ms)
1087  nrf_delay_ms(delay_interval_ms);
1088  // Set idle (else the send_to_service will be blocked) and transmit it
1089  comm_cmdbuf_set_state(&fromLora, COMM_STATE_IDLE);
1090  if (send_to_service(toRelayBuffer, toRelayBufferLength, REPLY_NONE, SEND_1))
1091  DEBUG_PRINTF("RELAY %lu snr=%d\n", toRelayDevice, toRelaySNR);
1092  else {
1093  if (!sent_pending_outbound())
1094  restart_receive();
1095  }
1096  break;
1097  }
1098 
1099  case COMM_LORA_TXRPL1: {
1100  if (thisargisL("ok"))
1101  setstateL(COMM_LORA_TXRPL2);
1102  else
1103  setidlestateL();
1104  break;
1105  }
1106 
1107  case COMM_LORA_TXRPL2: {
1108  if (thisargisL("radio_tx_ok") || thisargisL("mac_tx_ok")) {
1109  setidlestateL();
1110  } else if (thisargisL("mac_rx")) {
1111  comm_cmdbuf_next_arg(&fromLora);
1112  // skip mac_rx
1113  thisargisL("*");
1114  comm_cmdbuf_next_arg(&fromLora);
1115  // skip port#
1116  thisargisL("*");
1117  char *rxdata = comm_cmdbuf_next_arg(&fromLora);
1118  if (rxdata[0] != '\0') {
1119  process_rx(rxdata);
1120  } else {
1121  // If we get an empty reply, try up to one more time
1122  // to see if it happens to come back to us on that xmit
1123  if (xmitReplyRetriesLeft > 0) {
1124  int saveRetriesLeft = xmitReplyRetriesLeft;
1125  setstateL(COMM_STATE_IDLE);
1126  send_ping_to_service(REPLY_NONE);
1127  // send_ping_to_service resets xmitReplyRetriesLeft, so
1128  // here we restore and decrement it.
1129  xmitReplyRetriesLeft = saveRetriesLeft - 1;
1130  } else {
1131  setidlestateL();
1132  }
1133  }
1134  } else {
1135  DEBUG_PRINTF("tx2 reply ?? %s\n", &fromLora.buffer[fromLora.args]);
1136  // Record this as an error because it means that something the caller
1137  // thought was transmitted silently got dropped.
1138  stats()->errors_lora++;
1139  setidlestateL();
1140  }
1141  if (loraInitEverCompleted && !awaitingTTServeReply)
1142  comm_oneshot_completed();
1143  break;
1144  }
1145 
1146  case COMM_LORA_RXRPL: {
1147  if (thisargisL("ok")) {
1148  // this is expected response from initiating the rcv,
1149  // so just reset the buffer and keep waiting for a message to come in
1150  setstateL(COMM_LORA_RXRPL);
1151  } else if (thisargisL("radio_err")) {
1152  // Re-enable serial output now that it's safe to do so
1153  serial_transmit_enable(true);
1154  // We're done waiting for reply
1155  if (awaitingTTServeReply)
1156  comm_oneshot_completed();
1157  // Cancel any pending ttserve reply; it didn't come.
1158  awaitingTTServeReply = false;
1159  // Expected from receive timeout of WDT seconds.
1160  // Note that if we were awaiting a ping reply, we need to stop now
1161  // because we didn't receive it.
1162  if (awaitingTTGateReply) {
1163  awaitingTTGateReply = false;
1164  // Try one more time, because waiting for a single reply can be fragile
1165  if (xmitReplyRetriesLeft > 0) {
1166  int saveRetriesLeft = xmitReplyRetriesLeft;
1167  setstateL(COMM_STATE_IDLE);
1168  DEBUG_PRINTF("Retrying ping to Lora gateway\n");
1169  send_ping_to_service(REPLY_TTGATE);
1170  // send_ping_to_service resets xmitReplyRetriesLeft, so
1171  // here we restore and decrement it.
1172  xmitReplyRetriesLeft = saveRetriesLeft - 1;
1173  } else {
1174  // Either try the other mode, or try this one again.
1175  if (LoRaWAN_mode_try_the_other_on_failure)
1176  processstateL(COMM_LORA_LORAWANREQ);
1177  else
1178  processstateL(COMM_LORA_LORAREQ);
1179  }
1180  } else {
1181  // If there's a pending outbound, transmit it (which will change state)
1182  // else restart the receive.
1183  if (!sent_pending_outbound())
1184  restart_receive();
1185  }
1186  } else if (thisargisL("busy")) {
1187  // Re-enable serial output now that it's safe to do so
1188  serial_transmit_enable(true);
1189  // This is not at all expected, but it means that we're
1190  // moving too quickly and we should try again.
1191  nrf_delay_ms(MICROCHIP_LONG_DELAY_MS);
1192  restart_receive();
1193  } else if (thisargisL("radio_rx")) {
1194  // Re-enable serial output now that it's safe to do so
1195  serial_transmit_enable(true);
1196  comm_cmdbuf_next_arg(&fromLora);
1197  process_rx((char *) &fromLora.buffer[fromLora.args]);
1198  // We're done waiting for reply
1199  comm_oneshot_completed();
1200  } else {
1201  // Re-enable serial output now that it's safe to do so
1202  serial_transmit_enable(true);
