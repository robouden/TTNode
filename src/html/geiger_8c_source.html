162 // Measure geiger values by analyzing the buckets maintained by poller
163 void s_geiger_measure(void *s) {
164 
165  // Exit without completing sensor if the values aren't yet available
166  if (!s_geiger_get_value(NULL, NULL, NULL, NULL)) {
167 
168  // This is defensive coding for bad hardware, but if for any reason we've
169  // been measuring for much longer than we should ever be measuring, "complete"
170  // the measurement so that we don't hang in this sensor measurement indefinitely.
171  if (sensor_op_mode() != OPMODE_MOBILE)
172  if (!ShouldSuppress(&geigerSensorMeasurementBegan, GEIGER_FIXED_INTEGRATION_SECONDS*4)) {
173  DEBUG_PRINTF("GEIGER polling failed!\n");
174  stats()->errors_geiger++;
175  sensor_measurement_completed(s);
176  }
177 
178  return;
179  }
180 
181  // Error handling
182 #define MAXTESTCPM 500
183 #if G0!=0
184  if (!geiger0IsAvailable)
185  stats()->errors_geiger++;
186  else if (value0IsReportable && reportableValue0 > MAXTESTCPM)
187  stats()->errors_geiger++;
188 #endif
189 #if G1!=0
190  if (!geiger1IsAvailable)
191  stats()->errors_geiger++;
192  else if (value1IsReportable && reportableValue1 > MAXTESTCPM)
193  stats()->errors_geiger++;
194 #endif
195 
196  // Debugging
197  if (value0IsReportable && value1IsReportable) {
198  DEBUG_PRINTF("GEIGER reported %ld %ld\n", reportableValue0, reportableValue1);
199  } else if (value0IsReportable) {
200  DEBUG_PRINTF("GEIGER reported %ld -\n", reportableValue0);
201  } else if (value1IsReportable) {
202  DEBUG_PRINTF("GEIGER reported - %ld\n", reportableValue1);
203  }
204 
205  // Done
206  sensor_measurement_completed(s);
207 
208 }

210 // Update the buckets where we accumulate the data from the counts.
211 // This *must* be called every GEIGER_BUCKET_SECONDS.
212 void geiger_bucket_update() {
213  int i;
214  uint32_t interruptCount0, interruptCount1;
215 
216  // Grab the values from the interrupt counters, and clear them out
217  interruptCount0 = geiger0InterruptCount;
218  geiger0InterruptCount = 0;
219  interruptCount1 = geiger1InterruptCount;
220  geiger1InterruptCount = 0;
221 
222  // Take note of when the geigers become available
223 #define PULSE_DEBOUNCE 5
224  geiger0InterruptCount_total += interruptCount0;
225  if (!geiger0IsAvailable && geiger0InterruptCount_total > PULSE_DEBOUNCE) {
226  geiger0IsAvailable = true;
227  if (debug(DBG_SENSOR))
228  DEBUG_PRINTF("Geiger #0 detected\n");
229  }
230  geiger1InterruptCount_total += interruptCount1;
231  if (!geiger1IsAvailable && geiger1InterruptCount_total > PULSE_DEBOUNCE) {
232  geiger1IsAvailable = true;
233  if (debug(DBG_SENSOR))
234  DEBUG_PRINTF("Geiger #1 detected\n");
235  }
236 
237  // Process geiger settling and filling
238  if (bucketsLeftToFillAfterPowerOn > 0) {
239  --bucketsLeftToFillAfterPowerOn;
240 
241  if (bucketsLeftDuringSettling > 0) {
242  --bucketsLeftDuringSettling;
243  if (debug(DBG_SENSOR))
244  DEBUG_PRINTF("CPM settling (+%d +%d)\n", interruptCount0, interruptCount1);
245  return;
246  }
247 
248  } else {
249 
250  // Don't generate anything reportable if we're skipping
251  if (g_geiger_skip(NULL))
252  return;
253 
254 
255  }
256 
257  // Insert the up-to-date interrupt counters into the bucket
258  if (++currentBucket >= GEIGER_INTEGRATION_BUCKETS)
259  currentBucket = 0;
260  bucket0[currentBucket] = interruptCount0;
261  bucket1[currentBucket] = interruptCount1;
262 
263  // Sum up the bucket contents
264  uint32_t cpm0 = 0;
265  uint32_t cpm0buckets = 0;
266  if (geiger0IsAvailable) {
267  value0IsReportable = true;
268  for (i = 0; i < GEIGER_INTEGRATION_BUCKETS; i++) {
269  if (bucket0[i] == INVALID_COUNT)
270  value0IsReportable = false;
271  else {
272  cpm0 += bucket0[i];
273  cpm0buckets++;
274  }
275  }
276  if (value0IsReportable)
277  value0EverReportable = true;
278  }
279  uint32_t cpm1 = 0;
280  uint32_t cpm1buckets = 0;
281  if (geiger1IsAvailable) {
282  value1IsReportable = true;
283  for (i = 0; i < GEIGER_INTEGRATION_BUCKETS; i++) {
284  if (bucket1[i] == INVALID_COUNT)
285  value1IsReportable = false;
286  else {
287  cpm1 += bucket1[i];
288  cpm1buckets++;
289  }
290  }
291  if (value1IsReportable)
292  value1EverReportable = true;
293  }
294 
295  // Compute compensated means
296  float bucketsPerMinute = (float) 60 / GEIGER_BUCKET_SECONDS;
297  float mean, compensated, divisor, secondsPerBucketPerMinute;
298  lastValue0 = 0;
299  if (cpm0buckets) {
300  secondsPerBucketPerMinute = ((float) cpm0buckets) / bucketsPerMinute;
301  mean = (float) cpm0 / secondsPerBucketPerMinute;
302  divisor = 1 - (mean * 1.8833e-6);
303  if (divisor)
304  compensated = mean / divisor;
305  else
306  compensated = 0.0;
307  lastValue0 = (uint32_t) compensated;
308  if (value0IsReportable) {
309  reportableValue0 = lastValue0;
310  valuesHaveBeenUpdated = true;
311  }
312  }
313  lastValue1 = 0;
314  if (cpm1buckets) {
315  secondsPerBucketPerMinute = ((float) cpm1buckets) / bucketsPerMinute;
316  mean = (float) cpm1 / secondsPerBucketPerMinute;
317  divisor = 1 - (mean * 1.8833e-6);
318  if (divisor)
319  compensated = mean / divisor;
320  else
321  compensated = 0.0;
322  lastValue1 = (uint32_t) compensated;
323  if (value1IsReportable) {
324  reportableValue1 = lastValue1;
325  valuesHaveBeenUpdated = true;
326  }
327  }
328 
329  // Done
330  int totalIterations = geiger_integration_seconds()/GEIGER_BUCKET_SECONDS;
331  int percentComplete = (int) (((float) (totalIterations - bucketsLeftToFillAfterPowerOn) / totalIterations) * 100);
332  if (percentComplete < 0) percentComplete = 0;
333  if (geiger0IsAvailable && geiger1IsAvailable)
334  DEBUG_PRINTF("CPM %d %d (%d %d) %d%%\n", lastValue0, lastValue1, interruptCount0, interruptCount1, percentComplete);
335  else if (geiger0IsAvailable)
336  DEBUG_PRINTF("CPM %d - (%d %d) %d%%\n", lastValue0, interruptCount0, interruptCount1, percentComplete);
337  else if (geiger1IsAvailable)
338  DEBUG_PRINTF("CPM - %d (%d %d) %d%%\n", lastValue1, interruptCount0, interruptCount1, percentComplete);
339 
340 }

378 // Turn on geiger if it's not on
379 void geiger_power_on() {
380 
381  // Turn on the power. Note that we do our own
382  // power management outside the sensor package so that
383  // we can control whether or not we leave it turned on
384  // across many iterations.
385  if (!geigerPowerOn) {
386  int i;
387 
388  // Turn on the power
389 #ifdef POWER_PIN_GEIGER
390  gpio_power_set(POWER_PIN_GEIGER, true);
391  DEBUG_PRINTF("s-geiger power ON\n");
392 #endif
393  geigerPowerOn = true;
394 
395  // Init the buckets
396  currentBucket = 0;
397  for (i = 0; i < GEIGER_INTEGRATION_BUCKETS; i++) {
398  bucket0[i] = INVALID_COUNT;
399  bucket1[i] = INVALID_COUNT;
400  }
401 
402  // After powering on, allow settling for stabilization. When we're in mobile mode,
403  // power-on only happens up-front and the geiger stays running continuously.
404  bucketsLeftDuringSettling = GEIGER_SETTLING_SECONDS/GEIGER_BUCKET_SECONDS;
405  bucketsLeftToFillAfterPowerOn = geiger_integration_seconds()/GEIGER_BUCKET_SECONDS + bucketsLeftDuringSettling;
406 
407  // Init the current values
408  geiger0InterruptCount = 0;
409  geiger1InterruptCount = 0;
410 
411  // Clear whether or not the values are reportable
412  s_geiger_clear_measurement();
413 
414  }
