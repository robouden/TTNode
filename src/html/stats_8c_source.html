19 // Update uptime stats
20 void stats_update() {
21  if (!ShouldSuppress(&st.last_minute, 60)) {
22  st.uptime_minutes++;
23  if (st.uptime_minutes >= 60) {
24  st.uptime_minutes = 0;
25  st.uptime_hours++;
26  if (st.uptime_hours >= 24) {
27  st.uptime_hours = 0;
28  st.uptime_days++;
29  // Reset and update daily stats
30  if (st.uptime_days > 1) {
31  st.transmitted_fullday = st.transmitted_today;
32  st.max_transmitted_fullday = st.max_transmitted_today;
33  st.received_fullday = st.received_today;
34  st.messages_fullday = st.messages_today;
35  st.joins_fullday = st.joins_today;
36  st.denies_fullday = st.denies_today;
37  }
38  st.transmitted_today = 0;
39  st.max_transmitted_today = 0;
40  st.received_today = 0;
41  st.messages_today = 0;
42  st.joins_today = 0;
43  st.denies_today = 0;
44  // Restart the device when appropriate
45  if (storage()->restart_days != 0 && st.uptime_days >= storage()->restart_days) {
46  storage()->uptime_days += st.uptime_days;
47  storage_save(true);
48  io_request_restart();
49  }
50  }
51  }
52  }
53 }
