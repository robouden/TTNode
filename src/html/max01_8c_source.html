90 // Callback when TWI data has been read, or timeout
91 void max01_callback(ret_code_t result, twi_context_t *t) {
92  int16_t icombined;
93  uint16_t ucombined;
94 
95  // If error, flag that this I/O has been completed.
96  if (!twi_completed(t)) {
97  stats()->errors_max01++;
98  battery_set_soc_to_unknown();
99  sensor_measurement_completed(t->sensor);
100  return;
101  }
102 
103  // Compute Device name
104  bool isMax01 = (regDEVNAME[1] & 0x0f) == 0x01;
105  // Compute voltage by converting mV to V
106  ucombined = regVCELL[1] | (regVCELL[2] << 8);
107  float voltage = (float) ucombined * 0.078125 / 1000;
108  // Compute current by converting to mV/Ohm (mA)
109  icombined = regCURRENT[1] | (regCURRENT[2] << 8);
110  float current = (float) icombined * (0.0015625/0.01);
111 #ifdef scv1
112  current = -current;
113 #endif
114  // Compute SOC as %
115  ucombined = regREPSOC[1] | (regREPSOC[2] << 8);
116  float soc = (float) ucombined / 256;
117  // Compute Temp as degrees C
118  icombined = regTEMP[1] | (regTEMP[2] << 8);
119  float temp = (float) icombined / 256;
120  // Compute Capacity as mAh
121  ucombined = regREPCAP[1] | (regREPCAP[2] << 8);
122  float cap = (float) ucombined * 0.005/0.01;
123  // Compute Time To Empty in seconds
124  ucombined = regTTE[1] | (regTTE[2] << 8);
125  float tte = (float) ucombined * 5.625;
126  // Compute Time To Full in seconds
127  ucombined = regTTF[1] | (regTTF[2] << 8);
128  float ttf = (float) ucombined * 5.625;
129  // Extract misc things
130  uint16_t status = regSTATUS[1] | (regSTATUS[2] << 8);
131  uint16_t age = regAGE[1] | (regAGE[2] << 8);
132  uint16_t capacity = regCAPACITY[1] | (regCAPACITY[2] << 8);
133  uint16_t avcell = regAVCELL[1] | (regAVCELL[2] << 8);
134  uint16_t agef = regAGE_FORECAST[1] | (regAGE_FORECAST[2] << 8);
135  uint16_t vbat = regVBAT[1] | (regVBAT[2] << 8);
136 
137  // Set the string for stats
138  char buffer[128];
139  sprintf(buffer, "%.1f/%.1f/%.1f/%.1f/%.1f/%.1f/%.1f/%04x/%d/%d/%d/%d/%d",
140  voltage, current, soc, temp, cap, tte, ttf, status, age, capacity, avcell, agef, vbat);
141  strlcpy(stats()->battery, buffer, sizeof(stats()->battery)-1);
142 
143  // Store it into the bin IF AND ONLY IF nobody is currently sucking power on the UART if in oneshot mode
144  if (num_samples < PWR_SAMPLE_BINS) {
145  sampled_voltage += voltage;
146  sampled_soc += soc;
147  if (!comm_oneshot_currently_enabled() || gpio_current_uart() == UART_NONE) {
148  sampled_current += current;
149  num_current_samples++;
150  }
151  num_samples++;
152  first_sample = false;
153  }
154 
155  // If we're not done, request another measurement
156  if (num_samples < PWR_SAMPLE_BINS) {
157 
158  // We should never hit this, but this is a protective measure to ensure that
159  // we don't just sit here for too long and hold up other sensors from operating
160  if (num_polls++ < (PWR_SAMPLE_BINS*25))
161  measure_on_next_poll = true;
162  else {
163  DEBUG_PRINTF("MAX01 Aborting because of unreleased UART\n", gpio_uart_name(gpio_current_uart()));
164  battery_set_soc_to_unknown();
165  sensor_measurement_completed(t->sensor);
166  }
167 
168  // Ensure that we've got a good device
169  if (!isMax01) {
170  battery_set_soc_to_unknown();
171  stats()->errors_max01++;
172  DEBUG_PRINTF("MAX01 not detected! (0x%04x)\n", (regDEVNAME[2] << 8) | regDEVNAME[1]);
173  }
174 
175  // Debug
176 
177  if (debug(DBG_SENSOR_MAX))
178  DEBUG_PRINTF("MAX01 %.3fmA %.3fV %.3f%%\n", current, voltage, soc);
179  if (debug(DBG_SENSOR_SUPERMAX)) {
180  DEBUG_PRINTF("temp:%.3fC cap:%.3fmAh tte:%.3fs ttf:%.3fs\n", temp, cap, tte, ttf);
181  DEBUG_PRINTF("status:0x%04x age:%d C:%d avcell:%d agef:%d vbat:%d\n", status, age, capacity, avcell, agef, vbat);
182  }
183 
184  } else {
185 
186  // Compute the reported measurements
187  reported_voltage = sampled_voltage / num_samples;
188  reported_soc = sampled_soc / num_samples;
189  if (num_current_samples == 0)
190  reported_current = 0;
191  else
192  reported_current = sampled_current / num_current_samples;
193 
194  // When debugging current, just poll continuously
195 #ifdef CURRENTDEBUG
196 
197  // Start over
198  s_max01_clear_measurement();
199  measure_on_next_poll = true;
200 
201 #else
202  // Done
203  reported = ever_reported = true;
204 
205  // Tell the sensor package that we retrieved an SOC value, and what it is
206 #if 0
207  battery_set_soc(reported_soc);
208 #else
209  battery_set_soc(battery_soc_from_voltage(reported_voltage));
210 #endif
211 
212  // Flag that this I/O has been completed.
213  sensor_measurement_completed(t->sensor);
214 
215 #endif // !CURRENTDEBUG
216 
217  if (debug(DBG_SENSOR))
218  DEBUG_PRINTF("BAT %.2fV %.0f%% %.0fmA\n", reported_voltage, reported_soc, reported_current);
219 
220  }
221 
222 }

241 void s_max01_measure(void *s) {
242  static app_twi_transfer_t const transfers[] = {
243  APP_TWI_WRITE(MAX17201_I2C_ADDRESS, &regSTATUS[0], addrLen, APP_TWI_NO_STOP),
244  APP_TWI_READ(MAX17201_I2C_ADDRESS, &regSTATUS[1], dataLen, 0),
245  APP_TWI_WRITE(MAX17201_I2C_ADDRESS, &regREPCAP[0], addrLen, APP_TWI_NO_STOP),
246  APP_TWI_READ(MAX17201_I2C_ADDRESS, &regREPCAP[1], dataLen, 0),
247  APP_TWI_WRITE(MAX17201_I2C_ADDRESS, &regREPSOC[0], addrLen, APP_TWI_NO_STOP),
248  APP_TWI_READ(MAX17201_I2C_ADDRESS, &regREPSOC[1], dataLen, 0),
249  APP_TWI_WRITE(MAX17201_I2C_ADDRESS, &regAGE[0], addrLen, APP_TWI_NO_STOP),
250  APP_TWI_READ(MAX17201_I2C_ADDRESS, &regAGE[1], dataLen, 0),
251  APP_TWI_WRITE(MAX17201_I2C_ADDRESS, &regTEMP[0], addrLen, APP_TWI_NO_STOP),
252  APP_TWI_READ(MAX17201_I2C_ADDRESS, &regTEMP[1], dataLen, 0),
253  APP_TWI_WRITE(MAX17201_I2C_ADDRESS, &regVCELL[0], addrLen, APP_TWI_NO_STOP),
254  APP_TWI_READ(MAX17201_I2C_ADDRESS, &regVCELL[1], dataLen, 0),
255  APP_TWI_WRITE(MAX17201_I2C_ADDRESS, &regCURRENT[0], addrLen, APP_TWI_NO_STOP),
256  APP_TWI_READ(MAX17201_I2C_ADDRESS, &regCURRENT[1], dataLen, 0),
257  APP_TWI_WRITE(MAX17201_I2C_ADDRESS, &regCAPACITY[0], addrLen, APP_TWI_NO_STOP),
258  APP_TWI_READ(MAX17201_I2C_ADDRESS, &regCAPACITY[1], dataLen, 0),
259  APP_TWI_WRITE(MAX17201_I2C_ADDRESS, &regTTE[0], addrLen, APP_TWI_NO_STOP),
260  APP_TWI_READ(MAX17201_I2C_ADDRESS, &regTTE[1], dataLen, 0),
261  APP_TWI_WRITE(MAX17201_I2C_ADDRESS, &regAVCELL[0], addrLen, APP_TWI_NO_STOP),
262  APP_TWI_READ(MAX17201_I2C_ADDRESS, &regAVCELL[1], dataLen, 0),
263  APP_TWI_WRITE(MAX17201_I2C_ADDRESS, &regTTF[0], addrLen, APP_TWI_NO_STOP),
264  APP_TWI_READ(MAX17201_I2C_ADDRESS, &regTTF[1], dataLen, 0),
265  APP_TWI_WRITE(MAX17201_I2C_ADDRESS, &regDEVNAME[0], addrLen, APP_TWI_NO_STOP),
266  APP_TWI_READ(MAX17201_I2C_ADDRESS, &regDEVNAME[1], dataLen, 0),
267  APP_TWI_WRITE(MAX17201_I2C_ADDRESS, &regAGE_FORECAST[0], addrLen, APP_TWI_NO_STOP),
268  APP_TWI_READ(MAX17201_I2C_ADDRESS, &regAGE_FORECAST[1], dataLen, 0),
269  APP_TWI_WRITE(MAX17201_I2C_ADDRESS, &regVBAT[0], addrLen, APP_TWI_NO_STOP),
270  APP_TWI_READ(MAX17201_I2C_ADDRESS, &regVBAT[1], dataLen, 0),
271  };
272  static app_twi_transaction_t const transaction = {
273  .callback = twi_callback,
274  .p_user_data = "MAX01",
275  .p_transfers = transfers,
276  .number_of_transfers = sizeof(transfers) / sizeof(transfers[0])
277  };
278  if (!twi_schedule(s, max01_callback, &transaction)) {
279  stats()->errors_max01++;
280  battery_set_soc_to_unknown();
281  sensor_unconfigure(s);
282  }
283 }
