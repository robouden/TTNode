83 // Set the operating mode
84 bool sensor_set_op_mode(uint16_t op_mode) {
85 
86  // If we're setting it back to normal, but we're in BURN mode, just interpret it as returning to burn
87 #ifdef BURN
88  if (op_mode == OPMODE_NORMAL)
89  op_mode = OPMODE_TEST_BURN;
90 #endif
91 
92  // Do special work if we're switching into mobile mode
93  if (op_mode == OPMODE_MOBILE && operating_mode != op_mode) {
94 
95  if (storage()->gps_latitude != 0 || storage()->gps_longitude != 0) {
96  DEBUG_PRINTF("Mobile mode doesn't make sense with static GPS configuration.\n");
97  return false;
98  }
99 
100  // Tell others that we are beginning a new "drive" session
101  mobile_session++;
102 
103  // Tell the GPS module to improve its location
104  comm_gps_update();
105 
106  // Accelerate enabling the mobile modules
107  sensor_group_schedule_now(GPS_SENSOR_GROUP);
108 
109  // Initiate a service update, so that we send a new stamp value
110  // to the service so as to initiate a new "drive"
111  comm_initiate_service_update(true);
112 
113  }
114 
115  // Set the mode
116  operating_mode = op_mode;
117  return true;
118 
119 }

128 // Get the operating mode. Note that this is an extremely low level routine, so
129 // don't modify it to call anything that might potentially go recursive.
130 uint16_t sensor_op_mode() {
131 
132  // Make sure we act in normal mode if battery is truly dead, no matter what mode
133  // we are actually in. This blocks aggressive comms and sensor measurement
134  // behavior in many of the test modes.
135  if (battery_soc() < 10.0)
136  return OPMODE_NORMAL;
137 
138  // Allow it to be temporarily overridden
139  if (temporary_op_mode_duration_seconds != 0) {
140  if (ShouldSuppress(&temporary_op_mode_set_at, temporary_op_mode_duration_seconds))
141  return temporary_op_mode;
142  temporary_op_mode_duration_seconds = 0;
143  }
144 
145  // Return the actual mode
146  return(operating_mode);
147 
148 }

284 // Mark all sensors within an entire group as having been completed
285 bool sensor_group_completed(group_t *g) {
286  sensor_t **sp, *s;
287  bool somethingCompleted = false;
288  // Note that we support calling of sensor routines directly in absence
289  // of the sensor packaging having been involved, in which case this
290  // will be null
291  if (g == NULL)
292  return false;
293  // Turn off polling
294  g->state.is_polling_valid = false;
295  // Mark all sensors as completed
296  for (sp = &g->sensors[0]; (s = *sp) != END_OF_LIST; sp++)
297  if (s->state.is_configured && !s->state.is_completed) {
298  s->state.is_completed = true;
299  s->state.is_polling_valid = false;
300  somethingCompleted = true;
301  }
302  if (somethingCompleted && debug(DBG_SENSOR_MAX))
303  DEBUG_PRINTF("%s is completed.\n", g->name);
304  return (somethingCompleted);
305 }

407 // Get a group's repeat minutes, adjusted for debugging
408 uint16_t group_repeat_seconds(group_t *g) {
409  uint16_t bat_status = battery_status();
410  uint16_t repeat_seconds = 0;
411  repeat_t *r;
412 
413  // If overridden, use it
414  if (g->state.repeat_seconds_override != 0) {
415  return g->state.repeat_seconds_override;
416  }
417 
418  // Loop, finding the appropriate battery status
419  for (r = g->repeat;; r++) {
420  if ((bat_status & r->active_battery_status) != 0) {
421  repeat_seconds = r->repeat_seconds;
422  break;
423  }
424  }
425 
426  // Bug check
427  if (repeat_seconds == 0)
428  DEBUG_PRINTF("%s repeat seconds not found for %s\n", g->name, battery_status_name());
429 
430  // If we're testing, just double it
431  if (bat_status == BAT_TEST)
432  return (repeat_seconds/2);
433 
434  return(repeat_seconds);
435 }

459  group_t **gp, *g;
460  sensor_t **sp, *s;
461  uint32_t seconds_since_boot = get_seconds_since_boot();
462  char buffp[512];
463 
464  if (!fInit) {

537  strcat(buffp, "T");

539 
540  if (fVerbose)
541  DEBUG_PRINTF("%s %s\n", g->name, g->state.is_processing ? (g->state.is_settling ? "now settling" : "now sampling") : buff);

543  strcat(buffp, "] ");

570  strcat(buffp, fUploadNeeded ? "u" : "c");
571  }
572  }
573  strcat(buffp, ") ");
574  }
575 
576  }
577 
578  }
579  if (!fVerbose)
580  DEBUG_PRINTF("%s\n", buffp);
581

590 // Poll, advancing the state machine
591 void sensor_poll() {
592  static int inside_poll = 0;
593  bool groups_currently_active;
594  int pending;
595  group_t **gp, *g;
596  sensor_t **sp, *s;
597 
598  // Exit if we haven't yet initialized GPS, which is a big signal that we're not yet ready to proceed,
599  // except for the case of UGPS when we need sensor processing to acquire GPS
600 #ifndef UGPS
601  uint16_t status = comm_gps_get_value(NULL, NULL, NULL);
602  if (status != GPS_NOT_CONFIGURED && status != GPS_LOCATION_ABORTED)
603  if (status != GPS_LOCATION_FULL && status != GPS_LOCATION_PARTIAL)
604  return;
605 #endif
606 
607  // Initialize if we haven't yet done so
608  if (!fInit) {
609  sensor_init();
610  fInit = true;
611  }
612 
613  // Debug sensors
614  if (debug(DBG_SENSOR_POLL)) {
615  sensor_show_values(false);
616  }
617 
618  // Debug TWI
619 #ifdef TWIX
620  twi_status_check(false);
621 #endif
622 
623  // Exit if we're already inside the poller. This DOES happen if one of the handlers (such as an
624  // init handler) takes an incredibly long time because of, say, a retry loop.
625  if (inside_poll++ != 0) {
626  inside_poll--;
627  DEBUG_PRINTF("sensor_poll REENTRANCY!\n");
628  return;
629  }
630 
631  if (debug(DBG_SENSOR_SUPERDUPERMAX))
632  DEBUG_PRINTF("sensor_poll enter\n");
633 
634  // Loop over all configured sensors in all configured groups
635  groups_currently_active = 0;
636 
637  for (gp = &sensor_groups[0]; (g = *gp) != END_OF_LIST; gp++) {
638 
639  // If not configured, skip this group
640  if (!g->state.is_configured)
641  continue;
642 
643  // If test mode and not being tested, bail
644  if (sensor_op_mode() == OPMODE_TEST_SENSOR && !g->state.is_being_tested)
645  continue;
646 
647  // Are we completely idle?
648  if (!g->state.is_processing && !g->state.is_settling) {
649 
650  if (debug(DBG_SENSOR_SUPERDUPERMAX))
651  DEBUG_PRINTF("%s !processing !settling\n", g->name);
652 
653  // If we've requested test mode, don't schedule anything new
654  if (fTestModeRequested)
655  continue;
656 
657  // Skip if this group doesn't need to be processed right now
658  if (g->skip_handler != NO_HANDLER && sensor_op_mode() != OPMODE_TEST_SENSOR)
659  if (g->skip_handler(g)) {
660  if (debug(DBG_SENSOR_SUPERDUPERMAX))
661  DEBUG_PRINTF("Skipping %s at its request.\n", g->name);
662  continue;
663  }
664 
665  // If ALL of the sensors in this group already have pending measurements,
666  // skip the group because it's senseless to keep measuring.
667  bool fSkipGroup = true;
668  int Sensors = 0;
669  for (sp = &(*gp)->sensors[0]; (s = *sp) != END_OF_LIST; sp++) {
670  if (s->state.is_configured) {
671  Sensors++;
672  if (s->upload_needed == NO_HANDLER || !s->upload_needed(s)) {
673  fSkipGroup = false;
674  break;
675  }
676  }
677  }
678  if (Sensors == 0)
679  g->state.is_configured = false;
680  if (fSkipGroup && sensor_op_mode() != OPMODE_TEST_SENSOR) {
681  if (debug(DBG_SENSOR_SUPERDUPERMAX)) {
682  if (Sensors == 0)
683  DEBUG_PRINTF("Skipping %s because no sensors are found.\n", g->name);
684  else
685  DEBUG_PRINTF("Skipping %s because all its sensors' uploads are pending.\n", g->name);
686  }
687  continue;
688  }
689 
690  // If this sensor group can only be run when nothing else is running, skip if busy
691  if (g->exclusive) {
692  if (sensor_group_busy()) {
693  if (debug(DBG_SENSOR_SUPERDUPERMAX))
694  DEBUG_PRINTF("Skipping %s because something else is being processed.\n", g->name);
695  continue;
696  }
697  } else {
698  if (sensor_group_any_exclusive_busy()) {
699  if (debug(DBG_SENSOR_SUPERDUPERMAX))
700  DEBUG_PRINTF("Skipping %s because exclusive is being processed.\n", g->name);
701  continue;
702  }
703  }
704 
705  // If this sensor group is a particular power hog and needs to be run
706  // only when other exclusives aren't powered on, skip the group if anyone else
707  // is currently powered on.
708  if (g->power_exclusive && sensor_group_any_exclusive_powered_on()) {
709  if (debug(DBG_SENSOR_SUPERDUPERMAX))
710  DEBUG_PRINTF("Skipping %s because something else is powered on.\n", g->name);
711  continue;
712  }
713 
714  // If this sensor group requires TWI and can only run when other exclusives
715  // aren't using TWI, skip the group if anyone else is currently using TWI.
716  if (g->twi_exclusive && sensor_group_any_exclusive_twi_on()) {
717  if (debug(DBG_SENSOR_SUPERDUPERMAX))
718  DEBUG_PRINTF("Skipping %s because something else is using TWI.\n", g->name);
719  continue;
720  }
721 
722  // If this sensor group requires a uart, but the UART is busy, skip the group
723  if (g->uart_required != UART_NONE && gpio_current_uart() != UART_NONE) {
724  if (debug(DBG_SENSOR_SUPERDUPERMAX))
725  DEBUG_PRINTF("Skipping %s because the required UART is busy.\n", g->name);
726  continue;
727  }
728 
729  // If this sensor group requests a uart (but is allowed to run without it if
730  // it CAN'T be granted), but the UART is busy, skip the group
731  if (comm_uart_switching_allowed() && g->uart_requested != UART_NONE && gpio_current_uart() != UART_NONE) {
732  if (debug(DBG_SENSOR_SUPERDUPERMAX))
733  DEBUG_PRINTF("Skipping %s because the requested UART is busy.\n", g->name);
734  continue;
735  }
736 
737  // If we're not in the right battery status, skip it
738  if ((battery_status() & g->active_battery_status) == 0) {
739  if (debug(DBG_SENSOR_SUPERDUPERMAX))
740  DEBUG_PRINTF("Skipping %s because 0x%04x doesn't map to %s\n", g->name, g->active_battery_status, battery_status_name());
741  continue;
742  }
743 
744  // If we're not in the right comms mode, skip it
745  if ((comm_mode() & g->active_comm_mode) == 0) {
746  if (debug(DBG_SENSOR_SUPERDUPERMAX))
747  DEBUG_PRINTF("Skipping %s because %04x doesn't map to active comm mode (%04x).\n", g->name, g->active_comm_mode, comm_mode());
748  continue;
749  }
750 
751  // If we're in the repeat idle period for this group, just go to next group
752  if (sensor_op_mode() != OPMODE_TEST_SENSOR)
753  if (ShouldSuppressConsistently(&g->state.last_repeated, group_repeat_seconds(g)))
754  continue;
755 
756  // Initialize sensor state and refresh configuration state
757  for (sp = &(*gp)->sensors[0]; (s = *sp) != END_OF_LIST; sp++) {
758  if (s->state.is_configured) {
759  s->state.is_settling = false;
760  s->state.is_processing = false;
761  s->state.is_completed = false;
762  }
763  }
764 
765  // Begin processing
766  g->state.is_processing = true;
767 
768  if (debug(DBG_SENSOR_SUPERDUPERMAX))
769  DEBUG_PRINTF("%s processing\n", g->name);
770 
771  // Power ON the module
772  if (g->power_set != NO_HANDLER) {
773  g->power_set(g->power_set_parameter, true);
774  g->state.is_powered_on = true;
775  // Delay a bit before proceeding to do anything at all
776  nrf_delay_ms(MAX_NRF_DELAY_MS);
777  if (debug(DBG_SENSOR_MAX))
778  DEBUG_PRINTF("%s power ON\n", g->name);
779  }
780 
781  // Select the UART if one is required or requested
782  if (g->uart_required != UART_NONE)
783  gpio_uart_select(g->uart_required);
784  if (comm_uart_switching_allowed() && g->uart_requested != UART_NONE)
785  gpio_uart_select(g->uart_requested);
786 
787  // Call the sensor power-on init functions
788  for (sp = &g->sensors[0]; (s = *sp) != END_OF_LIST; sp++) {
789 
790  // If not configured, don't bother initializing anything else
791  if (!s->state.is_configured)
792  continue;
793 
794  // If there's an init handler to be called after power is turned on, call it
795  if (s->init_power != NO_HANDLER) {
796  if (!s->init_power(s, s->init_parameter))
797  s->state.init_failures++;
798  else
799  s->state.init_failures = 0;
800  }
801 
802  }
803 
804  // Begin the settling period
805  g->state.last_settled = get_seconds_since_boot();
806  g->state.is_settling = true;
807 
808  if (debug(DBG_SENSOR_SUPERDUPERMAX))
809  DEBUG_PRINTF("%s settling\n", g->name);
810 
811  if (g->settling_seconds != 0 && debug(DBG_SENSOR_MAX))
812  DEBUG_PRINTF("Begin %s settling for %ds\n", g->name, g->settling_seconds);
813 
814  // Start the group app timer when the power is applied, because a core part of the
815  // settling period for certain TWI devices (i.e. GPS) is that you need to
816  // "warm them up" by pulling data out of them on a continuous basis.
817  if (g->poll_handler != NO_HANDLER && !g->poll_continuously && g->poll_during_settling) {
818  if (debug(DBG_SENSOR_MAX))
819  DEBUG_PRINTF("%s timer ON\n", g->name);
820  app_timer_start(g->state.group_timer.timer_id, APP_TIMER_TICKS(g->poll_repeat_milliseconds, APP_TIMER_PRESCALER), g);
821  // Release the poller so that it's ok to proceed
822  g->state.is_polling_valid = true;
823  }
824 
825  // Do the same for sensor timers
826  for (sp = &(*gp)->sensors[0]; (s = *sp) != END_OF_LIST; sp++) {
827 
828  // Skip unconfigured sensors
829  if (!s->state.is_configured)
830  continue;
831 
832  // Enable the poller
833  if (s->poll_handler != NO_HANDLER && !s->poll_continuously && s->poll_during_settling) {
834  if (debug(DBG_SENSOR_MAX))
835  DEBUG_PRINTF("%s timer ON\n", s->name);
836  app_timer_start(s->state.sensor_timer.timer_id, APP_TIMER_TICKS(s->poll_repeat_milliseconds, APP_TIMER_PRESCALER), s);
837  // Release the poller so that it's ok to proceed
838  s->state.is_polling_valid = true;
839  }
840 
841  }
842 
843  }
844 
845  // Are we in the settling period?
846  if (g->state.is_processing && g->state.is_settling) {
847  // groups_currently_active++;
848 
849  if (debug(DBG_SENSOR_SUPERDUPERMAX))
850  DEBUG_PRINTF("%s processing settling\n", g->name);
851 
852  // If we're in the settling idle period for this group, just go to the next group
853  if (g->settling_seconds != 0)
854  if (ShouldSuppress(&g->state.last_settled, g->settling_seconds))
855  continue;
856 
857  // Stop the settling period.
858  g->state.is_settling = false;
859 
860  // If there's a handler to be called after group settling, call it
861  if (g->done_settling != NO_HANDLER)
862  g->done_settling();
863  for (sp = &(*gp)->sensors[0]; (s = *sp) != END_OF_LIST; sp++)
864  if (s->state.is_configured && s->done_group_settling != NO_HANDLER)
865  s->done_group_settling();
866 
867  // Start the app timer when settling is over, if that's what was requested
868  if (g->poll_handler != NO_HANDLER && !g->poll_continuously && !g->poll_during_settling) {
869  if (debug(DBG_SENSOR_MAX))
870  DEBUG_PRINTF("%s timer ON\n", g->name);
871  app_timer_start(g->state.group_timer.timer_id, APP_TIMER_TICKS(g->poll_repeat_milliseconds, APP_TIMER_PRESCALER), g);
872  // Release the poller so that it's ok to proceed
873  g->state.is_polling_valid = true;
874  }
875 
876  // Do the same for sensor timers
877  for (sp = &(*gp)->sensors[0]; (s = *sp) != END_OF_LIST; sp++) {
878 
879  // Skip unconfigured sensors
880  if (!s->state.is_configured)
881  continue;
882 
883  // Enable the poller
884  if (s->poll_handler != NO_HANDLER && !s->poll_continuously && !s->poll_during_settling) {
885  if (debug(DBG_SENSOR_MAX))
886  DEBUG_PRINTF("%s timer ON\n", s->name);
887  app_timer_start(s->state.sensor_timer.timer_id, APP_TIMER_TICKS(s->poll_repeat_milliseconds, APP_TIMER_PRESCALER), s);
888  // Release the poller so that it's ok to proceed
889  s->state.is_polling_valid = true;
890  }
891 
892  }
893 
894  }
895 
896  // Is it time to do some processing?
897  if (g->state.is_processing && !g->state.is_settling) {
898  // groups_currently_active++;
899 
900  if (debug(DBG_SENSOR_SUPERDUPERMAX))
901  DEBUG_PRINTF("%s processing !settling\n", g->name);
902 
903  // Loop over all sensors in this group, looking for work to do
904  for (sp = &(*gp)->sensors[0]; (s = *sp) != END_OF_LIST; sp++) {
905 
906  // Skip unconfigured sensors
907  if (!s->state.is_configured)
908  continue;
909 
910  // If test mode and not being tested, bail
911  if (sensor_op_mode() == OPMODE_TEST_SENSOR && !s->state.is_being_tested)
912  continue;
913 
914  // Is this a candidate for initiating work?
915  if (!s->state.is_processing && !s->state.is_completed) {
916 
917  // Begin processing
918  s->state.is_processing = true;
919 
920  if (s->state.is_being_tested)
921  DEBUG_PRINTF("Now testing %s\n", s->name);
922 
923  // Begin the settling period
924  s->state.last_settled = get_seconds_since_boot();
925  s->state.is_settling = true;
926 
927  if (s->settling_seconds != 0 && debug(DBG_SENSOR))
928  DEBUG_PRINTF("Begin %s %s for %ds\n", s->name, g->poll_handler == NO_HANDLER ? "settling" : "sampling", s->settling_seconds);
929 
930  }
931 
932  // Are we in the settling period?
933  if (s->state.is_processing && s->state.is_settling) {
934 
935  // If we're in the settling idle period for this sensor, stop processing sensors
936  if (s->settling_seconds != 0)
937  if (ShouldSuppress(&s->state.last_settled, s->settling_seconds))
938  break;
939 
940  // Stop the settling period.
941  s->state.is_settling = false;
942 
943  // If there's a handler to be called after settling, call it
944  if (s->done_settling != NO_HANDLER)
945  s->done_settling();
946 
947  if (s->measure != NO_HANDLER && debug(DBG_SENSOR))
948  DEBUG_PRINTF("Measuring %s\n", s->name);
949 
950  }
951 
952  // Keep measuring the sensor until it reports that it has "completed"
953  if (s->state.is_processing && !s->state.is_completed && !s->state.is_settling) {
954 
955  // Initiate the measurement.
956  if (s->measure != NO_HANDLER)
957  s->measure(s);
958 
959  }
960 
961  // Is this one processing? If so, we don't want to move beyond it
962  if (s->state.is_processing && !s->state.is_completed)
963  break;
964 
965  } // loop over sensors
966 
967 
968  } // if work to do for a given sensor group
969 
970  // Count the number of sensors with work left to do
971  for (pending = 0, sp = &(*gp)->sensors[0]; (s = *sp) != END_OF_LIST; sp++)
972  if (s->state.is_configured)
973  if (!s->state.is_completed)
974  if (sensor_op_mode() != OPMODE_TEST_SENSOR || s->state.is_being_tested)
975  pending++;
976 
977  if (debug(DBG_SENSOR_SUPERDUPERMAX))
978  DEBUG_PRINTF("%s still not completed\n", g->name);
979 
980  // If none of the sensors has any work pending, we're done with this group.
981  if (pending == 0) {
982 
983  // Stop the app timer if one had been requested
984  if (g->poll_handler != NO_HANDLER && !g->poll_continuously) {
985  app_timer_stop(g->state.group_timer.timer_id);
986  if (debug(DBG_SENSOR_MAX))
987  DEBUG_PRINTF("%s timer OFF\n", g->name);
988  // Stop the poller so that it doesn't do any more processing
989  g->state.is_polling_valid = false;
990  }
991 
992  for (sp = &g->sensors[0]; (s = *sp) != END_OF_LIST; sp++) {
993 
994  // If not configured, don't bother initializing anything else
995  if (!s->state.is_configured)
996  continue;
997 
998  // Do the same for sensor pollers
999  if (s->poll_handler != NO_HANDLER && !s->poll_continuously) {
1000  app_timer_stop(s->state.sensor_timer.timer_id);
1001  if (debug(DBG_SENSOR_MAX))
1002  DEBUG_PRINTF("%s timer OFF\n", s->name);
1003  // Stop the poller so that it doesn't do any more processing
1004  s->state.is_polling_valid = false;
1005  }
1006 
1007  }
1008 
1009  // Call the sensor power-off preparation functions
1010  for (sp = &g->sensors[0]; (s = *sp) != END_OF_LIST; sp++) {
1011 
1012  // If not configured, don't bother initializing anything else
1013  if (!s->state.is_configured)
1014  continue;
1015 
1016  // If there's a term handler to be called before power is turned off, call it
1017  if (s->term_power != NO_HANDLER) {
1018  if (!s->term_power())
1019  s->state.term_failures++;
1020  else
1021  s->state.term_failures = 0;
1022  }
1023 
1024  }
1025 
1026  // Deselect the UART if one was selected
1027  if (g->uart_required != UART_NONE)
1028  gpio_uart_select(UART_NONE);
1029  if (comm_uart_switching_allowed() && g->uart_requested != UART_NONE)
1030  gpio_uart_select(UART_NONE);
1031 
1032  // Power OFF the module
1033  if (g->power_set != NO_HANDLER) {
1034  g->power_set(g->power_set_parameter, false);
1035  g->state.is_powered_on = false;
1036  if (debug(DBG_SENSOR_MAX))
1037  DEBUG_PRINTF("%s power OFF\n", g->name);
1038  }
1039 
1040  // Clear our own state, setting us to idle.
1041  g->state.is_processing = false;
1042 
1043  // At the very end of group processing, satisfy any sensor deconfiguration requests
1044  int configured_sensors = 0;
1045  for (sp = &(*gp)->sensors[0]; (s = *sp) != END_OF_LIST; sp++)
1046  if (s->state.is_configured) {
1047  if (s->state.is_requesting_deconfiguration)
1048  s->state.is_configured = false;
1049  else
1050  configured_sensors++;
1051  }
1052 
1053  // If there are no sensors left to process, request deconfiguration of the group
1054  if (configured_sensors == 0)
1055  g->state.is_requesting_deconfiguration = true;
1056 
1057  // Now that the group is quiescent, deconfiguration is requested, do it
1058  if (g->state.is_requesting_deconfiguration)
1059  g->state.is_configured = false;
1060 
1061  // Done
1062  if (debug(DBG_SENSOR_SUPERDUPERMAX))
1063  DEBUG_PRINTF("%s !processing\n", g->name);
1064  if (debug(DBG_SENSOR_MAX))
1065  DEBUG_PRINTF("%s completed\n", g->name);
1066 
1067  }
1068 
1069  } // Looping across groups
1070 
1071  // If no groups are currently active and test mode was requested, we
1072  // can now enter it.
1073  if (fTestModeRequested) {
1074  if (groups_currently_active == 0) {
1075  fTestModeRequested = false;
1076  sensor_set_op_mode(OPMODE_TEST_SENSOR);
1077  DEBUG_PRINTF("Sensor Test Mode now active\n");
1078  } else {
1079  DEBUG_PRINTF("Sensor Test Mode waiting for %d sensors to complete\n", groups_currently_active);
1080  }
1081  }
1082 
1083  // Done
1084 
1085  if (debug(DBG_SENSOR_SUPERDUPERMAX))
1086  DEBUG_PRINTF("sensor_poll exit\n");
1087  inside_poll--;
1088 
1089  return;
1090 
1091 }
1092 
1093 // Initialize
1094 void sensor_init() {
1095  group_t **gp, *g;
1096  sensor_t **sp, *s;
1097  STORAGE *c = storage();
1098  uint32_t init_time = get_seconds_since_boot();
1099 
1100  // Loop over all sensors in all sensor groups
1101  for (gp = &sensor_groups[0]; (g = *gp) != END_OF_LIST; gp++) {
1102 
1103  // If not configured, don't bother initializing anything else
1104  g->state.is_requesting_deconfiguration = false;
1105  g->state.is_configured = (g->storage_product == c->product);
1106  if (!g->state.is_configured)
1107  continue;
1108 
1109  // Modify the sensor parameters to reflect what's in the storage parameters
1110  g->state.repeat_seconds_override = 0;
1111  char *psp, *pgn;
1112  psp = c->sensor_params;
1113  while (true) {
1114  // Exit if nothing left to parse in sensor parameters
1115  if (*psp == '\0')
1116  break;
1117  // Compare what we're sitting at with the current group name
1118  pgn = g->name;
1119  while (true) {
1120  if (*psp == '\0' || *pgn == '\0' || *psp != *pgn)
1121  break;
1122  psp++;
1123  pgn++;
1124 
1125  }
1126  // If we fully recognize the group name, process it
1127  if (*pgn == '\0' && *psp == '.') {
1128  // See if it's a subfield that we recognize
1129 #define repeat_field ".r="
1130  if (memcmp(psp, repeat_field, sizeof(repeat_field)) == 0) {
1131  psp += sizeof(repeat_field);
1132  uint16_t v = (uint16_t) strtol(psp, &psp, 0);
1133  if (debug(DBG_SENSOR))
1134  DEBUG_PRINTF("%s override repeat with %d minutes\n", g->name, v);
1135  g->state.repeat_seconds_override = (uint16_t) v*60;
1136  }
1137  }
1138  // Skip to the next psp parameter
1139  while (*psp != '\0')
1140  if (*psp++ == '/')
1141  break;
1142  }
1143 
1144  // Init the state of the sensor group
1145  g->state.is_settling = false;
1146  g->state.is_processing = false;
1147  g->state.is_polling_valid = false;
1148 
1149  // If it's to be sensed immediately, do it, else base repeats on when init started
1150  if (g->sense_at_boot)
1151  g->state.last_repeated = 0;
1152  else {
1153  g->state.last_repeated = init_time;
1154 #if defined(FAST)
1155  g->state.last_repeated = 0;
1156 #endif
1157  }
1158 
1159  // Power OFF the module as its initial state
1160  if (g->power_set == NO_HANDLER)
1161  g->state.is_powered_on = true;
1162  else {
1163  g->power_set(g->power_set_parameter, false);
1164  g->state.is_powered_on = false;
1165  if (debug(DBG_SENSOR_MAX))
1166  DEBUG_PRINTF("%s power OFF\n", g->name);
1167  }
1168 
1169  // If this group requested an app timer, create it
1170  if (g->poll_handler != NO_HANDLER) {
1171  memset(&g->state.group_timer.timer_data, 0, sizeof(g->state.group_timer.timer_data));
1172  g->state.group_timer.timer_id = &g->state.group_timer.timer_data;
1173  app_timer_create(&g->state.group_timer.timer_id, APP_TIMER_MODE_REPEATED, g->poll_handler);
1174  // Start it at init if we're polling continuously
1175  if (g->poll_continuously) {
1176  app_timer_start(g->state.group_timer.timer_id, APP_TIMER_TICKS(g->poll_repeat_milliseconds, APP_TIMER_PRESCALER), g);
1177  // Release the poller so that it's ok to proceed
1178  g->state.is_polling_valid = true;
1179  }
1180  // The code below is defensive programming because there's a known-buggy behavior in the
1181  // nRF handling of app timers, such that if we stop the timer before a single tick has happened,
1182  // the stop fails to "take". So we ensure that the settling period is at a minimum of the timer period plus slop.
1183  uint32_t min_settling_seconds = (g->poll_repeat_milliseconds / 1000) + 5;
1184  if (g->settling_seconds != 0 && g->settling_seconds < min_settling_seconds)
1185  g->settling_seconds = min_settling_seconds;
1186  }
1187 
1188  // Loop over all sensors in the group
1189  uint16_t configured_sensors = 0;
1190  for (sp = &(*gp)->sensors[0]; (s = *sp) != END_OF_LIST; sp++) {
1191 
1192  // If not configured, don't bother initializing anything else
1193  s->state.is_requesting_deconfiguration = false;
1194  s->state.is_configured = ((s->storage_sensor_mask & c->sensors) != 0);
1195  if (!s->state.is_configured)
1196  continue;
1197  configured_sensors++;
1198 
1199  // Initialize sensor state
1200  s->state.is_settling = false;
1201  s->state.is_processing = false;
1202  s->state.is_completed = false;
1203  s->state.init_failures = 0;
1204  s->state.term_failures = 0;
1205  s->state.is_polling_valid = false;
1206 
1207  // If this sensor requested an app timer, create it
1208  if (s->poll_handler != NO_HANDLER) {

1210  s->state.sensor_timer.timer_id = &s->state.sensor_timer.timer_data;
