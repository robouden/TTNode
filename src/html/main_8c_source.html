24 // Primary power-sensitive work dispatcher
25 void power_manage(void) {
26 
27 #ifdef NRF52
28  // From nRF5 SDK v11.0.0 release_notes.txt Release Date: Week 10, 2016
29  // Found in our code because the very first TWI transaction caused the
30  // FPU exception, thus causing massive battery drain.
31  // The release_notes say:
32  // - When the FPU is in use, it triggers the FPU_IRQn interrupt when one of
33  // the six exception flags (IDC, IXC, UFC, OFC, DZC, IOC) is set.
34  // The FPU interrupt will always set the pending flag (even if the
35  // interrupt is not enabled), irrespective of whether the user is
36  // interested in the exception bit.
37  // The pending flag then prevents the SoftDevice from going into low
38  // power mode when sd_app_evt_wait() is called.
39  // Therefore, always clear the exception bits and the pending interrupt
40  // before calling sd_app_evt_wait().
41 #define FPU_EXCEPTION_MASK 0x0000009F
42  __set_FPSCR(__get_FPSCR() & ~(FPU_EXCEPTION_MASK));
43  (void) __get_FPSCR();
44  NVIC_ClearPendingIRQ(FPU_IRQn);
45 #endif
46 
47 #if defined(SCHEDDEBUG)
48 #ifdef LED_PIN_YEL
49  gpio_pin_set(LED_PIN_YEL, true);
50 #endif
51 #ifdef LED_PIN_RED
52  gpio_pin_set(LED_PIN_RED, false);
53 #endif
54 #endif
55 
56  sd_app_evt_wait();
57 
58 #if defined(SCHEDDEBUG)
59 #ifdef LED_PIN_YEL
60  gpio_pin_set(LED_PIN_YEL, false);
61 #endif
62 #ifdef LED_PIN_RED
63  gpio_pin_set(LED_PIN_RED, true);
64 #endif
65 #endif
66 
67  // Do work enqueued during interrupt service routines
68  app_sched_execute();
69 
70 #if defined(SCHEDDEBUG)
71 #ifdef LED_PIN_RED
72  gpio_pin_set(LED_PIN_RED, false);
73 #endif
74 #ifdef LED_PIN_YEL
75  gpio_pin_set(LED_PIN_YEL, false);
76 #endif
77 #endif
78 
79 }
