83 // Returns a value based on the battery status. Note that these values
84 // are defined in sensor.h as a bit mask, so they can be tested via "==" or switch
85 // OR via bitwise-&, as opposed to *needing* to do == or a switch statement.
86 // Note that this is an extremely low level routine, so
87 // don't modify it to call anything that might potentially go recursive.
88 uint16_t battery_status() {
89 
90  // Handle certain compile-time test modes
91 #if defined(BATTERY_FULL)
92  return BAT_FULL;
93 #endif
94 #if defined(BATTERY_NORMAL)
95  return BAT_NORMAL;
96 #endif
97 
98  // Exit if battery is truly dead, no matter what mode we are in
99 #if BATTERY_AUTOADJUST
100  if (lastKnownBatterySOC != 0 && lastKnownBatterySOC < 10.0)
101  return BAT_DEAD;
102 #endif
103 
104  // Special-case certain operating modes
105 
106  switch (sensor_op_mode()) {
107  case OPMODE_TEST_BURN:
108  return BAT_BURN;
109  case OPMODE_TEST_FAST:
110  return BAT_TEST;
111  case OPMODE_MOBILE:
112  return BAT_MOBILE;
113  case OPMODE_TEST_DEAD:
114  return BAT_NO_SENSORS;
115  }
116 
117  // Exit if we can't auto-adjust
118 #if !BATTERY_AUTOADJUST
119  return BAT_NORMAL;
120 #endif
121 
122  // If it's never yet been set, just treat as normal
123  if (lastKnownBatterySOC == 0)
124  return BAT_NORMAL;
125 
126  // Recovery mode
127  if (batteryRecoveryMode) {
128  if (lastKnownBatterySOC < 70.0)
129  return BAT_EMERGENCY;
130  batteryRecoveryMode = false;
131  return BAT_NORMAL;
132  } else {
133 
134  // Important note: Sadly, I learned the hard way that because of
135  // the internal chemistry of LIPO batteries, they must NEVER be
136  // allowed to discharge below 3.0V per cell or else they will
137  // suffer internal damage. Copper shunts may form within the
138  // cells that may cause an electrical short. Therefore, this
139  // code goes through extraordinary lengths to ensure that we
140  // cease draining the battery when it gets low.
141 
142  if (lastKnownBatterySOC < 20.0) {
143  batteryRecoveryMode = true;
144  return BAT_EMERGENCY;
145  }
146  }
147 
148  // Danger mode
149  if (lastKnownBatterySOC < 60.0)
150  return BAT_LOW;
151 
152  // Danger mode
153  if (lastKnownBatterySOC < 40.0)
154  return BAT_WARNING;
155 
156  // Determine if this is a full battery, debouncing it between min & max
157  if (lastKnownBatterySOC < BATTERY_SOC_HIGHPOWER_MIN) {
158  fullBatteryRecoveryMode = true;
159  return BAT_NORMAL;
160  }
161  if (fullBatteryRecoveryMode && lastKnownBatterySOC < BATTERY_SOC_HIGHPOWER_MAX)
162  return BAT_NORMAL;
163  fullBatteryRecoveryMode = false;
164  return BAT_FULL;
165 
166 }

168 // Compute a simulated SOC value based on voltage data
169 
170 // Note that our goal is that 100% means "normal full", however
171 // we will try to actively do higher-power activities while modulating the
172 // charging to between HIGHPOWER_MIN-HIGHPOWER_MAX, while always doing high-power activities
173 // above HIGHPOWER_MAX under the assumption that this means we're plugged-in.
174 //
175 // Important note: Sadly, I learned the hard way that because of
176 // the internal chemistry of LIPO batteries, they must NEVER be
177 // allowed to discharge below 3.2V per cell or else they will
178 // suffer internal damage. Copper shunts may form within the
179 // cells that may cause an electrical short.
180 //
181 // http://batteryuniversity.com/learn/article/how_to_prolong_lithium_based_batteries
182 //
183 // Further, note that there is NO CODE in this project that explicitly tests
184 // for battery voltages. Everything is based on SOC, so if the
185 // fuel gauge driver uses voltage to compute SOC, this code's behavior
186 // is super-critical to overall device performance.
187 float battery_soc_from_voltage(float voltage) {
188 #ifdef scv0
189  float minV = 3.5;
190  float maxV = 4.0;
191 #endif
192 #ifdef scv1
193  float minV = 3.5;
194  float maxV = 3.9;
195 #endif
196 #ifdef scv2
197  float minV = 3.5;
198  float maxV = 4.0;
199 #endif
200  float soc;
201  float curV = voltage;
202  if (curV < minV)
203  curV = 0;
204  else
205  curV -= minV;
206  maxV -= minV;
207  soc = (curV * 100.0) / maxV; // Assume linear drain because of our device's behavior
208 
209  // Done
210  return(soc);
211 }
