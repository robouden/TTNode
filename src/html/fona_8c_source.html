427 // Return true if transmitting would be pointless
428 bool fona_is_busy() {
429 
430  // Exit if we're in the middle of sending a different message,
431  // This can happen because transmits take a finite amount of time,
432  // and we might be being asked to transmit both for the local
433  // geiger counter as well as the URSC-connected bGeigie.
434  if (fromFona.state != COMM_STATE_IDLE) {
435 
436 #ifdef FLOWTRACE
437  DEBUG_PRINTF("(Can't send - busy %d!)\n", fromFona.state);
438 #endif
439 
440  return true;
441  }
442 
443  // If a transmit is in-progress, the state might be idle when
444  // we are awaiting the next phase of the transmit process.
445  // Indicate that we're busy so that we don't try to come back
446  // and transmit something else.
447  if (deferred_active != 0) {
448 
449  // If it's been active for too long, this is an indication that
450  // either the fona module is misbehaving or there was some uart issue.
451  // So as defensively not to get "stuck", this will abort if it is
452  // active for far, far too long.
453  if (!ShouldSuppress(&deferred_active, 60)) {
454  deferred_active = 0;
455  DEBUG_PRINTF("Warning: Fona transmit timeout!\n");
456  return false;
457  }
458 
459  // We are still waiting for transmit to complete
460  return true;
461 
462  }
463 
464 #ifdef FONAGPS
465  // Exit if we're waiting to shut off the GPS
466  if (gpsSendShutdownCommandWhenIdle)
467  return true;
468 
469  // Exit if we're waiting to update the location. If we've been
470  // waiting too long, though, just give up.
471  if (gpsUpdateLocation) {
472  if (ShouldSuppress(&fonaInitLastInitiated, gpsHaveLocation ? GPS_ABORT_IMPROVE_MINUTES*60L : GPS_ABORT_FIRST_MINUTES*60L))
473  return true;
474  gpsUpdateLocation = false;
475  }
476 #endif
477 
478  // Not busy
479  return false;
480 }

482 // Transmit a well-formed protocol buffer to the LPWAN as a message
483 bool fona_send_to_service(uint8_t *buffer, uint16_t length, uint16_t RequestType) {
484  char command[64];
485 
486  // Exit if we're not yet initialized
487  if (!comm_is_initialized())
488  return false;
489 
490  // If we're busy doing something else, drop this
491  if (fona_is_busy())
492  return false;
493 
494  // If there's already a deferred command, drop this
495  if (deferred_active != 0)
496  return false;
497 
498  // If this is too long for buffers, exit
499  if (length > sizeof(deferred_iobuf))
500  return false;
501 
502  // Now that we're committed, if this is a request that requires a reply, remember that we're doing so.
503  awaitingTTServeReply = (RequestType != REPLY_NONE);
504 
505  // Set up the deferred data
506  deferred_active = get_seconds_since_boot();
507  deferred_iobuf_length = length;
508  memcpy(deferred_iobuf, buffer, length);
509  deferred_request_type = RequestType;
510 
511  // If this is a transmit-only request, do it via UDP, else TCP/HTTP
512  if (deferred_request_type == REPLY_NONE) {
513 
514 #ifdef FLOWTRACE
515  DEBUG_PRINTF("UDP send: %d\n", length);
516 #endif
517 
518  // Bump stats about what we've transmitted
519  stats_io(length, 0);
520 
521  // Transmit it, expecting the deferred handler to finish this.
522  deferred_callback_requested = true;
523  deferred_done_after_callback = true;
524  sprintf(command, "at+cipsend=0,%u,\"%s\",%u", deferred_iobuf_length, service_udp_ipv4, SERVICE_UDP_PORT);
525  fona_send(command);
526  setstateF(COMM_FONA_MISCRPL);
527 
528  } else {
529 
530 #if USETCP
531 
532 #ifdef FLOWTRACE
533  DEBUG_PRINTF("TCP send: %d\n", length);
534 #endif
535 
536  // Bump stats about what we've transmitted
537  stats_io(length, 0);
538 
539  // Transmit it, after which the "cipsend" will process the deferred iobuf
540  ip_open_retries = 8;
541  comm_set_connect_state(CONNECT_STATE_APP_SERVICE);
542  sprintf(command, "at+cipopen=1,\"TCP\",\"%s\",%u", service_tcp_ipv4, SERVICE_TCP_PORT);
543  fona_send(command);
544  setstateF(COMM_FONA_CIPOPENRPL2);
545 
546 #else
547 
548 #ifdef FLOWTRACE
549  DEBUG_PRINTF("HTTP send: %d\n", length);
550 #endif
551 
552  // Transmit it, expecting to receive a callback at fona_http_start_send() after
553  // the session is open.
554  sprintf(command, "at+chttpsopse=\"%s\",%u,1", SERVICE_HTTP_ADDRESS, SERVICE_HTTP_PORT);
555  fona_send(command);
556  setstateF(COMM_FONA_CHTTPSOPSERPL);
557 
558 #endif // USETCP
559 
560  }
561 
562  // Done
563  return true;
564 }

566 // Initiate the HTTP send now that the session is open.
567 // Note that this REPLACES the contents of deferred_iobuf with a hexified version of same.
568 #if !USETCP
569 void fona_http_start_send() {
570  char command[64];
571  char hiChar, loChar, body[sizeof(deferred_iobuf)*2+50+1];
572  uint16_t i, header_length, hexified_length, total_length;
573 
574  // The hexified data length will be the original data * 2 (because of hexification)
575  hexified_length = deferred_iobuf_length * 2;
576 
577  // Put together a minimalist HTTP header and command to transmit it
578  sprintf(body, "POST %s HTTP/1.1\r\nHost: %s:%d\r\nUser-Agent: TTNODE\r\nContent-Length: %d\r\n\r\n",
579  SERVICE_HTTP_TOPIC, SERVICE_HTTP_ADDRESS, SERVICE_HTTP_PORT, hexified_length);
580 
581  // Compute the remaining lengths
582  header_length = strlen(body);
583 
584  // Hexify into the body, and add trailing \r\n, and a null term which is useful for %s printing when debugging
585  total_length = header_length;
586  for (i=0; i<deferred_iobuf_length; i++) {
587  if ( (i + header_length) >= (sizeof(deferred_iobuf) - sizeof("00\r\n")) )
588  break;
589  HexChars(deferred_iobuf[i], &hiChar, &loChar);
590  body[total_length++] = hiChar;
591  body[total_length++] = loChar;
592  }
593  body[total_length] = '\0';
594 
595  // Bump stats about what we've transmitted
596  stats_io(total_length, 0);
597 
598  // Move it back into the iobuf
599  deferred_iobuf_length = total_length;
600  memcpy(deferred_iobuf, body, deferred_iobuf_length);
601 
602  // Generate a command
603  deferred_callback_requested = true;
604  sprintf(command, "at+chttpssend=%u", deferred_iobuf_length);
605  fona_send(command);
606 
607 }
608 #endif // !USETCP

651 // Process the stuff in the deferred iobuf
652 void fona_process_received() {
653  uint8_t buffer[CMD_MAX_LINELENGTH];
654  uint16_t msgtype;
655 
656  // Regardless of what we received, indicate that we're no longer waiting for
657  // a TTServe reply, because we only want to listen for a single receive
658  // window, for power reasons If we don't pick it up now, we'll pick it up
659  // eventually on the next message to the service.
660  awaitingTTServeReply = false;
661 
662  // Null-terminate the io buffer
663  if (deferred_iobuf_length == sizeof(deferred_iobuf))
664  deferred_iobuf_length--;
665  deferred_iobuf[deferred_iobuf_length] = '\0';
666 
667  // Only do this if we got something back
668  if (deferred_iobuf_length != 0) {
669 
670  // Bump stats about what we've received on the wire
671  stats_io(0, deferred_iobuf_length);
672 
673  // Decode the message
674  msgtype = comm_decode_received_message((char *)deferred_iobuf, NULL, buffer, sizeof(buffer) - 1, NULL);
675  if (msgtype != MSG_REPLY_TTSERVE) {
676  // This can happen if we get an HTTP error in the body
677  deferred_iobuf[deferred_iobuf_length] = '\0';
678  DEBUG_PRINTF("?: %s\n", deferred_iobuf);
679  } else {
680 
681  // Process it
682  recv_message_from_service((char *)buffer);
683 
684  }
685 
686  }
687 
688  // We're now done.
689  deferred_active = 0;
690  comm_oneshot_completed();

734 // Do modem reset processing
735 bool fona_needed_to_be_reset() {
736  uint32_t secondsSinceBoot = get_seconds_since_boot();
737 
738 #ifdef FONAGPS
739  // Check to see if Fona GPS needed to be shut down, and do it if appropriate.
740  if (gpsSendShutdownCommandWhenIdle && fromFona.state == COMM_STATE_IDLE) {
741  gpsSendShutdownCommandWhenIdle = false;
742  fona_send("at+cgpsinfo=0");
743  setstateF(COMM_FONA_CGPSINFO2RPL);
744  return true;
745  }
746 #endif
747 
748  // Check to see if the Fona card is simply missing or powered off
749  if (fona_received_since_powerup == 0 && secondsSinceBoot > BOOT_DELAY_UNTIL_INIT) {
750  if (!fonaLock && !fonaInitCompleted && fonaInitInProgress ) {
751  DEBUG_PRINTF("CELL is non-responsive.\n");
752  fonaNoNetwork = true;
753 #ifdef FONAGPS
754  fona_gps_shutdown();
755 #endif
756  fona_shutdown();
757  comm_oneshot_completed();
758  return true;
759  }
760  }
761 
762  // Initialize the entire cmd subsystem only after the chip has
763  // had a chance to stabilize after boot.
764  if (!fonaInitCompleted && !fonaInitInProgress && secondsSinceBoot > BOOT_DELAY_UNTIL_INIT) {
765  DEBUG_PRINTF("reset\n");
766  fona_reset(false);
767  return true;
768  }
769 
770  // See if we should reset the watchdog because of clock wraparound
771  if (watchdog_set_time >= secondsSinceBoot)
772  fona_watchdog_reset();
773 
774  // Reset all state and the modem if appropriate,
775  // handling clock wrap. Only do this, though, if
776  // the modem is in a non-idle state, because we
777  // don't want to be performing resets of completely
778  // idle devices.
779  if (secondsSinceBoot >= CELL_WATCHDOG_SECONDS) {
780  uint32_t watchdog_seconds = CELL_WATCHDOG_SECONDS;
781  // Extend to 5m for super-long operations such as DFU file transfer
782  if (watchdog_extend)
783  watchdog_seconds = 300;
784  // If we've gone over the watchdog time, reset the world
785  if ((secondsSinceBoot - watchdog_set_time) > watchdog_seconds)
786  if (fromFona.state != COMM_STATE_IDLE) {
787  DEBUG_PRINTF("WATCHDOG: Fona stuck st=%d cc=%d b=%d,%d,%d '%s'\n", fromFona.state, fromFona.complete, fromFona.busy_length, fromFona.busy_nextput, fromFona.busy_nextget, fromFona.buffer);
788  // If we're in oneshot mode, use a much bigger stick to reset it, just for good measure
789  // This ensures that the uart switch is set appropriately.
790  // We need to do a hardware reset to close currently open sessions
791  fonaForceFullHardwareReset = true;
792  if (!comm_oneshot_currently_enabled())
793  fona_reset(true);
794  else {
795  comm_deselect("fona reset");
796  comm_reselect();
797  }
798  if (fRecordingStats)
799  stats()->errors_fona++;
800  return true;
801  }
802  }
803 
804  // Not reset
805  return false;
806 
807 }

958 // Primary state processing of the command buffer
959 void fona_process() {
960 
961  // If there was a spurious app sched event remaining after powering off the module, ignore it
962  if (gpio_current_uart() != UART_FONA)
963  return;
964 
965  // If it's not complete, just exit.
966  if (!fromFona.complete)
967  return;
968 
969  // If extreme debugging
970  if (debug(DBG_RX))
971  DEBUG_PRINTF("<%d %s\n", fromFona.state, fromFona.buffer);
972 
973  // If we've never been initialized and yet we're being called here,
974  // just ignore it because we're in an odd state.
975  if (fromFona.state != COMM_FONA_RESETREQ && !fonaInitInProgress && !fonaInitCompleted) {
976  setidlestateF();
977  return;
978  }
979 
980  // If displaying version info
981  if (atiMode && fromFona.buffer[0] != '+' && fromFona.buffer[0] != 'O')
982  DEBUG_PRINTF("%s\n", fromFona.buffer);
983 
991 
992  switch (fromFona.state) {
993 
997 
998  case COMM_FONA_RESETREQ: {
999  // If we get here when powered off, force ourselves idle
1000  if (comm_is_deselected()) {
1001  fona_reset(true);
1002  break;
1003  }
1004  // Initialize comms
1005  if (fonaFirstResetAfterInit) {
1006  fonaFirstResetAfterInit = false;
1007  DEBUG_PRINTF("CELL initializing\n");
1008  } else {
1009  stats()->resets++;
1010  DEBUG_PRINTF("Cell reset\n");
1011  }
1012  fona_watchdog_reset();
1013  fonaNoNetwork = false;
1014  fonaInitCompleted = false;
1015  fonaInitInProgress = true;
1016  fonaInitLastInitiated = get_seconds_since_boot();
1017  deferred_active = 0;
1018  deferred_callback_requested = false;
1019  deferred_done_after_callback = false;
1020  awaitingTTServeReply = false;
1021  serial_transmit_enable(true);
1022  if (apn[0] == '\0')
1023  strcpy(apn, storage()->carrier_apn);
1024  gpio_indicate(INDICATE_CELL_INITIALIZING);
1025  // Regardless of the state previously, we MUST
1026  // blast the chip immediately with the desire to
1027  // turn off flow control. These settings
1028  // are persistent across chip resets, and it
1029  // is critical that they both be done before the
1030  // reset so that it will come out of the reset
1031  // in a mode where we can communicate with it
1032  // regardless of the hardware flow control config.
1033  fona_send("at+cgfunc=11,0");
1034  setstateF(COMM_FONA_CGFUNCRPL1);
1035  break;
1036  }
1037 
1038  case COMM_FONA_CGFUNCRPL1: {
1039  // Save time and energy if we just powered on
1040  // the module by skipping a high-overhead reset.
1041  // This is necessary, however, if triggered
1042  // by a subsequent watchdog operation.
1043  // We only do this, however, if we already have
1044  // the location, because we've observed that we
1045  // need to do a creset to set the GPS function
1046  // to stabilize, as evidenced by an ERROR(120)
1047  // resulting from the very first at+cgps=1
1048  bool fDoOptimizedReset = false;
1049 #ifdef FONAGPS
1050  gpsSendShutdownCommandWhenIdle = false;
1051  if (gpsHaveLocation && !gpsUpdateLocation)
1052  if (fonaDFUInProgress || !fonaForceFullHardwareReset)
1053  fDoOptimizedReset = true;
1054 #else
1055  if (fonaDFUInProgress || !fonaForceFullHardwareReset)
1056  fDoOptimizedReset = true;
1057 #endif
1058  fonaForceFullHardwareReset = false;
1059  if (fDoOptimizedReset) {
1060  fona_send("ate0");
1061  setstateF(COMM_FONA_ECHORPL);
1062  } else {
1063  DEBUG_PRINTF("Fona: full reset\n");
1064  fona_send("at+creset");
1065  setstateF(COMM_FONA_CRESETRPL);
1066  }
1067  break;
1068  }
1069 
1070  case COMM_FONA_CRESETRPL: {
1071  // Process start up-front, else we'd go recursive because of commonreplyF()
1072  if (thisargisF("start")) {
1073  seenF(0x01);
1074  fRecordingStats = true;
1075  } else if (thisargisF("+cpin: ready"))
1076  seenF(0x02);
1077  else if (thisargisF("pb done")) {
1078  // Wait until 1 second after when we think we're done
1079  // This seems to be necessary else we get a +CME ERROR: SIM busy
1080  nrf_delay_ms(1000);
1081  seenF(0x04);
1082  } else if (commonreplyF())
1083  break;
1084  if (allwereseenF(0x07)) {
1085  fona_send("ate0");
1086  setstateF(COMM_FONA_ECHORPL);
1087  }
1088  break;
1089  }
1090 
1091  case COMM_FONA_STARTRPL: {
1092  fona_send("ate0");
1093  setstateF(COMM_FONA_ECHORPL);
1094  break;
1095  }
1096 
1097  case COMM_FONA_ECHORPL: {
1098  if (commonreplyF())
1099  break;
1100  if (thisargisF("ok")) {
1101  if (serial_hwfc_enabled()){
1102  // Enable hardware flow control & reconfigure GPIO
1103  fona_send("at+cgfunc=11,1");
1104  } else {
1105  // Again ensure that flow control is disabled
1106  fona_send("at+cgfunc=11,0");
1107  }

1109  }

1111  }

1118  // Enable hardware flow control & reconfigure GPIO
1119  fona_send("at+ifc=2,2");
1120  setstateF(COMM_FONA_IFCRPL2);
1121  }
1122  } else {

1124  // an error if we've not first done an at+cgfunc=11,1
1125  processstateF(COMM_FONA_IFCRPL2);
1126  }
1127  break;
1128  }
1129

1132  break;

1134  fona_send("at+cgfunc=1,0");

1153  // Otherwise, let's just jump to verifying carrier connectivity.
1154  if (!gpsHaveLocation || gpsUpdateLocation) {
1155  fona_send("at+cgps=1");

1157  break;
1158  }
1159 #endif

1161  fona_send("at+cpsi=5");

1168  // ignore error reply because it may have been user-enabled via at+cgpsauto=1
1169  if (thisargisF("error") || thisargisF("ok"))
1170  seenF(0x01);
1171  if (commonreplyF())
1172  break;

1174  fona_send("at+cgpsinfo=10");

1177  break;

1182  case COMM_FONA_CGPSINFORPL: {
1183  if (commonreplyF())
1184  break;

1186  seenF(0x01);
1187  if (allwereseenF(0x01)) {
1188  // Settle down from reset. This appears to be necessary, else
1189  // we see ourselves getting stuck in this state.
1190  nrf_delay_ms(750);

1193 #ifdef FONANOSIM
