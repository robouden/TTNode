140 void log_debug_write_hex(uint32_t value) {
141  uint8_t nibble;
142  uint8_t i = 8;
143  debug_putchar('0');
144  debug_putchar('x');
145  while ( i-- != 0 ) {
146  nibble = (value >> (4 * i)) & 0x0F;
147  debug_putchar( (nibble > 9) ? ('A' + nibble - 10) : ('0' + nibble) );
148  }
149 }
