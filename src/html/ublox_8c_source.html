36 // UBLOXM8 commands
37 // Addresses FD and FE contain the length that's readable at FF
38 #define UBLOXM8_GETLEN_ADDR 0xFD

67 // Callback that's activated when TWI data has been read.
68 // Unlike other callbacks, GPS does its transactions inside the poller, and ensures
69 // that the measured data is always available in its context structure.
70 void gps_callback(ret_code_t result, twi_context_t *t) {
71  uint16_t i, j;
72 #ifdef GPSDEBUG
73  static uint32_t lastDebugOutput = 0;
74  int d;
75  bool doDebugOutput = false;
76  char DebugOutput1[32], DebugOutput2[32];
77  DebugOutput1[0] = DebugOutput2[0] = '\0';
78 #endif
79 
80  // Mark as completed
81  if (!twi_completed(t))
82  return;
83 
84  // Remember the number of times that we attempted to fetch data from the GPS
85  ioGPS.gpsDataAttempts++;
86 
87  // Parse the buffer, looking for either of these. (We'd rather have gpgga because it has altitude, but we may not lock)
88  // $GPGGA,<time>,<latitude>,<ns>,<longitude>,<ew>,<quality>,<numsat>,<hdop>,<alt>,<m>
89  // $GPGLL,<latitude>,<ns>,<longitude>,<ew>
90  // https://www.u-blox.com/sites/default/files/products/documents/u-blox7-V14_ReceiverDescrProtSpec_(GPS.G7-SW-12001)_Public.pdf
91 
92  for (i = 0; i < sizeof(ioGPS.buffer); i++) {
93 
94  // Done? This is the expected delimeter.
95  if (ioGPS.buffer[i] == 0xff)
96  break;
97 
98  // Remember that we've actually looked at some data.
99  ioGPS.gpsDataParsed++;
100  gpio_indicate(INDICATE_GPS_CONNECTING);
101 
102  // Look for the beginning of the $GPGGA NMEA string
103  if ((ioGPS.buffer[i] == '$') && ((sizeof(ioGPS.buffer) - i) >= 6) && (memcmp(&ioGPS.buffer[i], "$GPGGA", 6) == 0)) {
104  char *lat, *ns, *lon, *ew, *alt, *fix;
105  bool haveLat, haveLon, haveNS, haveEW, haveAlt, haveFix;
106  uint16_t commas;
107 
108 #ifdef GPSDEBUG
109  for (d=0; d<sizeof(DebugOutput1)-1; d++)
110  if (ioGPS.buffer[i+d] < ' ')
111  DebugOutput1[d] = '\0';
112  else
113  DebugOutput1[d] = ioGPS.buffer[i+d];
114  DebugOutput1[sizeof(DebugOutput1)-1] = '\0';
115  doDebugOutput = true;
116 #endif
117 
118  commas = 0;
119  haveLat = haveLon = haveNS = haveEW = haveAlt = haveFix = false;
120  lat = ns = lon = ew = alt = fix = "";
121 
122  for (j = i; j < sizeof(ioGPS.buffer); j++)
123  if (ioGPS.buffer[j] == ',') {
124  ioGPS.buffer[j] = '\0';
125  commas++;
126  if (commas == 1) {
127  // Sitting at comma before Time
128  }
129  if (commas == 2) {
130  // Sitting at comma before Lat
131  lat = (char *) &ioGPS.buffer[j + 1];
132  }
133  if (commas == 3) {
134  // Sitting at comma before NS
135  haveLat = (lat[0] != '\0');
136  ns = (char *) &ioGPS.buffer[j + 1];
137  }
138  if (commas == 4) {
139  // Sitting at comma before Lon
140  haveNS = (ns[0] != '\0');
141  lon = (char *) &ioGPS.buffer[j + 1];
142  }
143  if (commas == 5) {
144  // Sitting at comma before EW
145  haveLon = (lon[0] != '\0');
146  ew = (char *) &ioGPS.buffer[j + 1];
147  }
148  if (commas == 6) {
149  // Sitting at comma before Quality
150  haveEW = (ew[0] != '\0');
151  fix = (char *) &ioGPS.buffer[j + 1];
152  }
153  if (commas == 7) {
154  // Sitting at comma before NumSat
155  // 1 is valid GPS fix, 2 is valid DGPS fix
156  haveFix = (fix[0] == '1' || fix[0] == '2');
157  }
158  if (commas == 8) {
159  // Sitting at comma before Hdop
160  }
161  if (commas == 9) {
162  // Sitting at comma before Alt
163  alt = (char *) &ioGPS.buffer[j + 1];
164  }
165  if (commas == 10) {
166  // Sitting at comma before M unit
167  haveAlt = (alt[0] != '\0');
168  break;
169  }
170  }
171 
172  // If we've got what we need, process it and exit.
173  if (haveFix && haveLat && haveNS & haveLon && haveEW) {
174  float fLatitude = GpsEncodingToDegrees(lat, ns);
175  float fLongitude = GpsEncodingToDegrees(lon, ew);
176  if (fLatitude != 0 || fLongitude != 0) {
177  ioGPS.haveLocation = true;
178  ioGPS.gpsLatitude = fLatitude;
179  ioGPS.gpsLongitude = fLongitude;
180  if (haveAlt) {
181  ioGPS.gpsAltitude = atof(alt);
182  ioGPS.haveFullLocation = true;
183  }
184  }
185  }
186 
187  } // if GPGGA
188 
189  // Look for the beginning of the $GPGLL NMEA string
190  if ((ioGPS.buffer[i] == '$') && ((sizeof(ioGPS.buffer) - i) >= 6) && (memcmp(&ioGPS.buffer[i], "$GPGLL", 6) == 0)) {
191  char *lat, *ns, *lon, *ew;
192  bool haveLat, haveLon, haveNS, haveEW;
193  uint16_t commas;
194 
195 #ifdef GPSDEBUG
196  for (d=0; d<sizeof(DebugOutput2)-1; d++)
197  if (ioGPS.buffer[i+d] < ' ')
198  DebugOutput2[d] = '\0';
199  else
200  DebugOutput2[d] = ioGPS.buffer[i+d];
201  DebugOutput2[sizeof(DebugOutput2)-1] = '\0';
202  doDebugOutput = true;
203 #endif
204 
205  commas = 0;
206  haveLat = haveLon = haveNS = haveEW = false;
207  lat = ns = lon = ew = "";
208 
209  for (j = i; j < sizeof(ioGPS.buffer); j++)
210  if (ioGPS.buffer[j] == ',') {
211  ioGPS.buffer[j] = '\0';
212  commas++;
213  if (commas == 1) {
214  // Sitting at comma before Lat
215  lat = (char *) &ioGPS.buffer[j + 1];
216  }
217  if (commas == 2) {
218  // Sitting at comma before NS
219  haveLat = (lat[0] != '\0');
220  ns = (char *) &ioGPS.buffer[j + 1];
221  }
222  if (commas == 3) {
223  // Sitting at comma before Lon
224  haveNS = (ns[0] != '\0');
225  lon = (char *) &ioGPS.buffer[j + 1];
226  }
227  if (commas == 4) {
228  // Sitting at comma before EW
229  haveLon = (lon[0] != '\0');
230  ew = (char *) &ioGPS.buffer[j + 1];
231  }
232  if (commas == 5) {
233  // done
234  haveEW = (ew[0] != '\0');
235  break;
236  }
237  }
238 
239  // If we've got what we need, process it and exit.
240  if (haveLat && haveNS & haveLon && haveEW) {
241  float fLatitude = GpsEncodingToDegrees(lat, ns);
242  float fLongitude = GpsEncodingToDegrees(lon, ew);
243  if (fLatitude != 0 || fLongitude != 0) {
244  ioGPS.haveLocation = true;
245  ioGPS.gpsLatitude = fLatitude;
246  ioGPS.gpsLongitude = fLongitude;
247  }
248  }
249 
250  } // if GPGLL
251 
252  } // Loop over iobuf
253 
254 #ifdef GPSDEBUG
255  if (ioGPS.gpsDataParsed == 0) {
256  if (!ShouldSuppress(&lastDebugOutput, 5)) {
257  DEBUG_PRINTF("No GPS data\n");
258  }
259  } else if (doDebugOutput) {
260  if (!ShouldSuppress(&lastDebugOutput, 5)) {
261  char DebugOutput[128];
262  sprintf(DebugOutput, "GPS: %s %s\n", DebugOutput1, DebugOutput2);
263  DEBUG_PRINTF(DebugOutput);
264  }
265  }
266 #endif
267 
268 } // Notification of TWI result

275 // GPS poller
276 void s_gps_poll(void *s) {
277 
278  // Exit if we're not supposed to be here
279  if (!sensor_is_polling_valid(s))
280  return;
281 
282  // If we've already got the full location, terminate the polling just to save battery life
283  if ((comm_gps_get_value(NULL, NULL, NULL) == GPS_LOCATION_FULL) || ioGPS.gpsShutdown) {
284  sensor_measurement_completed(s);
285  DEBUG_PRINTF("TWI GPS acquired.\n");
286  return;
287  }
288 
289  // If the GPS hardware isn't even present, terminate the polling to save battery life.
290  if (ioGPS.gpsDataAttempts > 100 && ioGPS.gpsDataParsed == 0) {
291  sensor_measurement_completed(s);
292  DEBUG_PRINTF("TWI GPS shutdown. (not found)\n");
293  return;
294  }
295 
296  // Make sure it appears that we are connecting to GPS
297  gpio_indicate(INDICATE_GPS_CONNECTING);
298 
299  // This assumes, as in the ublox spec in 10.5.1.1, that the default address pointer at startup is 0xff, so all we need to do
300  // is to start reading. Also, prefill the buffer with ff because that represents "no more data" for the ublox.
301  memset(ioGPS.buffer, 0xff, sizeof(ioGPS.buffer));
302  ioGPS.address[0] = UBLOXM8_GETDATA_ADDR;
303  static app_twi_transfer_t transfers[] = {
304  APP_TWI_WRITE(UBLOXM8_I2C_ADDRESS, &ioGPS.address[0], UBLOXM8_ADDRESS_LEN, APP_TWI_NO_STOP),
305  APP_TWI_READ(UBLOXM8_I2C_ADDRESS, &ioGPS.buffer[0], sizeof(ioGPS.buffer), 0)
306  };
307  static app_twi_transaction_t transaction = {
308  .callback = twi_callback,
309  .p_user_data = "UBLOX",
310  .p_transfers = transfers,
311  .number_of_transfers = sizeof(transfers) / sizeof(transfers[0])
312  };
313  if (!twi_schedule(s, gps_callback, &transaction))
314  sensor_unconfigure(s);
315 }

346 uint16_t s_gps_get_value(float *lat, float *lon, float *alt) {
347  if (!ioGPS.haveLocation)
