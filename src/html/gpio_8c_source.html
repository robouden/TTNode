141 // Turn power on or off for a pin
142 void gpio_power_set (uint16_t pin, bool fOn) {
143 
144  // If we're debugging power, just hammer everything modified
145  // below to be ON.
146 #ifdef POWERDEBUG_OUTPUTS_ON
147  gpio_pin_set(pin, true);
148  gpio_cfg_output(POWER_PIN_PS_5V);
149  gpio_pin_set(POWER_PIN_PS_5V, true);
150  gpio_cfg_output(POWER_PIN_PS_BAT);
151  gpio_pin_set(POWER_PIN_PS_BAT, true);
152  gpio_cfg_output(POWER_PIN_TWI);
153  gpio_pin_set(POWER_PIN_TWI, true);
154  return;
155 #endif
156 
157  // Debug modes to force the power of sensors on or off
158  if (fOn && fSensorPowerForceOff)
159  return;
160  if (!fOn && fSensorPowerForceOn)
161  return;
162 
163  // On solarcast, turn on or off the _PS_ pins based on those pins that depend upon them
164 #ifdef scv1
165  static uint32_t pins_enabled = 0L; // All are cleared in gpio_init()
166  static uint32_t pin_mask_5V = POWER_PINS_REQUIRING_PS_5V;
167  static uint32_t pin_mask_BAT = POWER_PINS_REQUIRING_PS_BAT;
168  static uint32_t pin_mask_TWI = POWER_PINS_REQUIRING_TWI;
169  uint32_t pin_mask_before = pins_enabled;
170 
171  // Remember the state of which pins are powered on
172  if (fOn)
173  pins_enabled |= (1 << pin);
174  else
175  pins_enabled &= ~(1 << pin);
176 
177  // Do special processing where, in order to deliver power, more than one pin is necessary
178  if ((pin_mask_before & pin_mask_TWI) == 0 && (pins_enabled & pin_mask_TWI) != 0) {
179  gpio_pin_set(POWER_PIN_TWI, true);
180  } else if ((pin_mask_before & pin_mask_TWI) != 0 && (pins_enabled & pin_mask_TWI) == 0) {
181  gpio_pin_set(POWER_PIN_TWI, false);
182  }
183  if ((pin_mask_before & pin_mask_5V) == 0 && (pins_enabled & pin_mask_5V) != 0) {
184 #ifndef POWERDEBUG_DISABLE_5V
185  gpio_cfg_output(POWER_PIN_PS_5V);
186  gpio_pin_set(POWER_PIN_PS_5V, true);
187 #endif
188  } else if ((pin_mask_before & pin_mask_5V) != 0 && (pins_enabled & pin_mask_5V) == 0) {
189  gpio_pin_set(POWER_PIN_PS_5V, false);
190  gpio_cfg_input(POWER_PIN_PS_5V);
191  }
192  if ((pin_mask_before & pin_mask_BAT) == 0 && (pins_enabled & pin_mask_BAT) != 0) {
193  gpio_cfg_output(POWER_PIN_PS_BAT);
194  gpio_pin_set(POWER_PIN_PS_BAT, true);
195  } else if ((pin_mask_before & pin_mask_BAT) != 0 && (pins_enabled & pin_mask_BAT) == 0) {
196  gpio_pin_set(POWER_PIN_PS_BAT, false);
197  gpio_cfg_input(POWER_PIN_PS_BAT);
198  }
199 
200 #ifdef POWERDEBUG_DISABLE_AIR
201  if (pin == POWER_PIN_AIR && fOn)
202  return;
203 #endif
204 
205 #endif // scv1
206 
207  // Turn the actual power pin on or off
208  gpio_pin_set(pin, fOn);
209 
210 }

221 // Handle motion detection
222 bool gpio_motion_sense(uint16_t command) {
223 #if !defined(MOTIONX)
224  return false;
225 #else
226  static bool fMotionArmed = false;
227  static bool fMotionSensed = false;
228 
229  switch (command) {
230 
231  // Arm the motion detection
232  case MOTION_ARM:
233  // If (and ONLY if) there had previously been motion sensed,
234  // refresh the GPS position.
235  if (fMotionSensed) {
236  // If we're in a mode supporting auto-motion, switch operating mode
237  if (storage()->wan == WAN_FONA_PLUS_MOBILE)
238  sensor_set_op_mode(OPMODE_NORMAL);
239  // Update the GPS plus stats
240  comm_gps_update();
241  stats()->motion_events++;
242  } else if (!fMotionArmed) {
243  DEBUG_PRINTF("Motion NOW ARMED\n");
244  }
245  // Prepare for sensing
246  fMotionSensed = false;
247  fMotionArmed = true;
248  break;
249 
250  // Disarm so that we won't pay any attention to motion
251  case MOTION_DISARM:
252  fMotionArmed = false;
253  break;
254 
255  case MOTION_QUERY_PIN:
256 #if defined(SENSE_PIN_MOTION) && defined(TWILIS3DH)
257  return(nrf_gpio_pin_read(SENSE_PIN_MOTION));
258 #else
259  return(false);
260 #endif
261 
262  // Update the state of the motion sensing, based on the motion pin
263  case MOTION_UPDATE:
264  if (fMotionArmed) {
265  bool fMotionNowSensed = false;
266 
267  // Sense the state of the pin if we're configured for it
268  fMotionNowSensed = gpio_motion_sense(MOTION_QUERY_PIN);
269 #ifdef MOTIONDEBUGMAX
270  if (fMotionNowSensed && debug(DBG_SENSOR))
271  DEBUG_PRINTF("Motion detected by GPIO\n");
272 #endif
273 
274  // If we've sensed motion, refresh the debounce timer
275  if (fMotionNowSensed)
276  MotionDetectedTime = get_seconds_since_boot();
277 
278  // Handle state changes
279  if (!fMotionSensed && fMotionNowSensed) {
280 
281  // Remember that it was sensed
282  fMotionSensed = true;
283  if (debug(DBG_SENSOR_MAX))
284  DEBUG_PRINTF("MOTION SENSED\n");
285 
286  // If we're in a mode supporting auto-motion, set operating mode
287  if (storage()->wan == WAN_FONA_PLUS_MOBILE)
288  sensor_set_op_mode(OPMODE_MOBILE);
289 
290  } else if (fMotionSensed && !fMotionNowSensed) {
291 
292  // Debounce. When it truly has stabilized, re-arm.
293  if (MotionDetectedTime != 0) {
294  if (!ShouldSuppress(&MotionDetectedTime, MOTION_STABLE_MINUTES * 60L)) {
295  gpio_motion_sense(MOTION_ARM);
296  MotionDetectedTime = 0;
297  }
298  }
299  }
300 
301  }
302  break;
303 
304  // See if motion is yet armed
305  case MOTION_QUERY_ARMED:
306  return fMotionArmed;
307 
308  }
309  return fMotionSensed;
310 #endif
311 }

313 // See if we're within the interval after motion was detected
314 bool gpio_in_motion() {
315  if (MotionDetectedTime == 0)
316  return false;
317  if (!WouldSuppress(&MotionDetectedTime, MOTION_STABLE_MINUTES * 60L))
318  return false;

391 // Primary app timer
392 #ifdef INDICATORS
393 void indicator_timer_handler(void *p_context) {
394  uint32_t color, mask;
395 
396  if (indicator_toggle == 1) {
397  indicator_toggle = 0;
398  mask = 0x01010100;
399  } else {
400  indicator_toggle = 1;
401  mask = 0x02020200;
402  }
403 
404  // Give priority to the COMM color unless it's still unknown
405  if (indicator_blinky)
406  color = BOTH_ALTERNATE;
407  else if ((indicator_comm_color & ~LEDMODE) != BLACK && (indicator_comm_color & ~LEDMODE) != BOTH_SOLID)
408  color = indicator_comm_color;
409  else
410  color = indicator_gps_color;
411  color &= mask;
412 
413  // Turn on or off the LEDs based on the mask, efficiently
414  bool newRed = (color & 0x0000ff00) != 0;
415  bool newYel = (color & 0x00ff0000) != 0;
416  if (!lastKnown) {
417 #ifdef LED_PIN_RED
418  gpio_pin_set(LED_PIN_RED, newRed);
419 #endif
420 #ifdef LED_PIN_YEL
421  gpio_pin_set(LED_PIN_YEL, newYel);
422 #endif
423  } else {
424 #ifdef LED_PIN_RED
425  if (newRed != lastRed)
426  gpio_pin_set(LED_PIN_RED, newRed);
427 #endif
428 #ifdef LED_PIN_YEL
429  if (newYel != lastYel)
430  gpio_pin_set(LED_PIN_YEL, newYel);
431 #endif
432  }
433  lastRed = newRed;
434  lastYel = newYel;
435  lastKnown = true;
436 
437 
438 }
439 #endif

441 // Turn indicators to a certain state
442 void gpio_indicate(uint32_t what) {
443 #ifdef INDICATORS
444 
445  // If this is a special request to flash so that we can identify the device, do it
446  if (what == INDICATE_BLINKY) {
447  indicator_blinky = true;
448  if (indicator_app_timer_started)
449  app_timer_stop(indicator_timer);
450  app_timer_start(indicator_timer, APP_TIMER_TICKS(150, APP_TIMER_PRESCALER), NULL);
451  indicator_app_timer_started = true;
452  indicator_shutdown = false;
453  }
454 
455  // Set the color so the next poll changes it
456  if ((what & COMM) != 0)
457  indicator_comm_color = what;
458  if ((what & GPS) != 0)
459  indicator_gps_color = what;
460 
461  // Turn things off if no longer needed
462  if (what == INDICATE_GPS_CONNECTED)
463  gpio_indicator_no_longer_needed(GPS);
464  else if (what == INDICATE_CELL_CONNECTED || what == INDICATE_LORA_CONNECTED || what == INDICATE_LORAWAN_CONNECTED)
465  gpio_indicator_no_longer_needed(COMM);
466 
467 #endif
468 }

547 // UART Selector
548 void gpio_uart_select(uint16_t which) {
549  uint32_t speed = UART_BAUDRATE_BAUDRATE_Baud57600;
550  bool hwfc = HWFC;
551  uint16_t prev_uart_selected = last_uart_selected;
552  last_uart_selected = which;
553 
554 #ifdef DEBUGSELECT
555  DEBUG_PRINTF("UART SELECT %s\n", gpio_uart_name(which));
556 #endif
557 
558  // Deconfigure the UART on the Nordic driver
559  serial_init(0, false);
560 
561  // Allow settling
562  nrf_delay_ms(500);
563 
564  // Power-off modules as appropriate
565 #if defined(LORA) && defined(POWER_PIN_LORA)
566  if (which != UART_LORA)
567  gpio_power_set(POWER_PIN_LORA, false);
568 #endif
569 #ifdef CELLX
570  if (which != UART_FONA)
571  gpio_power_set(POWER_PIN_CELL, false);
572 #endif
573 #ifdef UGPS
574  if (which != UART_GPS) {
575  // Note that we NEVER turn off the GPS while in mobile mode,
576  // so we don't lose our fix.
577  if (sensor_op_mode() != OPMODE_MOBILE)
578  gpio_power_set(POWER_PIN_GPS, false);
579  }
580 #endif
581 
582  // Disable all serial input coming through the mux
583 #ifdef USX
584  gpio_pin_set(UART_DESELECT, true);
585 #endif
586 
587  // Select the appropriate port on the (still-disabled) uart mux
588 #ifdef LORA
589  if (which == UART_LORA) {
590  hwfc = HWFC;
591  speed = UART_BAUDRATE_BAUDRATE_Baud57600;
592 #if defined(USX) && defined(USLORA)
593  gpio_pin_set(UART_SELECT_A, (UART_SELECT_PIN_A & USLORA) != 0);
594  gpio_pin_set(UART_SELECT_B, (UART_SELECT_PIN_B & USLORA) != 0);
595 #endif
596  }
597 #endif
598 #ifdef CELLX
599  if (which == UART_FONA) {
600  hwfc = HWFC;
601  speed = UART_BAUDRATE_BAUDRATE_Baud9600;
602 #if defined(USX) && defined(USFONA)
603  gpio_pin_set(UART_SELECT_A, (UART_SELECT_PIN_A & USFONA) != 0);
604  gpio_pin_set(UART_SELECT_B, (UART_SELECT_PIN_B & USFONA) != 0);
605 #endif
606  }
607 #endif
608 #if defined(PMSX) && PMSX==IOUART
609  if (which == UART_PMS) {
610  speed = UART_BAUDRATE_BAUDRATE_Baud9600;
611  hwfc = false;
612 #if defined(USX) && defined(USPMS)
613  gpio_pin_set(UART_SELECT_A, (UART_SELECT_PIN_A & USPMS) != 0);
614  gpio_pin_set(UART_SELECT_B, (UART_SELECT_PIN_B & USPMS) != 0);
615 #endif
616  }
617 #endif
618 #ifdef UGPS
619  if (which == UART_GPS) {
620  speed = UART_BAUDRATE_BAUDRATE_Baud9600;
621  hwfc = false;
622 #if defined(USX) && defined(USGPS)
623  gpio_pin_set(UART_SELECT_A, (UART_SELECT_PIN_A & USGPS) != 0);
624  gpio_pin_set(UART_SELECT_B, (UART_SELECT_PIN_B & USGPS) != 0);
625 #endif
626  }
627 #endif
628 
629  // Re-enable comms
630  if (which != UART_NONE) {
631 
632  // Initialize the UART
633  serial_init(speed, hwfc);
634 
635  // Allow serial traffic to stabilize after selecting speed
636  nrf_delay_ms(1000);
637 
638  // Enable the uart mux, which starts data flowing
639 #ifdef USX
640  gpio_pin_set(UART_DESELECT, false);
641 #endif
642 
643  }
644 
645  // Power-on modules as appropriate
646 #if defined(LORA) && defined(POWER_PIN_LORA)
647  if (which == UART_LORA)
648  gpio_power_set(POWER_PIN_LORA, true);
649 #endif
650 #ifdef CELLX
651  if (which == UART_FONA)
652  gpio_power_set(POWER_PIN_CELL, true);
653 #endif
654 #ifdef UGPS
655  if (which == UART_GPS)
