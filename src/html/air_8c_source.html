5 // Composite Virtual Particle Sensor
6 //
7 // I have gathered enough data that I've come to the conclusion that the most reliable way of
8 // gathering and comparing measurements from the two particle counters is to run them concurrently,
9 // having the PMS sensor "breathing the exhaust" of the OPC sensor. This works well because the
10 // OPC sensor's physical design is focused around inhanling air through a panel-mounted tube,
11 // exhaling it from its fan. The PMS sensor is not as carefully designed to separate intake from
12 // exhaust, and its design (with intake and exhaust relatively near each other) more-or-less suggests
13 // that it is measuring the nature of the ambient air surrounding it.
14 //
15 // In any case, this module creates a virtual "composite sensor" out of both the OPC and PMS sensors,
16 // ensuring that they run concurrently. It also handles various error paths in which one or the other
17 // sensors is either unconfigured or otherwise unavailable.

63 // Show the current value
64 bool s_air_show_value(uint32_t when, char *buffer, uint16_t length) {
65  bool is_opc, is_pms;
66  char msg_opc[128], msg_pms[128];
67  is_opc = s_opc_show_value(when, msg_opc, sizeof(msg_opc)-1);
68  is_pms = s_pms_show_value(when, msg_pms, sizeof(msg_pms)-1);
69  if (is_opc && is_pms) {
70  char msg[256];
71  sprintf(msg, "%s\n%s", msg_pms, msg_opc);
72  strlcpy(buffer, msg, length);
73  } else if (is_opc) {
74  strlcpy(buffer, msg_opc, length);
75  } else if (is_pms) {
76  strlcpy(buffer, msg_pms, length);
77  } else {
78  return false;
79  }
80  return true;
81 }

101 // Init sensor just after each power-on

105  if (s_opc_init(s, param))

109 #ifdef PMSX
110 
111  // In the case (specifically LoRaWAN mode) in which the UART cannot be switched
112  // to PMS even though sensor-defs.h "requested" it, don't even try. We will
113  // simply run with OPC. This should never happen because of code in the sensor
114  // scheduler that requests UART, but this is just defensive coding.
115 #if PMSX==IOUART
116  if (gpio_current_uart() != UART_PMS) {
117  if (debug(DBG_SENSOR))

119  return false;
120  } else {

122  pms_active = true;
123  }
124 #else
