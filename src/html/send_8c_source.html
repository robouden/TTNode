45 // Send buffer, for cellular use.
46 // The size allocated here was conservatively computed by assuming
47 // that generally the worst case is:
48 // - No more than 100 bytes generally per message, typically 25-75 bytes

111 // Get the stamp ID of a message. Don't call this unless it's stampable.
112 uint32_t stamp_id(ttproto_Telecast *message) {
113  char buffer[64];
114 
115  // Only include lat/lon when present and NOT in mobile mode
116  if (message->has_latitude && message->has_longitude && sensor_op_mode() != OPMODE_MOBILE)
117  sprintf(buffer, "%f,%f,%lu,%lu",
118  message->latitude,
119  message->longitude,
120  message->captured_at_date,
121  message->captured_at_time);
122  else
123  sprintf(buffer, "%lu,%lu",
124  message->captured_at_date,
125  message->captured_at_time);
126 
127  return(crc32_compute((uint8_t *)buffer, strlen(buffer), NULL));

129 }

131 // Create a stamp from the stamp fields
132 bool stamp_create(ttproto_Telecast *message) {
133  static uint32_t mobile_session_id = 12345; // init to something unlikely
134 
135  if (stampable(message)) {
136 
137  // If we're in mobile mode, do special processing
138  if (sensor_op_mode() == OPMODE_MOBILE) {
139 
140  // First, REMOVE the location info so that
141  // we can stamp messages that are in motion.
142  message->has_latitude = false;
143  message->has_longitude = false;
144  message->has_altitude = false;
145 
146  // Next, update the motion time offset ("Drive ID") if we've started a new session
147  if (mobile_session_id != sensor_get_mobile_session_id()) {
148  mobile_session_id = sensor_get_mobile_session_id();
149  mobile_session_time_offset = message->motion_began_offset = message->captured_at_offset;
150  }
151 
152  }
153 
154  // Create the stamp repeatedly even if it has changed, just to
155  // cover the (very rare) case where the server admin purges stamps
156  uint32_t message_id = stamp_id(message);
157 
158  // Save the stamp info locally
159  stamp_message_id = message_id;
160  stamp_message = *message;
161  stamp_message_valid = true;
162 
163  // Apply the stamp metadata so that the service stores it
164  message->stamp = stamp_message_id;
165  message->has_stamp = true;
166  message->stamp_version = STAMP_VERSION;
167  message->has_stamp_version = true;
168  return true;
169 
170  }
171  return false;
172 }

198  message->has_motion_began_offset = false;

200  message->has_test = false;
201 
202  return true;
203 
204  }
205  }
206 
207  return false;

211 // Reset the buffer
212 void send_buff_reset() {
213 
214  // No messages
215  buff_hdr[0] = BUFF_FORMAT_PB_ARRAY;
216  buff_hdr[1] = 0;
217  buff_hdr_used = sizeof(buff_hdr[0]) + sizeof(buff_hdr[1]);
218 
219  // Always start filling the buffer leaving room for the header
220  // which we will ultimately copy into the buffer before doing
221  // the UDP I/O.
222  buff_data_used = 0;
223  buff_data_left = sizeof(buff_data) - sizeof(buff_hdr);
224  buff_data_base = buff_data + sizeof(buff_hdr);
225  buff_pdata = buff_data_base;
226  buff_response_type = REPLY_NONE;
227 
228  // Done
229  buff_initialized = true;
230 
231 }

233 // Determine if we should avoid filling any more, out of caution
234 bool send_buff_is_full(uint16_t anticipated) {
235 
236  // Initialize if we've never done so
237  if (!buff_initialized)
238  send_buff_reset();
239 
240  // The "buffer left" is just a bit smaller than the
241  // full buffer, just to be conservative.
242  uint16_t max_buffer_size = sizeof(buff_data) - 250;
243 
244  // If we've already overflowed, indicate so
245  uint16_t hdr_anticipated = buff_hdr_used + (anticipated ? 1 : 0);
246  if ((hdr_anticipated + buff_data_used + anticipated) > max_buffer_size)
247  return true;
248 
249  // There's room
250  return false;
251 
252 }

258  if (!buff_initialized)

267 // Prepare the buff for writing
268 uint8_t *send_buff_prepare_for_transmit(uint16_t *lenptr, uint16_t *response_type_ptr) {
269 
270  // Copy the header to be contiguous with the data
271  uint8_t count = buff_hdr[1];
272  uint8_t header_size = sizeof(buff_hdr[0]) + sizeof(buff_hdr[1]) + count;
273  uint8_t *header = buff_data_base - header_size;
274  memcpy(header, buff_hdr, header_size);
275 
276  // Return the pointer to the buffer and length to be transmitted
277  if (lenptr != NULL)
278  *lenptr = header_size + buff_data_used;
279  if (response_type_ptr != NULL)
280  *response_type_ptr = buff_response_type;
281  return header;
282 
283 }

285 // Append a protocol buffer to the send buffer
286 bool send_buff_append(uint8_t *ptr, uint8_t len, uint16_t response_type) {
287 
288  // Initialize if we've never yet done so
289  if (!buff_initialized)
290  send_buff_reset();
291 
292  // Exit if we've exceeded MTU
293  if (send_buff_is_full(len))
294  return false;
295 
296  // Exit if we've appended too many
297  if (buff_hdr[1] >= (sizeof(buff_hdr) - (sizeof(buff_hdr[0]) + sizeof(buff_hdr[1])) ))
298  return false;
299 
300  // Exit if the body of the buffer is full
301  if (buff_data_left < len)
302  return false;
303 
304  // Remember these in case we need to pop this append
305  buff_pop_hdr = buff_hdr[1];
306  buff_pop_pdata = buff_pdata;
307  buff_pop_hdr_used = buff_hdr_used;
308  buff_pop_data_used = buff_data_used;
309  buff_pop_data_left = buff_data_left;
310  buff_pop_response_type = buff_response_type;
311 
312  // Append to the buffer
313  memcpy(buff_pdata, ptr, len);
314  buff_pdata += len;
315  buff_data_used += len;
316  buff_data_left -= len;
317 
318  // Append to the header
319  buff_hdr[1]++;
320  buff_hdr[sizeof(buff_hdr[0])+buff_hdr[1]] = len;
321  buff_hdr_used++;
322 
323  // Set response type, overriding NONE with what is desired
324  if (response_type != REPLY_NONE)
325  buff_response_type = response_type;
326 
327  // If we've buffered at least 3 items, force a reply response type simply because
328  // this means that we've been offline for quite a while and it would be good to give
329  // the service a chance to send us a command.
330  if (buff_hdr[1] > 3)
331  buff_response_type = REPLY_TTSERVE;
332 
333  // Done
334  return true;
335 
336 }

1391 // Transmit a binary message to the service on the currently-active transport, even
1392 // if the transport is busy or if we're in init.
1393 bool send_to_service_unconditionally(uint8_t *buffer, uint16_t length, uint16_t RequestType, uint16_t RequestFormat) {

1395  bool fTransmitted = false;
1396 
1397  // Ensure that we don't go recursive. This can happen if we try to transmit
1398  // something from a timer interrupt or bluetooth interrupt, when in fact
1399  // we are already trying to transmit something. Rather than queue it,
1400  // we just drop it on the floor. This is acceptable for the type of things
1401  // that we transmit.
1402  if (transmitInProgress++ != 0) {
1403  transmitInProgress--;
1404  return false;
1405  }
1406 
1407  // If the request format is to "send 1", we need to reformat it to be a "batch send" request because
1408  // that's the only format that we support as of March 2017. We do this by using the send buffer, which
1409  // we know isn't in use.
1410  if (RequestFormat == SEND_1) {
1411  send_buff_reset();
1412  send_buff_append(buffer, length, RequestType);
1413  buffer = send_buff_prepare_for_transmit(&length, &RequestType);
1414  }

1417  // when in mobile mode, although it works well in any mode.

1419  fTransmitted = db_put(buffer, length, RequestType);

1424  if (RequestFormat == SEND_1)
