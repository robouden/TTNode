84 // Transmit a byte to the LPWAN device
85 void serial_send_byte(uint8_t databyte) {
86  bool fTransmitted = false;
87 
88  // Exit if not initialized
89  if (!fSerialInit)
90  return;
91 
92  // Exit if we're temporarily disabled because
93  // we know that we'll cause a uart error if
94  // we try sending to the other device.
95  if (fTransmitDisabled)
96  return;
97 
98 #ifndef DISABLE_UART
99 
100  // Send it. Note that if we are linking against the NON-fifo version of app_uart,
101  // it is EXPECTED that this will come back with an error if we are busy.
102  // As such, we'll do a retry for a reasonable period of time. We don't want
103  // to hang here forever, though, because we might just be transmitting at a time
104  // when the target device is not selected by the uart multiplexer.
105  int i;
106  for (i=0; i<3; i++) {
107 
108  // It is expected that we will have material overrun problems on output when
109  // tthere is a large block of stuff being output (such as during bootloader).
110  // Since there's no app_uart equivalent of nrf_drv_uart_tx_in_progress(), we
111  // just do a nominal delay on EVERY byte transmitted. Given how little we
112  // transmit to Lora/Fona devices, this is not harmful even in the non-bootloader
113  // case. (Note that at 9600 baud, it takes roughly 1ms to transmit
114  // a single character - so this is quite a generous delay at that or faster
115  // baud rates.)
116  nrf_delay_ms(2);
117 
118  // Transmit, and we're done if it's successful
119  if (app_uart_put(databyte) == NRF_SUCCESS) {
120  fTransmitted = true;
121  break;
122  }
123 
124  }
125 
126 #endif // DISABLE_UART
127 
128  // Debugging
129  if (!fTransmitted)
130  DEBUG_PRINTF("SSB Error\n");
131 
132 #if defined(DEBUG_USES_UART) && !( defined(NSDKV10) || defined(NSDKV11) )
133  NRF_LOG_RAW_INFO("%c", databyte);
134 #endif
135 
136 }

138 // Check and clear uart errors
139 bool serial_uart_error_check(bool fClearOnly) {
140  bool wereErrors;
141 
142  // Silently clear out if that's what is being requested
143  if (fClearOnly) {
144 #ifdef SERIALRECEIVEDEBUG
145  debug_buff[0] = '\0';
146  debug_chars = 0;
147 #endif
148  uart_errors = 0;
149  return false;
150  }
151 
152  // Debug
153 
154 #ifdef SERIALRECEIVEDEBUG
155  if (debug_buff[0] != '\0') {
156  DEBUG_PRINTF("%d %s\n", debug_total, debug_buff);
157  debug_buff[0] = '\0';
158  debug_chars = 0;
159  }
160 #endif
161 
162  // We can't really do much without a UART, so be persistent about reporting
163  if (uart_init_err_code != 0)
164  DEBUG_PRINTF("UART init error: 0x%04x\n", uart_init_err_code);
165 
166  // If no uart is selected, ignore reporting the expected large # of errors
167  if (gpio_current_uart() == UART_NONE)
168  wereErrors = false;
169  else
170  wereErrors = (uart_errors > 5);
171 
172 #ifdef SERIALRECEIVEDEBUG
173  if (wereErrors)
174  DEBUG_PRINTF("%d more UART(%d) errors\n", uart_errors, gpio_current_uart());
175 #endif
176 
177  uart_errors = 0;
178  return(wereErrors);
179 }

193 // Handle serial input events
194 #ifndef DISABLE_UART
195 void uart_event_handler(app_uart_evt_t *p_event) {
196  uint8_t databyte;
197 
198  // Exit if we're in the middle of switching uarts
199  if (!fSerialInit)
200  return;
201 
202  switch (p_event->evt_type) {
203 
204  case APP_UART_DATA_READY: {
205  int i;
206  // We're called here at interrupt level, with serial
207  // interrupts disabled. On the one hand, if we only process
208  // a single byte per interrupt, we may never catch up
209  // if in fact the fifo starts to fill up and we are behind.
210  // On the other hand, if we stay here too long (such as a
211  // "while app_uart_get()==success"), we could cause significant
212  // problems staying in the interrupt handler for too long.
213  // This number is chosen as a "reasonable" number that
214  // enables us to consume several bytes from the rx queue
215  // without hanging us in here for too too long.
216  for (i=0; i<4; i++) {
217  if (app_uart_get(&databyte) != NRF_SUCCESS)
218  break;
219 #ifdef SERIALRECEIVEDEBUG
220  if (gpio_current_uart() != UART_NONE)
221  add_to_debug_log(databyte);
222 #endif
223  switch (gpio_current_uart()) {
224 #if defined(PMSX) && PMSX==IOUART
225  case UART_PMS:
226  pms_received_byte(databyte);
227  break;
228 #endif
229 #ifdef LORA
230  case UART_LORA:
231  lora_received_byte(databyte);
232  break;
233 #endif
234 #ifdef FONA
235  case UART_FONA:
236  fona_received_byte(databyte);
237  break;
238 #endif
239 #ifdef UGPS
240  case UART_GPS:
241  s_ugps_received_byte(databyte);
242  break;
243 #endif
244  }
245  }
246  break;
247  }
248 
249  case APP_UART_COMMUNICATION_ERROR:
250  if (gpio_current_uart() != UART_NONE)
251  uart_errors++;
252  break;
253 
254  case APP_UART_FIFO_ERROR:
255  DEBUG_CHECK(p_event->data.error_code);
256  break;
257 
258  default:
259  break;
260  }
261 }
262 #endif // DISABLE_UART

269 // Init the serial I/O subsystem
270 void serial_init(uint32_t speed, bool hwfc) {
271 
272  // Close it and re-open it if reinitializing
273  if (fSerialInit) {
274 
275  // If we are doing uart debugging, don't mess things up by doing a reset or init
276 #ifdef DEBUG_USES_UART
277  return;
278 #endif
279 
280  // Close the UART.
281  fSerialInit = false;
282  fTransmitDisabled = true;
283 #ifndef DISABLE_UART
284  app_uart_close();
285 #endif
286 
287  // Disable the pins, to ensure there is no leakage path
288  gpio_cfg_input(RX_PIN);
289  gpio_cfg_input(TX_PIN);
290 #if HWFC
291  gpio_cfg_input(RTS_PIN);
292  gpio_cfg_input(CTS_PIN);
293 #endif
294 
295  }
296 
297  // If we're just shutting down the UART, exit
298  if (speed == 0)
299  return;
300 
301  // Init
302 #ifndef DISABLE_UART
303 
304  // If not configured for HWFC, turn it off
305 #if !HWFC
306  hwfc = false;
307 #endif
308 
309 #if HWFC
310  app_uart_comm_params_t comm_params = {
311  RX_PIN,
312  TX_PIN,
313 #if ( defined(NSDKV10) || defined(NSDKV11) )
314  (uint8_t)RTS_PIN,
315  (uint8_t)CTS_PIN,
316 #else
317  RTS_PIN,
318  CTS_PIN,
319 #endif
320  APP_UART_FLOW_CONTROL_ENABLED,
321  false, // NO PARITY
322  speed
323  };
324 #else
325  app_uart_comm_params_t comm_params = {
326  RX_PIN,
327  TX_PIN,
328 #if ( defined(NSDKV10) || defined(NSDKV11) )
329  (uint8_t)UART_PIN_DISCONNECTED,
330  (uint8_t)UART_PIN_DISCONNECTED,
331 #else
332  UART_PIN_DISCONNECTED,
333  UART_PIN_DISCONNECTED,
334 #endif
335  APP_UART_FLOW_CONTROL_DISABLED,
336  false, // NO PARITY
337  speed
338  };
339 #endif
340 
341  if (!hwfc) {
342  comm_params.flow_control = APP_UART_FLOW_CONTROL_DISABLED;
343 #if defined(NSDKV10) || defined(NSDKV11)
344  comm_params.rts_pin_no = (uint8_t) UART_PIN_DISCONNECTED;
345  comm_params.rts_pin_no = (uint8_t) UART_PIN_DISCONNECTED;
346 #else
347  comm_params.rts_pin_no = UART_PIN_DISCONNECTED;
348  comm_params.cts_pin_no = UART_PIN_DISCONNECTED;
349 #endif
350  }
351 
352  // Save this so the higher levels can query it
353  fHWFC = hwfc;
354 
355  // Init the uart
356  app_uart_buffers_t buffer_params;
357  app_irq_priority_t priority;
358  static uint8_t rx_buf[UART_RX_BUF_SIZE];
359  static uint8_t tx_buf[UART_TX_BUF_SIZE];
360  buffer_params.rx_buf = rx_buf;
361  buffer_params.rx_buf_size = sizeof (rx_buf);
362  buffer_params.tx_buf = tx_buf;
363  buffer_params.tx_buf_size = sizeof (tx_buf);
364  priority = APP_IRQ_PRIORITY_LOW;
365  uart_init_err_code = app_uart_init(&comm_params, &buffer_params, uart_event_handler, priority);
366 
367 #ifdef SERIALRECEIVEDEBUG
368  char *baudstr, *flowstr;
369  switch (comm_params.flow_control) {
370  case APP_UART_FLOW_CONTROL_DISABLED:
371  flowstr = "disabled";
372  break;
373  case APP_UART_FLOW_CONTROL_ENABLED:
374  flowstr = "enabled";
375  break;
376  default:
377  flowstr = "?";
378  break;
379  }
380  switch (comm_params.baud_rate) {
381  case UART_BAUDRATE_BAUDRATE_Baud9600:
382  baudstr = "9600";
383  break;
384  case UART_BAUDRATE_BAUDRATE_Baud19200:
385  baudstr = "19200";
386  break;
387  case UART_BAUDRATE_BAUDRATE_Baud57600:
388  baudstr = "57600";
389  break;
390  case UART_BAUDRATE_BAUDRATE_Baud115200:
391  baudstr = "115200";
392  break;
393  default:
394  baudstr = "?";
395  break;
396  }
397  DEBUG_PRINTF("UART=%s hwfc=%s err=%d\n", baudstr, flowstr, uart_init_err_code);
398 #endif
399 
400 #endif // DISABLE_UART
401 
402  fSerialInit = true;
403  fTransmitDisabled = false;
404 
405 }
