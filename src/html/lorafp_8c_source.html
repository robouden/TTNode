419 void configureUS915(uint8_t fsb)
420 {
421  uint8_t ch;
422  uint8_t chLow = fsb > 0 ? (fsb - 1) * 8 : 0;
423  uint8_t chHigh = fsb > 0 ? chLow + 7 : 71;
424  uint8_t ch500 = fsb + 63;
425  for (ch = 0; ch < 72; ch++)
426  {
427  if (ch == ch500 || (ch <= chHigh && ch >= chLow))
428  {
429  sendChSet(MAC_CHANNEL_STATUS, ch, "on");
430  if (ch < 63)
431  {
432  sendChSet(MAC_CHANNEL_DRRANGE, ch, "0 3");
433  }
434  }
435  else
436  {
437  sendChSet(MAC_CHANNEL_STATUS, ch, "off");
438  }
439  }
440  sendMacSet(MAC_PWRIDX, TTN_PWRIDX_US915);
441 }

511 void sendCommand(uint8_t table, uint8_t index, bool appendSpace)
512 {
513  char command[100];
514  switch (table)
515  {
516  case MAC_TABLE:
517  strcpy_P(command, (char *)pgm_read_word(&(mac_table[index])));
518  break;
519  case MAC_GET_SET_TABLE:
520  strcpy_P(command, (char *)pgm_read_word(&(mac_options[index])));
521  break;
522  case MAC_JOIN_TABLE:
523  strcpy_P(command, (char *)pgm_read_word(&(mac_join_mode[index])));
524  break;
525  case MAC_CH_TABLE:
526  strcpy_P(command, (char *)pgm_read_word(&(mac_ch_options[index])));
527  break;
528  case MAC_TX_TABLE:
529  strcpy_P(command, (char *)pgm_read_word(&(mac_tx_table[index])));
530  break;
531  case SYS_TABLE:
532  strcpy_P(command, (char *)pgm_read_word(&(sys_table[index])));
533  break;
534  case RADIO_TABLE:
535  strcpy_P(command, (char *)pgm_read_word(&(radio_table[index])));
536  break;
537  default:
538  return;
539  }
540  modemStream->write(command);
541  if (appendSpace)
542  {
543  modemStream->write(" ");
544  }
545 }

560 bool sendChSet(uint8_t index, uint8_t channel, const char *value)
561 {
562  clearReadBuffer();
563  char ch[5];
564  if (channel > 9)
565  {
566  ch[0] = ((channel - (channel % 10)) / 10) + 48;
567  ch[1] = (channel % 10) + 48;
568  ch[2] = '\0';
569  }
570  else
571  {
572  ch[0] = channel + 48;
573  ch[1] = '\0';
574  }
575  debugPrint(F(SENDING));
576  sendCommand(MAC_TABLE, MAC_PREFIX, true);
577  sendCommand(MAC_TABLE, MAC_SET, true);
578  sendCommand(MAC_GET_SET_TABLE, MAC_CH, true);
579  sendCommand(MAC_CH_TABLE, index, true);
580  modemStream->write(ch);
581  modemStream->write(" ");
582  modemStream->write(value);
583  modemStream->write(SEND_MSG);
584  debugPrint(channel);
585  debugPrint(F(" "));
586  debugPrintLn(value);
587  return waitForOk();
588 }
