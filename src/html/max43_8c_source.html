31 // MAX17043 Data Sheet
32 // https://datasheets.maximintegrated.com/en/ds/MAX17043-MAX17044.pdf
33 
34 // MAX17043 7-Bit I2C Address
35 #define MAX17043_I2C_ADDRESS 0x36
36 #define MAX17043_ADDRESS_LEN 1
37 #define MAX17043_DATA_LEN 2
38 
39 // MAX17043 Register Definitions - all contain 2 bytes of data and span 2 addresses
40 // R - 12-bit A/D measurement of battery voltage
41 #define MAX17043_VCELL 0x02
42 // R - 16-bit state of charge (SOC)
43 #define MAX17043_SOC 0x04
44 // W - Sends special commands to IC
45 #define MAX17043_MODE 0x06
46 
47 // I/O buffer
48 typedef struct {
49  // Address for TWI transfer
50  uint8_t address[MAX17043_ADDRESS_LEN];
51  // Buffer used for transfers (we only use a few bytes of it)
52  uint8_t buffer[MAX17043_DATA_LEN];
53  // State
54  bool batteryReportedVoltage;
55  float batteryVoltage;
56  bool batteryReportedSOC;
57  float batterySOC;
58 } max17043_data_t;
59 static max17043_data_t ioVoltage;
60 static max17043_data_t ioSOC;
61 
62 // TWI initialization
63 static bool fTWIInitV = false;
64 static bool fTWIInitS = false;

66 // Callback when TWI data has been read, or timeout
67 void voltage_callback(ret_code_t result, twi_context_t *t) {
68  float voltage;
69  uint16_t vCell;
70 
71  if (!twi_completed(t)) {
72  sensor_measurement_completed(t->sensor);
73  return;
74  }
75 
76  // Extract battery voltage, noting that MSB comes before LSB on TWI
77  vCell = (ioVoltage.buffer[0] << 8) | ioVoltage.buffer[1];
78  vCell = (vCell) >> 4;
79  voltage = (float) vCell / 800.0;
80  ioVoltage.batteryVoltage = voltage;
81  ioVoltage.batteryReportedVoltage = true;
82 
83  // Flag that this I/O has been completed.
84  sensor_measurement_completed(t->sensor);
85 }

92 // Measure voltage
93 void s_max43_voltage_measure(void *s) {
94  ioVoltage.address[0] = MAX17043_VCELL;
95  static app_twi_transfer_t const transfers[] = {
96  APP_TWI_WRITE(MAX17043_I2C_ADDRESS, &ioVoltage.address[0], MAX17043_ADDRESS_LEN, APP_TWI_NO_STOP),
97  APP_TWI_READ(MAX17043_I2C_ADDRESS, &ioVoltage.buffer[0], MAX17043_DATA_LEN, 0)
98  };
99  static app_twi_transaction_t const transaction = {
100  .callback = twi_callback,
101  .p_user_data = "MAX43-V",
102  .p_transfers = transfers,
103  .number_of_transfers = sizeof(transfers) / sizeof(transfers[0])
104  };
105  if (!twi_schedule(s, voltage_callback, &transaction))
106  sensor_unconfigure(s);
107 }

141 // Callback when TWI data has been read
142 void soc_callback(ret_code_t result, twi_context_t *t) {
143  float percent;
144  uint16_t soc;
145 
146  if (!twi_completed(t)) {
147  sensor_measurement_completed(t->sensor);
148  return;
149  }
150 
151  // Extract battery SOC, noting that MSB comes before LSB on TWI
152  soc = (ioSOC.buffer[0] << 8) | ioSOC.buffer[1];
153  percent = (soc & 0xFF00) >> 8;
154  percent += (float) (((uint8_t) soc) / 256.0);
155  ioSOC.batterySOC = percent;
156  ioSOC.batteryReportedSOC = true;
157  battery_set_soc(ioSOC.batterySOC);
158 
159  // Flag that this I/O has been completed.
160  sensor_measurement_completed(t->sensor);
161 }
