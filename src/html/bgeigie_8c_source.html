39 // Process a completed bGeigie command
40 void bgeigie_complete() {
41 
42  // If we're not in an idle state, let's not process commands for it
43  if (fromGeigie.state != COMM_STATE_IDLE)
44  return;
45 
46  // Process incoming from Instrument that we did or didn't expect in an IDLE state
47  if (comm_cmdbuf_this_arg_is(&fromGeigie, "$*")) {
48  // Leave nextarg set up as the entire first argument to the NMEA string, ready for parsing.
49  comm_cmdbuf_this_arg_is(&fromGeigie, "*");
50  fromGeigie.state = COMM_BGEIGIE_XMIT;
51  } else {
52  fromGeigie.state = COMM_BGEIGIE_UNKNOWN;
53  }
54 
55 }

119  UNUSED_VARIABLE(RadiationiValid);
120  UNUSED_VARIABLE(NumSat);
121  UNUSED_VARIABLE(HDOP);
122 
123  // Onliy do all this work if communicating wouldn't be pointless
124  if (!comm_is_busy()) {
125 
126  // See if we should suppress it, else transmit it
127  SuppressionSeconds = geigieSuppressionSeconds + io_get_random(DESYNCHRONIZATION_SECONDS);
128 
129  if (!ShouldSuppress(&lastGeigieTransmitTime, SuppressionSeconds)) {
130  uint16_t status;
131  float fLatitude, fLongitude;
132 
133  // Allocate space on the stack to store the message data.
134  ttproto_Telecast message = ttproto_Telecast_init_zero;
135

138

145  message.device_id = DeviceID;

148  strlcpy(message.captured_at, DateTimeISO, sizeof(message.captured_at));
149 
150  message.has_lnd_7318u = true;
151  message.lnd_7318u = atoi(RadiationCPM);
152 
153  fLatitude = GpsEncodingToDegrees(Latitude, LatitudeNS);
154  fLongitude = GpsEncodingToDegrees(Longitude, LongitudeEW);
155  if (fLatitude != 0 && fLongitude != 0 && GPSValid) {
156  message.latitude = fLatitude;
157  message.longitude = fLongitude;
158  message.altitude = atoi(Altitude);
159  message.has_latitude = message.has_longitude = message.has_altitude = true;
160  }
161 
162  /* Encode and transmit the message */
163  status = pb_encode(&stream, ttproto_Telecast_fields, &message);
164  if (!status)
165  DEBUG_PRINTF("pb_encode: %s\n", PB_GET_ERROR(&stream));
166  else {

169  }
170 
171  }
172 
173  }
174 
175  // Done
176  comm_cmdbuf_set_state(&fromGeigie, COMM_STATE_IDLE);
177  break;
178 
179  }

181  } //switch
