46 void btdebug_send_string(char *str) {
47 
48  // Exit if not yet initialized
49  if (!init)
50  return;
51 
52  // Short circuit all of this if we can't even send to BT
53  if (!can_send_to_bluetooth())
54  return;
55 
56  // Send welcome message immediately (yes, this goes recursive)
57  welcome_message();
58 
59  // If we're optimizing power, we don't even want to come down this path
60  // because we're wasting time. Just swallow the character.
61  if (io_optimize_power())
62  return;
63 
64  // This is completely defensive programming. There is no way that
65  // we should ever go recursive, but this is for future-proofing.
66  if (recursion++ != 0) {
67  --recursion;
68  return;
69  }
70 
71  // Loop over input chars
72  while (*str != '\0') {
73 
74 #ifdef BTDEBUG_BYPASS_BUFFERING
75 
76  // This causes huge problems if done at driver level, but
77  // occasionally this mode can be useful when debugging
78  // things that crash the MCU and thus don't give sufficient
79  // time for the output to get to bluetooth.
80  send_byte_to_bluetooth(*str++);
81  UNUSED_VARIABLE(output_buffer_fill_next);
82 
83 #else
84 
85  // Exit if output buffer overrun
86  if (output_buffer_used >= sizeof(output_buffer)) {
87  break;
88  }
89 
90  // Append to output buffer
91  output_buffer[output_buffer_fill_next++] = *str++;
92  if (output_buffer_fill_next >= sizeof(output_buffer))
93  output_buffer_fill_next = 0;
94  output_buffer_used++;
95 
96 #endif
97 
98  }
99 
100  // If we need the timer, start it.
101 #ifndef BTDEBUG_BYPASS_BUFFERING
102  if (output_buffer_used != 0 && !timer_started) {
103  if (NRF_SUCCESS == app_timer_start(btdebug_timer, BTDEBUG_TIMER_INTERVAL, NULL))
104  timer_started = true;
105 #ifdef INDICATORS
106  if (!gpio_indicators_are_active())
107  gpio_pin_set(LED_PIN_RED, true);
108 #endif
109  }
110 #endif
111 
112  // Done
113  --recursion;
114 
115 }

117 // Timer
118 void btdebug_timer_handler(void *p_context) {
119 
120  // Exit if we're already in here
121  if (timer_interrupt_being_serviced++ != 0) {
122  --timer_interrupt_being_serviced;
123  return;
124  }
125 
126  // If nothing is left after debouncing, shut down the timer
127  if (output_buffer_used == 0) {
128 
129  // Exit if we're still debouncing
130  if (timer_debounce_count != 0) {
131  if (--timer_debounce_count != 0) {
132  --timer_interrupt_being_serviced;
133  return;
134  }
135  }
136 
137  // Disable the timer
138  if (timer_started) {
139  timer_started = false;
140  app_timer_stop(btdebug_timer);
141  }
142 #ifdef INDICATORS
143  if (!gpio_indicators_are_active())
144  gpio_pin_set(LED_PIN_RED, false);
145 #endif
146  --timer_interrupt_being_serviced;
147  return;
148 
149  }
150 
151  // Since we've got something to output, reset the debounce timer
152  timer_debounce_count = BTDEBUG_TIMER_DEBOUNCE_MILLISECONDS / BTDEBUG_TIMER_MILLISECONDS;
153 
154  // Drain until the earlier of empty or asked to pause
155  while (output_buffer_used) {
156 
157  // Grab the next byte
158  uint8_t databyte = output_buffer[output_buffer_drain_next++];
159  if (output_buffer_drain_next >= sizeof(output_buffer))
160  output_buffer_drain_next = 0;
161  output_buffer_used--;
162 
163  // Send it, but (and this is defensive coding) don't allow any
164  // output to be appended while we are inside the bluetooth subsystem.
165  recursion++;
166  bool fPause = send_byte_to_bluetooth(databyte);
167  recursion--;
168 
169  // Force a pause if instructed to do so
170  if (!fPause)
171  break;
172 
173  }
174 
175  // Done
176  --timer_interrupt_being_serviced;
177 
178 }
