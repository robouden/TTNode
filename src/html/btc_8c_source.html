27 // Function for handling Handle Value Notification received from the SoftDevice.
28 void on_hvx(btc_t *p_btc, ble_evt_t *p_ble_evt) {
29  ble_gattc_evt_hvx_t *p_notif = &p_ble_evt->evt.gattc_evt.params.hvx;
30 
31  // Check if this is OUR notification
32  if (p_ble_evt->evt.gattc_evt.params.hvx.handle == p_btc->rx_handle) {
33 
34  // Once we've received data, flag that we do NOT want to optimize power
35  // by shutting down BT listens, even if this particular connection
36  // happens to drop.
37  io_power_stay_suboptimal();
38 
39  /* Received data! */
40  if (comm_can_send_to_service())
41  for (uint32_t i = 0; i < p_notif->len; i++)
42  bgeigie_received_byte(p_notif->data[i]);
43 
44  /* Dispatch, just for good measure */
45  btc_evt_t btc_evt;
46  btc_evt.evt_type = BTC_EVT_NOTIFICATION;
47  p_btc->evt_handler(p_btc, &btc_evt);
48 
49  }
50 
51 }

59 // BTC event handler called from our general BT event handler
60 void btc_on_ble_evt(btc_t *p_btc, ble_evt_t *p_ble_evt) {
61 
62  if ((p_btc == NULL) || (p_ble_evt == NULL)) {
63  return;
64  }
65 
66  switch (p_ble_evt->header.evt_id) {
67  case BLE_GAP_EVT_CONNECTED:
68  p_btc->conn_handle = p_ble_evt->evt.gap_evt.conn_handle;
69  break;
70 
71  case BLE_GATTC_EVT_HVX:
72  on_hvx(p_btc, p_ble_evt);
73  break;
74 
75  case BLE_GATTC_EVT_WRITE_RSP:
76  on_write_response(p_btc, p_ble_evt);
77  break;
78 
79  default:
80  break;
81  }
82 }

84 // Handles events for the controller
85 void btc_evt_handler(btc_t *p_btc, btc_evt_t *p_btc_evt) {
86 
87  switch (p_btc_evt->evt_type) {
88  case BTC_EVT_DISCOVERY_COMPLETE: {
89 
90  if (debug(DBG_BT))
91  DEBUG_PRINTF("BTC service discovered\r\n");
92 
93  // Initiate bonding.
94 #ifndef NOBONDING
95  {
96  ret_code_t err_code;
97  err_code = pm_conn_secure(p_btc->conn_handle, false);
98  if (err_code != NRF_ERROR_INVALID_STATE) {
99  DEBUG_CHECK(err_code);
100  }
101  }
102 #endif
103 
104  // Service discovered. Configure the CCCD to enable HVX notification.
105  ret_code_t err_code;
106  err_code = cccd_configure(p_btc->conn_handle, p_btc->cccd_handle, true);
107  DEBUG_CHECK(err_code);
108 
109  }
110  break; // BTC_EVT_DISCOVERY_COMPLETE
111 
112  case BTC_EVT_NOTIFICATION: {
113  // Nothing to do for now because it's taken care of in hvx handler
114  } break; // BTC_EVT_NOTIFICATION
115 
116  default:
117  if (debug(DBG_BT))
118  DEBUG_PRINTF("BTC Service event ID=%d\n", p_btc_evt->evt_type);
119  // No implementation needed.
120  break;
121  }
122 }

124 // BTC initialization
125 void btc_init(btc_t *p_btc) {
126  uint32_t err_code;
127  uint8_t *uuid128;
128  ble_uuid128_t btc_service_uuid = BTC_SERVICE_UUID;
129  ble_uuid128_t btc_rx_uuid = BTC_RX_CHAR_UUID;
130 
131  // Add custom base UUIDs by pulling out the magic numbers that
132  // the bluetooth stack utilizes as "shortcut" UUID's rather than
133  // the full 128-bit UUIDs. The only time that this will break
134  // down is if we ever have a true conflict between those exact
135  // bytes between multiple UUID's that we're trying to register.
136  uuid128 = (uint8_t *) &btc_service_uuid;
137  p_btc->service_uuid.uuid = uuid128[12] | (uuid128[13] << 8);
138  err_code = sd_ble_uuid_vs_add(&btc_service_uuid, &p_btc->service_uuid.type);
139  DEBUG_CHECK(err_code);
140  uuid128 = (uint8_t *) &btc_rx_uuid;
141  p_btc->rx_uuid.uuid = uuid128[12] | (uuid128[13] << 8);
142  err_code = sd_ble_uuid_vs_add(&btc_rx_uuid, &p_btc->rx_uuid.type);
143  DEBUG_CHECK(err_code);
144 
145  // Initialize discovery
146  {
147 #ifdef NSDKV10
148  ret_code_t errcode = ble_db_discovery_init();
149 #else
150  ret_code_t errcode = ble_db_discovery_init(db_discover_evt_handler);
151 #endif
152  DEBUG_CHECK(errcode);
153  }
154 
155  // Begin discovery to try to find the service and its attributes
156  p_btc->evt_handler = btc_evt_handler;
157  p_btc->conn_handle = BLE_CONN_HANDLE_INVALID;
158  p_btc->cccd_handle = BLE_GATT_HANDLE_INVALID;
159 #ifdef NSDKV10
160  err_code = ble_db_discovery_evt_register(&p_btc->service_uuid, db_discover_evt_handler);
161 #else
162  err_code = ble_db_discovery_evt_register(&p_btc->service_uuid);
163 #endif
164  DEBUG_CHECK(err_code);
165 
166 }
