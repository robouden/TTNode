497 
498  switch (ble_adv_evt) {
499  case BLE_ADV_EVT_FAST:
500  if (debug(DBG_BT))
501  DEBUG_PRINTF("Advertising\n");
502  break;
503  case BLE_ADV_EVT_IDLE: {
504  ret_code_t err_code;
505  if (debug(DBG_BT))
506  DEBUG_PRINTF("Advertising idle; restarting\n");
507  m_advertising = false;
508  if (!io_optimize_power()) {

511  DEBUG_CHECK(err_code);
512  m_advertising = true;
513  }
514  break;
515  }
516  default:
517  if (debug(DBG_BT))
518  DEBUG_PRINTF("Advertising event: %d\n", ble_adv_evt);
519  break;
520  }
521 }

1618 // a string. The string will be be sent over BLE when the last character received was a
