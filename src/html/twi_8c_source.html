54 // Allocate a context block
55 static twi_context_t *find_transaction(char *c) {
56  int i;
57 
58  // Attempt to find the transaction
59  for (i=0; i<(sizeof(transaction) / sizeof(transaction[0])); i++)
60  if (transaction[i].comment != NULL)
61  if (strcmp(transaction[i].comment, c) == 0)
62  return &transaction[i];
63 
64  // Allocate a slot
65  for (i=0; i<(sizeof(transaction) / sizeof(transaction[0])); i++)
66  if (transaction[i].comment == NULL) {
67  transaction[i].index = i;
68  transaction[i].comment = c;
69  transaction[i].transactions_scheduled = 0;
70  transaction[i].transactions_completed = 0;
71  transaction[i].sensor = NULL;
72  transaction[i].callback = NULL;
73  return &transaction[i];
74  }
75 
76  // Can't happen
77  while (true) {
78  DEBUG_PRINTF("*** Array not big enough for the number of types of transaction\n");
79  nrf_delay_ms(500);
80  }
81 
82 }

90 // Bump errors and append to error log
91 void report_err() {
92  int i;
93  stats()->errors_twi++;
94  stats()->errors_twi_info[0] = '\0';
95  for (i=0; i<(sizeof(transaction) / sizeof(transaction[0])); i++)
96  if (transaction[i].comment != NULL && transaction[i].comment[0] != '~' && transaction[i].callback != NULL) {
97  if (transaction[i].sched_error || transaction[i].transaction_error) {
98  char buff[40];
99  sprintf(buff, "%s%s:%s%ld",
100  stats()->errors_twi_info[0] == '\0' ? "" : " ",
101  transaction[i].comment,
102  transaction[i].sched_error ? "S" : "C",
103  transaction[i].sched_error ? transaction[i].sched_error : transaction[i].transaction_error);
104  // Only copy whole errors into the buffer
105  if ((strlen(stats()->errors_twi_info)+strlen(buff)) < (sizeof(stats()->errors_twi_info)-2))
106  strcat(stats()->errors_twi_info, buff);
107  }
108  }

111 // Check the state of TWI
112 void twi_status_check(bool fVerbose) {
113  static uint32_t suppress = 0;
114 
115  // Don't allow recursion because of DEBUG_PRINTF
116  disable_twi_debug_printf++;
117 
118  // Output critical debug messages periodically
119  if (!ShouldSuppress(&suppress, 300))
120  if (stats()->errors_twi_info[0] != '\0')
121  DEBUG_PRINTF("TWI errors: %s\n", stats()->errors_twi_info);
122 
123  if (fVerbose) {
124  DEBUG_PRINTF("TWI idle=%d init=%d t=%d se=%d ce=%d %s\n", app_twi_is_idle(&m_app_twi), InitCount, TransactionsInProgress, SchedulingErrors, CompletionErrors, stats()->errors_twi_info);
125  }
126 
127  // Display TWI transaction table
128  if (fVerbose) {
129  int i;
130  char buffer[512];
131  buffer[0] = '\0';
132  for (i=0; i<(sizeof(transaction) / sizeof(transaction[0])); i++)
133  if (transaction[i].comment != NULL && transaction[i].comment[0] != '~' && transaction[i].callback != NULL) {
134  char buff2[128];
135  sprintf(buff2, "%s(%ld/%ld:%ld/%ld) ", transaction[i].comment, transaction[i].transactions_scheduled, transaction[i].transactions_completed, transaction[i].sched_error, transaction[i].transaction_error);
136  strcat(buffer, buff2);
137  }
138  DEBUG_PRINTF("%s\n", buffer);
139  }
140 
141  // See if any pending transactions are in a stuck state
142  bool fTerminatedTWI = false;
143  if (TransactionsInProgress != 0) {
144  int i, j;
145  for (i=0; i<(sizeof(transaction) / sizeof(transaction[0])); i++) {
146  twi_context_t *t = &transaction[i];
147  // If the comment begins with "~", suppress errors and TWI reset
148  if (t->transaction_began != 0 && t->comment != NULL && t->comment[0] != '~') {
149  if (!WouldSuppress(&t->transaction_began, 30)) {
150  DEBUG_PRINTF("*** %s HUNG: unconfiguring, resetting TWI ***\n", t->comment);
151  // Substitute a special timeout error if we hang
152  if (t->transaction_error == NRF_SUCCESS) {
153  t->transaction_error = 99;
154  report_err();
155  }
156  // Unconfigure the sensor if not in burn test mode, else merely get it unstuck
157  if (t->sensor != NULL)
158  sensor_unconfigure(t->sensor);
159  if (!fTerminatedTWI) {
160  fTerminatedTWI = true;
161  // Drain all TWI inits to zero
162  while (twi_term());
163 #ifdef SSD
164  // Reset the display subsystem, because it cannot be deconfigured and
165  // if TWI is reset out from under it there is havoc.
166  ssd1306_force_reset();
167 #endif
168  // Abort in-progress transactions for ALL twi-based sensors
169  for (j=0; j<(sizeof(transaction) / sizeof(transaction[0])); j++)
170  if (transaction[j].transaction_began != 0) {
171  transaction[j].transaction_began = 0;
172  if (t->sensor != NULL)
173  sensor_abort(t->sensor);
174  }
175  }
176  // Clear local counters
177  InitCount = 0;
178  TransactionsInProgress = 0;
179  SchedulingErrors = CompletionErrors = 0;
180  }
181  }
182  }
183  }
184 
185  disable_twi_debug_printf--;
186  return;
187 }

198 // Our universal callback
199 void twi_callback(ret_code_t result, void *p_user_data) {
200 
201  // Don't allow recursion because of DEBUG_PRINTF
202  disable_twi_debug_printf++;
203 
204  // Find the transaction
205  twi_context_t *t = find_transaction(p_user_data);
206 
207  // Mark it as completed
208  t->transaction_began = 0;
209  t->transaction_error = result;
210  t->transactions_completed++;
211 
212  // Call the callback at app_sched level if we can
213 #ifdef TWI_APP_SCHED
214  uint16_t index = t->index;
215  if (app_sched_event_put(&index, sizeof(index), callback_sched) != NRF_SUCCESS) {
216  t->callback(result, t);
217  }
218 #else
219  t->callback(result, t);
220 #endif
221 
222  // Done
223  disable_twi_debug_printf--;
224 }

226 // Schedule a TWI transaction
227 bool twi_schedule(void *sensor, sensor_callback_t callback, app_twi_transaction_t const * p_transaction) {
228  int i;
229  twi_context_t *t;
230 
231  // Don't allow recursion because of DEBUG_PRINTF
232  disable_twi_debug_printf++;
233 
234  // Find this transaction
235  t = find_transaction(p_transaction->p_user_data);
236 
237  // This is a bug check that prevents one TWI transaction from being scheduled on top of
238  // an instance of itself. This will only protect us a single time, because we set the
239  // transaction_began to 0, however it is better than blocking TWI transactions indefinitely.
240  if (t->transaction_began != 0) {
241  DEBUG_PRINTF("%s TWI double-schedule\n", p_transaction->p_user_data);
242  nrf_delay_ms(250);
243  t->transaction_began = 0;
244  disable_twi_debug_printf--;
245  return false;
246  }
247  t->sensor = sensor;
248  t->callback = (app_twi_callback_t) callback;
249  for (i=0; i<5; i++) {
250  t->sched_error = app_twi_schedule(&m_app_twi, p_transaction);
251  if (t->sched_error != NRF_ERROR_BUSY)
252  break;
253  DEBUG_PRINTF("%s busy\n", p_transaction->p_user_data);
254  nrf_delay_ms(500);
255  }
256  if (t->sched_error != NRF_SUCCESS) {
257  SchedulingErrors++;
258  report_err();
259  disable_twi_debug_printf--;
260  return false;
261  }
262  t->transactions_scheduled++;
263  t->transaction_began = get_seconds_since_boot();
264  TransactionsInProgress++;
265  disable_twi_debug_printf--;
266  return true;
267 }

271
