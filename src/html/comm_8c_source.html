128 // Reset a command buffer
129 void comm_cmdbuf_reset(cmdbuf_t *cmd) {
130  uint8_t databyte;
131 
132  // Reset the buffer
133  cmd->length = 0;
134  cmd->buffer[0] = '\0';
135  cmd->args = 0;
136  cmd->complete = false;
137 
138  // If there was anything waiting in the busy buffer, process it.
139  while (cmd->busy_length) {
140  cmd->busy_length--;
141  databyte = cmd->busy_buffer[cmd->busy_nextget++];
142  if (cmd->busy_nextget >= sizeof(cmd->busy_buffer))
143  cmd->busy_nextget = 0;
144  if (comm_cmdbuf_received_byte(cmd, databyte))
145  break;
146  }
147 
148 }

150 // Set command buffer state
151 void comm_cmdbuf_set_state(cmdbuf_t *cmd, uint16_t newstate) {
152 
153  // Do watchdog processing on the LPWAN receive buffer.
154  // Remember the last time we changed the state of the
155  // LPWAN, and set the watchdog. If we fail to change
156  // state it means that we're stuck for some reason.
157  // For example, we've observed that sometimes a
158  // communications error will cause a failure between
159  // the tx1 and tx2 state, in which case we will
160  // never proceed. Since there is ALWAYS activity
161  // that should be occurring on this port, i.e.
162  // timeouts, we just use a sledgehammer if the
163  // watchdog expires.
164  if (cmd->state != newstate) {
165 #ifdef LORA
166  if (cmd->type == CMDBUF_TYPE_LORA)
167  lora_watchdog_reset();
168 #endif
169 #ifdef FONA
170  if (cmd->type == CMDBUF_TYPE_FONA)
171  fona_watchdog_reset();
172 #endif
173  }
174 
175  // Reset the buffer and set its new state
176  comm_cmdbuf_reset(cmd);
177  cmd->state = newstate;
178 
179  // For multi-reply commands, reset what we've received
180  cmd->recognized = 0;
181 
182 }

184 // Append a byte to a command buffer
185 bool comm_cmdbuf_append(cmdbuf_t *cmd, uint8_t databyte) {
186 
187  // Exit if not even initialized yet
188  if (!commInitialized)
189  return false;
190 
191  // If we're already complete, we need to buffer it.
192  if (cmd->complete) {
193  if (cmd->busy_length < sizeof(cmd->busy_buffer)) {
194  cmd->busy_buffer[cmd->busy_nextput++] = databyte;
195  if (cmd->busy_nextput >= sizeof(cmd->busy_buffer))
196  cmd->busy_nextput = 0;
197  cmd->busy_length++;
198  return false;
199  } else {
200  // Busy buffer overflow - just reset it and cause data loss
201  // because there's really nothing else we can do.
202  cmd->busy_length = 0;
203  cmd->busy_nextput = 0;
204  return false;
205  }
206  }
207 
208  // If it's a newline AND not a blank line AND within our line length limits, we're done
209  if (databyte == '\n') {
210  if (cmd->length != 0) {
211  cmd->complete = true;
212  return(true);
213  }
214  // Blank line
215  return(false);
216  }
217 
218  // Only append it if it's simple ASCII
219  if (databyte >= 0x20 && databyte < 0x7f) {
220  cmd->buffer[cmd->length++] = databyte;
221 
222  // If we've overflowed the buffer, it's an error, so let's take some intentional data loss
223  // rather than enqueueing bad work to be done. Very specifically, if we get flooded with
224  // data for whatever reason that is both within the ASCII range and isn't terminated
225  // by a newline, there's a possibility that we might either overflow the busy buffer
226  // or enqueue scheduler put's far too fast to handle. Just shunt this crap to ground.
227  if (cmd->length >= CMD_MAX_LINELENGTH)
228  cmd->length = 0;
229 
230  // Always leave the buffer null-terminated, so we can use string operations during parsing
231  cmd->buffer[cmd->length] = '\0';
232 
233  }
234 
235  // Not yet complete
236  return(false);
237 
238 }

240 // Test to see if a character is a separator for the purpose of args
241 bool isArgSeparator(uint8_t databyte, bool embeddedSpaces) {
242 
243  // Obvious separators
244  if (!embeddedSpaces && databyte == ' ')
245  return true;
246  if (databyte == ',' || databyte == ';')
247  return true;
248 
249  // Treat trash as a separator
250  if (databyte < 0x20 || databyte >= 0x7f)
251  return (true);
252 
253  // Not a separator
254  return (false);
255 
256 }

258 // Test to see if the current command argument matches a specific lowercase string.
259 // The behavior is normally to look for a full delimited word, but there are two special modes:
260 // 1) a "xxx*" wildcard mode that matches anything beginning with that string
261 // 2) a "*" 'token mode' that matches any whole word up to the next delimeter, null-terminating it
262 bool comm_cmdbuf_this_arg_is(cmdbuf_t *cmd, char *testCmd) {
263  bool testForWord, tokenMode, embeddedSpaces;
264  int i, testLen;
265 
266  // First, see if there are embedded spaces in testCmd.
267  testLen = strlen(testCmd);
268  embeddedSpaces = false;
269  for (i=0; i<testLen; i++) {
270  if (testCmd[i] == ' ') {
271  embeddedSpaces = true;
272  break;
273  }
274  }
275 
276  // See if we need to test for a space afterward
277  testForWord = true;
278  tokenMode = false;
279  if (testCmd[testLen - 1] == '*') {
280  testLen--;
281  // Does the asterisk mean 'wildcard mode', or 'token mode'?
282  if (testLen != 0)
283  testForWord = false;
284  else
285  tokenMode = true;
286  }
287 
288  // This method always leaves 'nextarg' pointing at the next thing to parse, or end-of-buffer
289  cmd->nextarg = cmd->args;
290 
291  // If looking for a testcmd match, start by scanning for testcmd in a case-insensitive matter
292  if (!tokenMode) {
293 
294  // Not enough room left in the command buffer.
295  if (testLen > (cmd->length - cmd->args)) {
296  return (false);
297  }
298 
299  // Iterate doing the comparison, with case folding.
300  // (It is assumed that the testCmd is lower-case.)
301  for (i = 0; i < testLen; i++) {
302  char testChar = (char) cmd->buffer[cmd->args + i];
303  if (testChar >= 'A' && testChar <= 'Z')
304  testChar += 'a' - 'A';
305  if (testCmd[i] != testChar) {
306  return (false);
307  }
308  }
309 
310  // We've now at least matched the full pattern, and so
311  // regardless of ultimate match conclusion, we've we should point at what's next
312  cmd->nextarg += testLen;
313 
314  // If we tested the full string, we're done.
315  if (testLen == (cmd->length - cmd->args)) {
316  return (true);
317  }
318 
319  }
320 
321  // In our special token mode, match ANY non-separator up to the next space or end-of-buffer
322  if (tokenMode)
323  while (cmd->nextarg < cmd->length && !isArgSeparator(cmd->buffer[cmd->nextarg], embeddedSpaces))
324  cmd->nextarg++;
325 
326  // If we need to test for a word delimiter and there is more left, look at it
327  if (testForWord && (cmd->nextarg < cmd->length)) {
328 
329  // Ensure there is at least one separator following the command
330  if (!isArgSeparator(cmd->buffer[cmd->nextarg], embeddedSpaces)) {
331  return (false);
332  }
333 
334  // Skip past contiguous separators
335  for (i = cmd->nextarg; i < cmd->length; i++) {
336  if (!isArgSeparator(cmd->buffer[i], embeddedSpaces))
337  break;
338  if (tokenMode)
339  cmd->buffer[i] = '\0';
340  }
341  cmd->nextarg = i;
342 
343  }
344 
345  // We've got us a match
346  return (true);
347 
348 }

526 // Get primary sampling interval, in seconds
527 uint32_t get_oneshot_interval() {
528  uint32_t suppressionSeconds = storage()->oneshot_minutes * 60;
529 
530  // If we're debugging and want to force a call, do it now
531  if (commCallNow)
532  return 1;
533 
534  // Depending upon battery level, optionally slow down the one-shot uploader.
535  // Note that if we fail to upload, this will naturally stop the sensors from
536  // re-sampling - and thus it will slow down the entire oneshot process as a
537  // desperate way of keeping the battery level in a reasonable state
538  switch (battery_status()) {
539  // If battery is dead, only one daily update
540  case BAT_DEAD:
541  suppressionSeconds = 24 * 60 * 60;
542  break;
543  // Only on a long interval if in danger
544  case BAT_EMERGENCY:
545  suppressionSeconds = 6 * 60 * 60;
546  break;
547  // Update 30m if just a warning
548  case BAT_WARNING:
549  suppressionSeconds = 30 * 60;
550  break;
551  // Full battery
552  case BAT_FULL:
553  if (suppressionSeconds > (ONESHOT_FAST_MINUTES * 60))
554  suppressionSeconds = (ONESHOT_FAST_MINUTES * 60);
555  break;
556  case BAT_BURN:
557  case BAT_TEST:
558  suppressionSeconds = 5 * 60;
559  break;
560  // Wicked-fast mobile mode
561  case BAT_MOBILE: {
562  uint16_t period_secs = sensor_get_mobile_upload_period();
563  if (period_secs == 0)
564  suppressionSeconds = 1;
565  else
566  suppressionSeconds = period_secs;
567  break;
568  }
569  }
570 
571  return(suppressionSeconds);
572 
573 }

671  int nextmin = nextsecs/60;
672  nextsecs -= nextmin*60;
673  DEBUG_PRINTF("Next stats update (%ldm) %s by %dm%ds\n", get_service_update_interval_minutes(), fOverdue ? "is overdue" : "will begin", nextmin, nextsecs);
674 
675  }

693 // Primary comms-related poller, called from our app timer
694 void comm_poll() {
695 
696  // Exit if the basic comms package has never yet been initialized.
697  if (!commEverInitialized)
698  return;
699 
700  // If a Oneshot completion was requested on the next poll, do it.
701  if (oneshotCompletedNextPoll) {
702  oneshotCompletedNextPoll = false;
703  oneshotCompleted = true;
704  }
705 
706  // Display failures periodically, as a debugging tool
707  mtu_status_check(false);
708 
709  // If we're waiting for our first select, process it.
710  if (commWaitingForFirstSelect) {

1002  // connection is active.

1005  if (comm_send_to_service(entry, entry_length, entry_request_type)) {
1006  db_get_release();

1147  if (commCallNow)
1148  return false;
1149

1151 #ifdef COMMS_FORCE_NONBUFFERED

1154  return false;
1155 #endif
1156

1158  if ((storage()->flags & FLAG_CONFIRM_ALL) != 0) {
1159  if (fVerbose)
1160  DEBUG_PRINTF("No: confirming all transactions\n");
1161  return false;
1162  }

1165  if (get_oneshot_cell_interval() == 0) {
1166  if (fVerbose)
1167  DEBUG_PRINTF("No: not oneshot mode\n");
1168  return false;
1169  }
1170 
1171  // We will only buffer when we are deselected
1172  if (!currently_deselected) {
1173  if (fVerbose)
1174  DEBUG_PRINTF("No: not deselected\n");
1175  return false;
1176  }
1177

1179  if (comm_mode() != COMM_FONA) {
1180  if (fVerbose)
1181  DEBUG_PRINTF("No: not fona\n");

1187  if (fVerbose)
1188  DEBUG_PRINTF("No: wrong operating mode\n");
1189  return false;
1190  }
1191

1193  // we proactively assume that there will be another message, and we don't want to
1194  // buffer if we're this close to the end.
1195 #define NEXT_MESSAGE_ALLOWANCE 150
1196  if (!db_enabled() && send_buff_is_full(NEXT_MESSAGE_ALLOWANCE)) {
1197  if (fVerbose)
1198  DEBUG_PRINTF("No: full buff\n");
1199  return false;
1200  }
1201

1203  // because all the uploads will look like they're at the same date/time
1204  if (!get_current_timestamp(NULL, NULL, NULL)) {
1205  if (fVerbose)
1206  DEBUG_PRINTF("No: no timestamp\n");
1207  return false;
1208  }
1209 
1210  // If we're forcing a flush, do it
1211  if (fFlushBuffers && !send_buff_is_empty()) {
1212  if (fVerbose)
1213  DEBUG_PRINTF("No: need to flush buffers\n");
1214  return false;
1215  }
1216 
1217  // If it's time to do a transmit to the service, don't buffer it
1218  if (!WouldSuppress(&comm_last_powered_up, get_oneshot_cell_interval())) {
1219  if (fVerbose)
1220  DEBUG_PRINTF("No: time for a oneshot\n");
1221  return false;
1222  }
1223 
1224  // If it's time to do a stats request, don't buffer it
1225  if (!WouldSuppress(&lastServiceUpdateTime, get_service_update_interval_minutes()*60)) {
1226  if (fVerbose)
1227  DEBUG_PRINTF("No: time for stats\n");
1228  return false;
1229  }
1230 
1231  // Done
1232  if (fVerbose)
1233  DEBUG_PRINTF("Would be buffered!\n");
1234 
1235  return true;
1236 
1237 }

1520 // Decode a hex-encoded received message, then unmarshal and process what's inside
1521 uint16_t comm_decode_received_message(char *msg, void *ttmessage, uint8_t *buffer, uint16_t buffer_length, uint16_t *bytesDecoded) {
1522  uint8_t bin[256], *pbin;
1523  int length;
1524  char hiChar, loChar;
1525  uint8_t databyte;
1526  uint16_t status;
1527  ttproto_Telecast tmessage;
1528  ttproto_Telecast *message = (ttproto_Telecast *) ttmessage;
1529 
1530  // Skip leading whitespace and control characters, to get to the hex
1531  while (*msg != '\0' && *msg <= ' ')
1532  msg++;
1533 
1534  // Convert hex to binary
1535  for (length = 0; length < sizeof(bin); length++) {
1536  hiChar = *msg++;
1537  loChar = *msg++;
1538  if (!HexValue(hiChar, loChar, &databyte))
1539  break;
1540  bin[length] = databyte;
1541  }
1542 
1543  // Return how many we've hex-decoded
1544  if (bytesDecoded != NULL)
1545  *bytesDecoded = length;
1546 
1547  // Zero out the structure to receive the decoded data
1548  if (message == NULL)
1549  message = &tmessage;
1550  memset(message, 0, sizeof(ttproto_Telecast));
1551 
1552  // Look at the first byte of what's been received, and see if it's in the new "array" format.
1553  // It will be this way if we're relaying a message.
1554  // If not, just process it as-is under the assumption that it's a single protocol buffer
1555  pbin = bin;
1556  if (bin[0] == BUFF_FORMAT_PB_ARRAY && bin[1] == 1) {
1557 
1558  // Process the message
1559  length = bin[2];
1560  pbin = &bin[3];
1561 
1562  DEBUG_PRINTF("Received %d-byte message\n", length);
1563 
1564  } else {
1565 
1566  DEBUG_PRINTF("Received message of unknown format 0x%02x 0x%02x 0x%02x\n", bin[0], bin[1], bin[2]);
1567  return MSG_NOT_DECODED;
1568 
1569  }
1570 
1571  // Create a stream that will write to our buffer.
1572  pb_istream_t stream = pb_istream_from_buffer(pbin, length);
1573 
1574  // Decode the message
1575  status = pb_decode(&stream, ttproto_Telecast_fields, message);
1576  if (!status) {
1577  DEBUG_PRINTF("pb_decode: %s\n", PB_GET_ERROR(&stream));
1578  return MSG_NOT_DECODED;
1579  }
1580 
1581  // Copy the message to the output buffer
1582  if (message->has_message)
1583  strlcpy((char *) buffer, message->message, buffer_length);
1584  else
1585  buffer[0] = '\0';
1586 
1587  // Do various things based on device type
1588  if (!message->has_device_type) {
1589 
1590  DEBUG_PRINTF("(ignoring message from %lu)\n", message->device_id);
1591  return MSG_SAFECAST;
1592 
1593  } else {
1594 
1595  switch (message->device_type) {
1596  case ttproto_Telecast_deviceType_UNKNOWN_DEVICE_TYPE:
1597  case ttproto_Telecast_deviceType_SOLARCAST:
1598  case ttproto_Telecast_deviceType_BGEIGIE_NANO:
1599  case ttproto_Telecast_deviceType_TTGATEPING:
1600  DEBUG_PRINTF("(ignoring message from %d)\n", message->device_id);
1601  return MSG_SAFECAST;
1602 
1603  case ttproto_Telecast_deviceType_TTGATE:
1604  // If it's from ttgate and directed at us, then it's a reply to our request
1605  if (message->has_device_id) {
1606  if (message->device_id == io_get_device_address())
1607  DEBUG_PRINTF("Received TTGATE message\n");
1608  else
1609  DEBUG_PRINTF("Received TTGATE message for device %lu\n", message->device_id);
1610  } else {
1611  DEBUG_PRINTF("Received an invalidly-formatted TTGATE message\n");
1612  }
1613  return MSG_REPLY_TTGATE;
1614 
1615  case ttproto_Telecast_deviceType_TTSERVE:
1616  // If it's from ttserve and directed at us, then it's a reply to our request
1617  if (message->has_device_id && message->device_id == io_get_device_address()) {
1618  DEBUG_PRINTF("Received TTSERVE message\n");
1619  return MSG_REPLY_TTSERVE;
1620  }
1621  DEBUG_PRINTF("Received TTServe message not intended for this device\n");
1622  return MSG_TELECAST;
1623 
1624  case ttproto_Telecast_deviceType_TTAPP:
1625  DEBUG_PRINTF("Received TTAPP message\n");
1626  return MSG_TELECAST;
1627 
1628  case ttproto_Telecast_deviceType_TTNODE:
1629  DEBUG_PRINTF("Received peer message\n");
1630  return MSG_TELECAST;
1631  }
1632  }
1633 
1634  DEBUG_PRINTF("Received unknown message\n");
1635  return MSG_NOT_DECODED;
1636 
1637 }
