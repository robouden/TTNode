198 // Callback when TWI data has been read, or timeout
199 void lis_callback(ret_code_t result, twi_context_t *t) {
200 
201  // If error, flag that this I/O has been completed.
202  if (!twi_completed(t)) {
203  fInit = false; // Force reinitialization
204  stats()->errors_lis++;
205  gpio_motion_sense(MOTION_DISARM);
206  sensor_measurement_completed(t->sensor);
207  return;
208  }
209 
210  // If we don't respond to the "who am I" command, something is wrong
211  if (who[1] != WHO_AM_I) {
212  fInit = false; // Force reinitialization
213  stats()->errors_lis++;
214  DEBUG_PRINTF("LIS: Bad response %02x from WHO AM I!\n", who[1]);
215  gpio_motion_sense(MOTION_DISARM);
216  sensor_measurement_completed(t->sensor);
217  return;
218  }
219 
220  // It's ok for the poller to validate that the interrupt is cleared
221  fPollerShouldClearInterrupt = true;
222  ConsecutiveClearInterrupts = 0;
223 
224 }

226 // Measure the sensor value
227 void s_lis_measure(void *s) {
228 
229  // Exit immediately if we don't need to do any measurement
230  if (!fArmForMotionSensing) {
231  if (debug(DBG_SENSOR))
232  DEBUG_PRINTF("LIS: already armed\n");
233  sensor_measurement_completed(s);
234  return;
235  }
236 
237  // clear out registers
238  reg1[1] = reg2[1] = reg3[1] = reg4[1] = reg5[1] = reg6[1] = int1_ths[1] = int1_cfg[1] = 0;
239 
240  // Set low-power wake mode
241  // Enable all axes
242  reg1[1] |= CTRL_REG1_ZEN | CTRL_REG1_YEN | CTRL_REG1_XEN;
243  // Enable 10 hz update rate
244  reg1[1] |= CTRL_REG1_ODR1;
245  // Enable low power mode
246  reg1[1] |= CTRL_REG1_LPEN;
247  // Enable high-pass filter on interrupt 1
248  reg2[1] |= CTRL_REG2_FDS | CTRL_REG2_HPIS1;
249  // Enable INT1
250  reg3[1] |= CTRL_REG3_I1_AOI1;
251  // Disable FIFO, enable latch interrupt on INT1_SRC
252  reg5[1] |= CTRL_REG5_LIR_INT1;
253  // Set Movement Threshold to 250mg
254  int1_ths[1] = 16;
255  // Configure the things that will interrupt us
256  int1_cfg[1] |= INT1_CFG_ZHIE_ZUPE | INT1_CFG_YHIE_YUPE | INT1_CFG_XHIE_XUPE;
257 
258  static app_twi_transfer_t const transfers[] = {
259  APP_TWI_WRITE(LIS_I2C_ADDRESS, &who[0], sizeof(who[0]), APP_TWI_NO_STOP),
260  APP_TWI_READ(LIS_I2C_ADDRESS, &who[1], sizeof(who[1]), 0),
261  APP_TWI_WRITE(LIS_I2C_ADDRESS, &reg1, sizeof(reg1), 0),
262  APP_TWI_WRITE(LIS_I2C_ADDRESS, &reg2, sizeof(reg2), 0),
263  APP_TWI_WRITE(LIS_I2C_ADDRESS, &reg3, sizeof(reg3), 0),
264  APP_TWI_WRITE(LIS_I2C_ADDRESS, &reg4, sizeof(reg4), 0),
265  APP_TWI_WRITE(LIS_I2C_ADDRESS, &reg5, sizeof(reg5), 0),
266  APP_TWI_WRITE(LIS_I2C_ADDRESS, &reg6, sizeof(reg6), 0),
267  APP_TWI_WRITE(LIS_I2C_ADDRESS, &int1_ths, sizeof(int1_ths), 0),
268  APP_TWI_WRITE(LIS_I2C_ADDRESS, &int1_cfg, sizeof(int1_cfg), 0),
269  APP_TWI_WRITE(LIS_I2C_ADDRESS, &int1_src[0], sizeof(int1_src[0]), APP_TWI_NO_STOP),
270  APP_TWI_READ(LIS_I2C_ADDRESS, &int1_src[1], sizeof(int1_src[1]), 0),
271  };
272  static app_twi_transaction_t const transaction = {
273  .callback = twi_callback,
274  .p_user_data = "LIS",
275  .p_transfers = transfers,
276  .number_of_transfers = sizeof(transfers) / sizeof(transfers[0])
277  };
278  if (!twi_schedule(s, lis_callback, &transaction)) {
279  stats()->errors_lis++;
280  sensor_unconfigure(s);
281  }
282 
283 }

301 void lis_poll_callback(ret_code_t result, twi_context_t *t) {
302 
303  // The poll request is no longer in progress
304  fPollRequestInProgress = false;
305 
306  if (!twi_completed(t)) {
307  fInit = false; // Force reinitialization
308  stats()->errors_lis++;
309  sensor_measurement_completed(t->sensor);
310  return;
311  }
312 
313  if (debug(DBG_SENSOR_SUPERMAX))
314  DEBUG_PRINTF("W%02x M%d A%02x C%02x T%02x | %02x%02x%02x%02x%02x%02x | %02x%02x %02x%02x %02x%02x | %02x%02x %02x%02x %02x%02x\n",
315  who[1], gpio_motion_sense(MOTION_QUERY_PIN), status_aux[1], int1_cfg[1], int1_ths[1],
316  reg1[1], reg2[1], reg3[1], reg4[1], reg5[1], reg6[1],
317  out_adc1_h[1], out_adc1_l[1], out_adc2_h[1], out_adc2_l[1], out_adc3_h[1], out_adc3_l[1],
318  out_x_h[1], out_x_l[1], out_y_h[1], out_y_l[1], out_z_h[1], out_z_l[1]);
319 
320  // Check the INT pin, and validate that it is clear for 10 consecutive polls
321  if (gpio_motion_sense(MOTION_QUERY_PIN)) {
322  ConsecutiveClearInterrupts = 0;
323  return;
324  }
325 
326  if (ConsecutiveClearInterrupts++ < MAX_CONSECUTIVE_CLEAR)
327  return;
328 
329  fPollerShouldClearInterrupt = false;
330 
331  // We've now been clear for 10 consecutive polls. We're done.
332  if (debug(DBG_SENSOR))
333  DEBUG_PRINTF("MOTION re-armed\n");
334 
335  gpio_motion_sense(MOTION_ARM);
336 
337  // Flag that this I/O has been completed
338 #ifdef MOTIONDEBUGMAX
339  // We're debugging the sensor, so don't mark it as complete so that we stay measuring forever
340 #else
341  sensor_measurement_completed(t->sensor);
342 #endif
343 
344 }

426 // Init sensor
427 bool s_lis_init(void *s, uint16_t param) {
428 
429  // Clear things used by the poller
430  fPollerShouldClearInterrupt = false;
431  fPollRequestInProgress = false;
432 
433  // Determine whether or not we are going to reset the chip
434  if (!fInit) {
435 
436  // If we've never initialized, let's arm
437  fArmForMotionSensing = true;
438  fInit = true;
439 
440  } else {
441 
442  // If we sense that motion has occurred, let's reset the interrupt so that
443  // future motion will be detected. Otherwise, we'll just do nothing
444  // because we're already armed and the pin doesn't need to be cleared
445  if (gpio_motion_sense(MOTION_QUERY_PIN))
446  fArmForMotionSensing = true;
