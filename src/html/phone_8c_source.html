38 // Device states

41 // Command buffer
42 static cmdbuf_t fromPhone;

44 // Process a "complete" command buffer, and if idle parse it to determine its first state
45 void phone_complete() {
46 
47  // If we're not in an idle state, let's not process commands for it
48  if (fromPhone.state != COMM_STATE_IDLE)
49  return;
50 
51  for (;;) {
52 
53  // If it begins with a slash, transmit this oon the wire as a pb-formatted "text message"
54  if (comm_cmdbuf_this_arg_is(&fromPhone, "/*")) {
55  // Skip to the actual text to be transmitted
56  comm_cmdbuf_next_arg(&fromPhone);
57  fromPhone.state = CMD_STATE_XMIT_PHONE_TEXT;
58  break;
59  }
60 
61  // Process our hard-wired commands
62 
63 #ifdef LORA
64  // Commands to be passed-through to the LPWAN chip - used for testing
65  if (comm_cmdbuf_this_arg_is(&fromPhone, "sys") || comm_cmdbuf_this_arg_is(&fromPhone, "mac") || comm_cmdbuf_this_arg_is(&fromPhone, "radio")) {
66  // Convert to lowercase because the LPWAN chip requires this
67  int i;
68  for (i = 0; i < fromPhone.length; i++)
69  if (fromPhone.buffer[i] >= 'A' && fromPhone.buffer[i] <= 'Z')
70  fromPhone.buffer[i] += 'a' - 'A';
71  // Enter command mode. This may fail the first time because it's busy,
72  // but then after the next timeout we will be in that state.
73  lora_enter_command_mode();
74  // Send to the LPWAN chip, even if it may fail because of the state we're in
75  lora_send((char *)&fromPhone.buffer[0]);
76  comm_cmdbuf_set_state(&fromPhone, COMM_STATE_IDLE);
77  break;
78  }
79 #endif // LORA
80 
81 #ifdef FONA
82  // Commands to be passed-through to the chip - used for testing
83  if (comm_cmdbuf_this_arg_is(&fromPhone, "at+*")) {
84  // Send to the chip, even if it may fail because of the state we're in
85  fona_send((char *)&fromPhone.buffer[0]);
86  comm_cmdbuf_set_state(&fromPhone, COMM_STATE_IDLE);
87  break;
88  }
89 #endif
90 
91  // Turn the display on or off
92 #ifdef SSD
93  if (comm_cmdbuf_this_arg_is(&fromPhone, "ssd")) {
94  comm_cmdbuf_next_arg(&fromPhone);
95  if (comm_cmdbuf_this_arg_is(&fromPhone,"on")) {
96  ssd1306_init();
97  } else if (comm_cmdbuf_this_arg_is(&fromPhone,"off")) {
98  ssd1306_term();
99  } else {
100  if (!ssd1306_active())
101  DEBUG_PRINTF("ssd <on/off>\n");
102  else
103  ssd1306_reset_display();
104  }
105  comm_cmdbuf_set_state(&fromPhone, COMM_STATE_IDLE);
106  break;
107  }
108 #endif
109 
110  // Battery level request
111  if (comm_cmdbuf_this_arg_is(&fromPhone, "bat")) {
112  if (sensor_group_schedule_now("g-basics")) {
113  DEBUG_PRINTF("Starting g-basics with max debugging enabled.\n");
114  debug_flags_set(DBG_SENSOR|DBG_SENSOR_MAX);
115  }
116  comm_cmdbuf_set_state(&fromPhone, COMM_STATE_IDLE);
117  break;
118  }
119 
120  // CPM measurement request
121 #ifdef GEIGERX
122  if (comm_cmdbuf_this_arg_is(&fromPhone, "rad") || comm_cmdbuf_this_arg_is(&fromPhone, "cpm") || comm_cmdbuf_this_arg_is(&fromPhone, "geiger")) {
123  if (sensor_group_schedule_now("g-geiger"))
124  DEBUG_PRINTF("Starting g-geiger\n");
125  comm_cmdbuf_set_state(&fromPhone, COMM_STATE_IDLE);
126  break;
127  }
128 #endif
129 
130  // Toggle a mode that makes it appear that no sensors are enabled
131  if (comm_cmdbuf_this_arg_is(&fromPhone, "dead")) {
132  uint16_t op_mode = sensor_op_mode();
133  if (op_mode != OPMODE_TEST_DEAD)
134  op_mode = OPMODE_TEST_DEAD;
135  else
136  op_mode = OPMODE_NORMAL;
137  sensor_set_op_mode(op_mode);
138  DEBUG_PRINTF("Sensor scheduling now %s\n", op_mode == OPMODE_NORMAL ? "ON" : "OFF");
139  comm_cmdbuf_set_state(&fromPhone, COMM_STATE_IDLE);
140  break;
141  }
142 
143  // Force sensor scheduling now
144  if (comm_cmdbuf_this_arg_is(&fromPhone, "q") || comm_cmdbuf_this_arg_is(&fromPhone, "sample") || comm_cmdbuf_this_arg_is(&fromPhone, "measure")) {
145  sensor_schedule_now();
146  comm_cmdbuf_set_state(&fromPhone, COMM_STATE_IDLE);
147  break;
148  }
149 
150  // Flip the display
151  if (comm_cmdbuf_this_arg_is(&fromPhone, "flip")) {
152  storage()->flags ^= FLAG_FLIP;
153  storage_save(true);
154  comm_cmdbuf_set_state(&fromPhone, COMM_STATE_IDLE);
155 #ifdef SSD
156  ssd1306_term();
157  ssd1306_init();
158 #endif
159  DEBUG_PRINTF("Display flipped.\n");
160  break;
161  }
162 
163  // Force cellular to test failover behavior
164  if (comm_cmdbuf_this_arg_is(&fromPhone, "fail")) {
165  comm_force_cell();
166  comm_cmdbuf_set_state(&fromPhone, COMM_STATE_IDLE);
167  break;
168  }
169 
170  // Force cellular
171 #if defined(FONA)
172  if (comm_cmdbuf_this_arg_is(&fromPhone, "fona")) {
173  comm_request_mode_on_reselect(COMM_FONA);
174  comm_cmdbuf_set_state(&fromPhone, COMM_STATE_IDLE);
175  break;
176  }
177 #endif
178 
179  // Force lora
180 #if defined(LORA)
181  if (comm_cmdbuf_this_arg_is(&fromPhone, "lora")) {
182  comm_request_mode_on_reselect(COMM_LORA);
183  comm_cmdbuf_set_state(&fromPhone, COMM_STATE_IDLE);
184  break;
185  }
186 #endif
187 
188  // Cancel a pending WAN request
189  if (comm_cmdbuf_this_arg_is(&fromPhone, "none")) {
190  comm_request_mode_on_reselect(COMM_NONE);
191  comm_cmdbuf_set_state(&fromPhone, COMM_STATE_IDLE);
192  break;
193  }
194 
195  // Comms Sensor State request
196  if (comm_cmdbuf_this_arg_is(&fromPhone, "ccc")) {
197  comm_show_state();
198  comm_cmdbuf_set_state(&fromPhone, COMM_STATE_IDLE);
199  break;
200  }
201 
202  // Show Sensor State request
203  if (comm_cmdbuf_this_arg_is(&fromPhone, "sss")) {
204  sensor_show_state(true);
205  comm_cmdbuf_set_state(&fromPhone, COMM_STATE_IDLE);
206  break;
207  }
208 
209  // Cause GPS to have a forced update, simulating motion
210  if (comm_cmdbuf_this_arg_is(&fromPhone, "gupdate") || comm_cmdbuf_this_arg_is(&fromPhone, "grefresh")) {
211  storage()->gps_latitude = storage()->gps_longitude = 0.0;
212  comm_gps_update();
213  comm_cmdbuf_set_state(&fromPhone, COMM_STATE_IDLE);
214  break;
215  }
216 
217  // GPS "set to fake data" request
218  if (comm_cmdbuf_this_arg_is(&fromPhone, "gfake")) {
219  char buffer[256];
220  storage_set_gps_params_as_string("1/1/1");
221  storage_save(true);
222  storage_get_gps_params_as_string(buffer, sizeof(buffer));
223  DEBUG_PRINTF("Now %s\n", buffer);
224  comm_cmdbuf_set_state(&fromPhone, COMM_STATE_IDLE);
225  break;
226  }
227 
228  // GPS abort and go to "last known Good GPS"
229  if (comm_cmdbuf_this_arg_is(&fromPhone, "glkg") || comm_cmdbuf_this_arg_is(&fromPhone, "gabort")) {
230  comm_gps_abort();
231  comm_cmdbuf_set_state(&fromPhone, COMM_STATE_IDLE);
232  break;
233  }
234 
235  // GPS request
236  if (comm_cmdbuf_this_arg_is(&fromPhone, "gps")) {
237  comm_cmdbuf_next_arg(&fromPhone);
238  switch (fromPhone.buffer[fromPhone.args]) {
239  case '\0': {
240  float lat, lon, alt;
241  uint16_t status = comm_gps_get_value(&lat, &lon, &alt);
242  if (status == GPS_LOCATION_FULL)
243  DEBUG_PRINTF("%.3f/%.3f/%.3f\n", lat, lon, alt);
244  else if (status == GPS_LOCATION_PARTIAL)
245  DEBUG_PRINTF("%.3f/%.3f\n", lat, lon);
246  else if (status == GPS_NO_LOCATION)
247  DEBUG_PRINTF("No location.\n");
248  else if (status == GPS_LOCATION_ABORTED)
249  DEBUG_PRINTF("Location aborted.\n");
250  else
251  DEBUG_PRINTF("No data.\n");
252  break;
253  }
254  }
255  comm_cmdbuf_set_state(&fromPhone, COMM_STATE_IDLE);
256  break;
257  }
258 
259  // GPIO test request
260  if (comm_cmdbuf_this_arg_is(&fromPhone, "gpio")) {
261  comm_cmdbuf_next_arg(&fromPhone);
262  if (fromPhone.buffer[fromPhone.args] != '\0') {
263  uint16_t num = atoi((char *)&fromPhone.buffer[fromPhone.args]);
264  uint16_t pin = num/10;
265  bool fOn = ((num & 0x01) != 0);
266  gpio_power_set(pin, fOn);
267  }
268  comm_cmdbuf_set_state(&fromPhone, COMM_STATE_IDLE);
269  break;
270  }
271 
272  // MTU test request
273  if (comm_cmdbuf_this_arg_is(&fromPhone, "mtu")) {
274  comm_cmdbuf_next_arg(&fromPhone);
275  if (fromPhone.buffer[fromPhone.args] != '\0') {
276  send_mtu_test(atoi((char *)&fromPhone.buffer[fromPhone.args]));
277  debug_flags_set(DBG_RX|DBG_TX);
278  }
279  comm_cmdbuf_set_state(&fromPhone, COMM_STATE_IDLE);
280  break;
281  }
282 
283  // Temperature/Humidity request
284 #if defined(TWIHIH6130) || defined(TWIBME0)
285  if (comm_cmdbuf_this_arg_is(&fromPhone, "temp") || comm_cmdbuf_this_arg_is(&fromPhone, "env")) {
286  float envTempC, envHumRH;
287  float envPress = 0.0;
288 #ifdef TWIHIH6130
289  s_hih6130_get_value(&envTempC, &envHumRH);
290 #endif
291 #ifdef TWIBME0
292  s_bme280_0_get_value(&envTempC, &envHumRH, &envPress);
293 #endif
294  DEBUG_PRINTF("%f degC, %f pctRH, %f Pa\n", envTempC, envHumRH, envPress);
295  comm_cmdbuf_set_state(&fromPhone, COMM_STATE_IDLE);
296  break;
297  }
298 #endif
299 
300  // Request state
301  if (comm_cmdbuf_this_arg_is(&fromPhone, "state")) {
302  comm_request_state();
303  comm_cmdbuf_set_state(&fromPhone, COMM_STATE_IDLE);
304  break;
305  }
306 
307  // Request statistics
308  if (comm_cmdbuf_this_arg_is(&fromPhone, "comms") || comm_cmdbuf_this_arg_is(&fromPhone, "comm")) {
309  stats_status_check(true);
310  comm_cmdbuf_set_state(&fromPhone, COMM_STATE_IDLE);
311  break;
312  }
313 
314  // Request statistics
315  if (comm_cmdbuf_this_arg_is(&fromPhone, "stats")) {
316  comm_initiate_service_update(false);
317  comm_cmdbuf_set_state(&fromPhone, COMM_STATE_IDLE);
318  break;
319  }
320 
321  // Request statistics
322  if (comm_cmdbuf_this_arg_is(&fromPhone, "buff")) {
323  comm_would_be_buffered(true);
324  comm_cmdbuf_set_state(&fromPhone, COMM_STATE_IDLE);
325  break;
326  }
327 
328  // Request full statistics
329  if (comm_cmdbuf_this_arg_is(&fromPhone, "hello")) {
330  comm_initiate_service_update(true);
331  comm_cmdbuf_set_state(&fromPhone, COMM_STATE_IDLE);
332  break;
333  }
334 
335  // Soft reset request
336  if (comm_cmdbuf_this_arg_is(&fromPhone, "reset")) {
337  comm_reset(true);
338  comm_cmdbuf_set_state(&fromPhone, COMM_STATE_IDLE);
339  break;
340  }
341 
342  // Force us to drop everything power hungry and optimize power
343  if (comm_cmdbuf_this_arg_is(&fromPhone, "drop")) {
344  io_force_optimize_power();
345  comm_cmdbuf_set_state(&fromPhone, COMM_STATE_IDLE);
346  break;
347  }
348 
349  // Turn off indicators, usually when measuring power
350  if (comm_cmdbuf_this_arg_is(&fromPhone, "ind")) {
351  gpio_indicators_off();
352  comm_cmdbuf_set_state(&fromPhone, COMM_STATE_IDLE);
353  break;
354  }
355 
356  // Force us to drop gps
357  if (comm_cmdbuf_this_arg_is(&fromPhone, "nogps")) {
358 #ifdef TWIUBLOXM8
359  s_gps_shutdown();
360 #endif
361 #ifdef FONAGPS
362  fona_gps_shutdown();
363 #endif
364 #ifdef UGPS
365  s_ugps_shutdown();
366 #endif
367  comm_cmdbuf_set_state(&fromPhone, COMM_STATE_IDLE);
368  break;
369  }
370 
371  // Request to enable verbose LPWAN rx/tx logging
372  if (comm_cmdbuf_this_arg_is(&fromPhone, "d")) {
373  char flags[40];
374  strcpy(flags, "");
375  if (debug(DBG_RX) && debug(DBG_TX))
376  strcat(flags, "C ");
377  else if (!debug(DBG_RX) && !debug(DBG_TX))
378  strcat(flags, "c ");
379  else {
380  strcat(flags, debug(DBG_RX) ? "RX " : "rx ");
381  strcat(flags, debug(DBG_TX) ? "TX " : "tx ");
382  }
383  strcat(flags, debug(DBG_COMM_MAX) ? "CX " : "cx ");
384  strcat(flags, debug(DBG_SENSOR) ? "S " : "s ");
385  strcat(flags, debug(DBG_SENSOR_MAX) ? "SX " : "sx ");
386  strcat(flags, debug(DBG_SENSOR_SUPERMAX) ? "SXX " : "sxx ");
387  strcat(flags, debug(DBG_SENSOR_SUPERDUPERMAX) ? "SXXX " : "sxxx ");
388  strcat(flags, debug(DBG_SENSOR_POLL) ? "SP " : "sp ");
389  strcat(flags, debug(DBG_GPS_MAX) ? "GX " : "gx ");
390  strcat(flags, debug(DBG_AIR) ? "A " : "a ");
391  strcat(flags, debug(DBG_BT) ? "B " : "b ");
392  DEBUG_PRINTF("DEBUG: %s\n", flags);
393  comm_cmdbuf_set_state(&fromPhone, COMM_STATE_IDLE);
394  break;
395  }
396  if (comm_cmdbuf_this_arg_is(&fromPhone, "0")) {
397  debug_flags_set(DBG_NONE);
398  DEBUG_PRINTF("ALL debug OFF\n");
399  comm_cmdbuf_set_state(&fromPhone, COMM_STATE_IDLE);
400  break;
401  }
402  if (comm_cmdbuf_this_arg_is(&fromPhone, "1")) {
403  debug_flags_set(DBG_COMMON);
404  DEBUG_PRINTF("Common debug ON\n");
405  comm_cmdbuf_set_state(&fromPhone, COMM_STATE_IDLE);
406  break;
407  }
408  if (comm_cmdbuf_this_arg_is(&fromPhone, "x")) {
409  debug_flags_set(DBG_COMMON|DBG_GPS_MAX|DBG_SENSOR_MAX|DBG_COMM_MAX);
410  DEBUG_PRINTF("ALL debug ON\n");
411  comm_cmdbuf_set_state(&fromPhone, COMM_STATE_IDLE);
412  break;
413  }
414  if (comm_cmdbuf_this_arg_is(&fromPhone, "rx")) {
415  DEBUG_PRINTF("RX toggled to %s\n", debug_flag_toggle(DBG_RX) ? "ON" : "OFF");
416  comm_cmdbuf_set_state(&fromPhone, COMM_STATE_IDLE);
417  break;
418  }
419  if (comm_cmdbuf_this_arg_is(&fromPhone, "tx")) {
420  DEBUG_PRINTF("TX toggled to %s\n", debug_flag_toggle(DBG_TX) ? "ON" : "OFF");
421  comm_cmdbuf_set_state(&fromPhone, COMM_STATE_IDLE);
422  break;
423  }
424  if (comm_cmdbuf_this_arg_is(&fromPhone, "c")) {
425  if (debug(DBG_RX) != debug(DBG_TX))
426  debug_flags_set(DBG_RX|DBG_TX);
427  DEBUG_PRINTF("COMM toggled to %s\n", debug_flag_toggle(DBG_RX|DBG_TX) ? "ON" : "OFF");
428  comm_cmdbuf_set_state(&fromPhone, COMM_STATE_IDLE);
429  break;
430  }
431  if (comm_cmdbuf_this_arg_is(&fromPhone, "cx")) {
432  DEBUG_PRINTF("COMMMAX toggled to %s\n", debug_flag_toggle(DBG_COMM_MAX) ? "ON" : "OFF");
433  comm_cmdbuf_set_state(&fromPhone, COMM_STATE_IDLE);
434  break;
435  }
436  if (comm_cmdbuf_this_arg_is(&fromPhone, "a")) {
437  DEBUG_PRINTF("AIR toggled to %s\n", debug_flag_toggle(DBG_AIR) ? "ON" : "OFF");
438  comm_cmdbuf_set_state(&fromPhone, COMM_STATE_IDLE);
439  break;
440  }
441  if (comm_cmdbuf_this_arg_is(&fromPhone, "b")) {
442  DEBUG_PRINTF("BT debug toggled to %s\n", debug_flag_toggle(DBG_BT) ? "ON" : "OFF");
443  comm_cmdbuf_set_state(&fromPhone, COMM_STATE_IDLE);
444  break;
445  }
446  if (comm_cmdbuf_this_arg_is(&fromPhone, "s")) {
447  DEBUG_PRINTF("SENSOR toggled to %s\n", debug_flag_toggle(DBG_SENSOR) ? "ON" : "OFF");
448  comm_cmdbuf_set_state(&fromPhone, COMM_STATE_IDLE);
449  break;
450  }

454  break;
455  }

458  comm_cmdbuf_set_state(&fromPhone, COMM_STATE_IDLE);
459  break;
460  }
461  if (comm_cmdbuf_this_arg_is(&fromPhone, "sxxx")) {
462  DEBUG_PRINTF("SENSORSUPERDUPERMAX toggled to %s\n", debug_flag_toggle(DBG_SENSOR_SUPERDUPERMAX) ? "ON" : "OFF");
463  comm_cmdbuf_set_state(&fromPhone, COMM_STATE_IDLE);
464  break;
465  }

468  comm_cmdbuf_set_state(&fromPhone, COMM_STATE_IDLE);
469  break;
470  }
471  if (comm_cmdbuf_this_arg_is(&fromPhone, "g")) {
472  DEBUG_PRINTF("GPSMAX toggled to %s\n", debug_flag_toggle(DBG_GPS_MAX) ? "ON" : "OFF");
473  comm_cmdbuf_set_state(&fromPhone, COMM_STATE_IDLE);
474  break;
475  }
476  if (comm_cmdbuf_this_arg_is(&fromPhone, "gx")) {
477  DEBUG_PRINTF("GPSMAX toggled to %s\n", debug_flag_toggle(DBG_GPS_MAX) ? "ON" : "OFF");
478  comm_cmdbuf_set_state(&fromPhone, COMM_STATE_IDLE);
479  break;
480  }
481 
482  // Get version
483  if (comm_cmdbuf_this_arg_is(&fromPhone, "ver")) {
484  DEBUG_PRINTF("%s\n", app_version());
485  comm_cmdbuf_set_state(&fromPhone, COMM_STATE_IDLE);
486  break;
487  }
488 
489  // Set test mode
490  if (comm_cmdbuf_this_arg_is(&fromPhone, "test") || comm_cmdbuf_this_arg_is(&fromPhone, "t")) {

517  }

538 
539  // Set mobile mode
540  if (comm_cmdbuf_this_arg_is(&fromPhone, "mobile")) {
541  comm_cmdbuf_next_arg(&fromPhone);

835 void phone_process() {
836 
837  // If it's not complete, just exit.
838  if (!fromPhone.complete)
839  return;
840 
841  // If we're idle, set the state based on buffer contents
842  if (fromPhone.state == COMM_STATE_IDLE)
843  phone_complete();
844 
845  switch (fromPhone.state) {
846 
847  case CMD_STATE_XMIT_PHONE_TEXT: {
848  uint16_t status;
849  uint8_t buffer[CMD_MAX_LINELENGTH];
850 
851  // Allocate space on the stack to store the message data.
852  ttproto_Telecast message = ttproto_Telecast_init_zero;
853 
854  /* Create a stream that will write to our buffer. */
855  pb_ostream_t stream = pb_ostream_from_buffer(buffer, sizeof(buffer));
856 
857  /* Build the message */
858  message.has_device_type = true;
859  message.device_type = ttproto_Telecast_deviceType_TTAPP;
860 
861  message.has_message = true;
862  strlcpy(message.message, (char *) &fromPhone.buffer[0], fromPhone.length);
863 
864  message.has_device_id = true;
865  message.device_id = io_get_device_address();
866 
867  // encode it and transmit it to TTSERVE
868  status = pb_encode(&stream, ttproto_Telecast_fields, &message);
869  if (!status)
870  DEBUG_PRINTF("pb_encode: %s\n", PB_GET_ERROR(&stream));
871  else
872  send_to_service(buffer, stream.bytes_written, REPLY_TTSERVE, SEND_1);
873 
874  comm_cmdbuf_set_state(&fromPhone, COMM_STATE_IDLE);
875  break;
876  }
