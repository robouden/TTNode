33 // HIH6130 7-Bit I2C Address

51 // Callback when TWI data has been read
52 void temp_callback(ret_code_t result, twi_context_t *t) {
53  uint8_t Hum_H, Hum_L, Temp_H, Temp_L, status;
54  uint16_t Hum_X, Temp_X;
55  float HumRH, TempC;
56 
57  if (!twi_completed(t)) {
58  sensor_measurement_completed(t->sensor);
59  return;
60  }
61 
62  // Read the temperature
63  status = (ioTemp.buffer[0] >> 6) & 0x03;
64  UNUSED_VARIABLE(status);
65  Hum_H = ioTemp.buffer[0] & 0x3f;
66  Hum_L = ioTemp.buffer[1];
67  Temp_H = ioTemp.buffer[2];
68  Temp_L = ioTemp.buffer[3];
69  Hum_X = (((uint16_t )Hum_H) << 8) | Hum_L;
70  Temp_X = ((((uint16_t)Temp_H) << 8) | Temp_L) >> 2;
71  if (Hum_X != 0 && Temp_X != 0) {
72  HumRH = (float) Hum_X * 6.10e-3;
73  TempC = (float) Temp_X * 1.007e-2 - 40.0;
74  // Update the temp at the cmd level
75  ioTemp.envTempC = TempC;
76  ioTemp.envHumRH = HumRH;
77  ioTemp.envReported = true;
78  }
79 
80  // Mark the I/O as being completed
81  sensor_measurement_completed(t->sensor);
82 }

89 // Measure temp
90 void s_hih6130_measure(void *s) {
91  memset(ioTemp.buffer, 0, sizeof(ioTemp.buffer));
92  static app_twi_transfer_t const transfers[] = {
93  APP_TWI_READ(HIH6130_ADDRESS, &ioTemp.buffer[0], HIH6130_DATA_LEN, 0)
94  };
95  static app_twi_transaction_t const transaction = {
96  .callback = twi_callback,
97  .p_user_data = "HIH",
98  .p_transfers = transfers,
99  .number_of_transfers = sizeof(transfers) / sizeof(transfers[0])
100  };
101  if (!twi_schedule(s, temp_callback, &transaction))
102  sensor_unconfigure(s);
103 }
