5 // Support for BT in its Controller role
6 // Minimal procedures that are necessary to kick a characteristic's CCCD into
7 // such a state wherein it generates HVX write events.
8 // CCCD is the Client Characteristic Configuration Descriptor, which configures
9 // the GATT client to tell the GATT server what packets should be sent to it.

19 // Must be a mask of continuous zeroes, followed by continuous sequence of ones: 000...111
20 #define TX_BUFFER_MASK 0x07
21 // Size of send buffer, which is 1 higher than the mask.
22 #define TX_BUFFER_SIZE (TX_BUFFER_MASK + 1)
23 // Length of the write message for CCCD
24 #define WRITE_MESSAGE_LENGTH BLE_CCCD_VALUE_LEN

55 // Function for passing any pending request from the buffer to the stack.
56 void cccd_tx_buffer_process(void) {
57 
58  if (m_tx_index != m_tx_insert_index) {
59  uint32_t err_code;
60 
61  if (m_tx_buffer[m_tx_index].type == READ_REQ) {
62  err_code = sd_ble_gattc_read(m_tx_buffer[m_tx_index].conn_handle,
63  m_tx_buffer[m_tx_index].req.read_handle,
64  0);
65  } else {
66  err_code = sd_ble_gattc_write(m_tx_buffer[m_tx_index].conn_handle,
67  &m_tx_buffer[m_tx_index].req.write_req.gattc_params);
68  }
69  if (err_code == NRF_SUCCESS) {
70  DEBUG_PRINTF("SD Read/Write API returns Success.\n");
71  m_tx_index++;
72  m_tx_index &= TX_BUFFER_MASK;
73  } else {
74  DEBUG_PRINTF("SD Read/Write API returns error. This message sending will be "
75  "attempted again..\n");
76  }
77  }
78 }

80 // Function for creating a message to be written to the CCCD
81 uint32_t cccd_configure(uint16_t conn_handle, uint16_t handle_cccd, bool enable) {
82 
83  DEBUG_PRINTF("Configuring CCCD. CCCD Handle = %d, Connection Handle = %d\n",
84  handle_cccd, conn_handle);
85 
86  tx_message_t *p_msg;
87  uint16_t cccd_val = enable ? BLE_GATT_HVX_NOTIFICATION : 0;
88 
89  p_msg = &m_tx_buffer[m_tx_insert_index++];
90  m_tx_insert_index &= TX_BUFFER_MASK;
91 
92  p_msg->req.write_req.gattc_params.handle = handle_cccd;
93  p_msg->req.write_req.gattc_params.len = WRITE_MESSAGE_LENGTH;
94  p_msg->req.write_req.gattc_params.p_value = p_msg->req.write_req.gattc_value;
95  p_msg->req.write_req.gattc_params.offset = 0;
96  p_msg->req.write_req.gattc_params.write_op = BLE_GATT_OP_WRITE_REQ;
97  p_msg->req.write_req.gattc_value[0] = LSB_16(cccd_val);
98  p_msg->req.write_req.gattc_value[1] = MSB_16(cccd_val);
99  p_msg->conn_handle = conn_handle;
100  p_msg->type = WRITE_REQ;
101 
102  cccd_tx_buffer_process();
103  return NRF_SUCCESS;
104 
105 }
