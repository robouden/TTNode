37 // Process a received message from the service
38 void recv_message_from_service(char *message) {
39  DEBUG_PRINTF("RECEIVED: %s\n", message);
40  if (memcmp(message, CMD_CFGDEV, strlen(CMD_CFGDEV)) == 0) {
41  storage_set_device_params_as_string(&message[strlen(CMD_CFGDEV)]);
42  } else if (memcmp(message, CMD_CFGGPS, strlen(CMD_CFGGPS)) == 0) {
43  storage_set_gps_params_as_string(&message[strlen(CMD_CFGGPS)]);
44  } else if (memcmp(message, CMD_CFGNET, strlen(CMD_CFGNET)) == 0) {
45  storage_set_service_params_as_string(&message[strlen(CMD_CFGNET)]);
46  } else if (memcmp(message, CMD_CFGTTN, strlen(CMD_CFGTTN)) == 0) {
47  storage_set_ttn_params_as_string(&message[strlen(CMD_CFGTTN)]);
48  } else if (memcmp(message, CMD_CFGLAB, strlen(CMD_CFGLAB)) == 0) {
49  storage_set_device_label_as_string(&message[strlen(CMD_CFGLAB)]);
50  } else if (memcmp(message, CMD_CFGSEN, strlen(CMD_CFGSEN)) == 0) {
51  storage_set_sensor_params_as_string(&message[strlen(CMD_CFGSEN)]);
52  } else if (memcmp(message, CMD_CFGDFU, strlen(CMD_CFGDFU)) == 0) {
53  storage_set_dfu_state_as_string(&message[strlen(CMD_CFGDFU)]);
54  } else if (memcmp(message, CMD_DFU, strlen(CMD_DFU)) == 0) {
55 #if !defined(DFU) || !defined(FONA)
56  DEBUG_PRINTF("DFU Requested, but firmware is not configured for DFU\n");
57  return;
58 #else
59  if (memcmp(message, CMD_DFUWITHCFG, strlen(CMD_DFUWITHCFG)) == 0)
60  storage_set_dfu_state_as_string(&message[strlen(CMD_DFUWITHCFG)]);
61  DEBUG_PRINTF("Initiating DFU of '%s'.\n", storage()->dfu_filename);
62  storage()->dfu_status = DFU_PENDING;
63 #endif
64  } else if (memcmp(message, CMD_BURN, strlen(CMD_BURN)) == 0) {
65  uint32_t duration = 60L; // Default to 1 hour
66  if (memcmp(message, CMD_BURNWITHMINS, strlen(CMD_BURNWITHMINS)) == 0)
67  duration = atol(&message[strlen(CMD_BURNWITHMINS)]);
68  DEBUG_PRINTF("Initiating temporary BURN mode for %ld minutes.\n", duration);
69  sensor_set_temporary_op_mode(OPMODE_TEST_BURN, duration * 60L);
70  return;
71  } else if (strcmp(message, CMD_RESTART) == 0) {
72  io_request_restart();
73  return;
74  } else if (strcmp(message, CMD_REBOOT) == 0) {
75  io_request_restart();
76  return;
77  } else if (strcmp(message, CMD_HELLO) == 0) {
78  comm_initiate_service_update(true);
79  return;
80  } else if (strcmp(message, CMD_DOWN) == 0) {
81  comm_force_cell();
82  return;
83  } else {
84  DEBUG_PRINTF("TTSERVE: %s\n", (char *) message);
85  return;
86  }
87  storage_save(true);
88  io_request_restart();
89 }
